3   import io.oasp.gastronomy.restaurant.general.common.api.datatype.Money;
4   import io.oasp.gastronomy.restaurant.salesmanagement.common.api.datatype.PaymentStatus;
5   import io.oasp.gastronomy.restaurant.salesmanagement.logic.api.to.BillEto;
6   import io.oasp.gastronomy.restaurant.salesmanagement.logic.api.to.OrderPositionEto;

8   
9   import java.util.List;
10  
11  /**
12   * Interface of UcManageBill to centralize documentation and signatures of methods.
13   *
14   * @author mvielsac
15   */
16  public interface UcManageBill {
17    /**
18     * Create a {@link BillEto} and save it in the database by linking the billed {@link OrderPositionEto order positions}
19     * with it. A bill also consists of the {@link Money total amount} as well as the {@link Money tip}, the waiter could
20     * fill in. The tip is at default 0.0.
21     *
22     * @param orderPositions is the {@link List} of {@link OrderPositionEto}, the billing process places on.
23     * @param tip is the waiter's {@link Money}
24     *
25     * @return the created {@link BillEto}
26     */
27    BillEto createBill(List<OrderPositionEto> orderPositions, Money tip);
28  
29    /**
30     * This method deletes a {@link BillEto} from the database by its given id.
31     *
32     * @param billId the ID of the {@link BillEto} that has to be deleted.
33     */
34    void deleteBill(Long billId);
35  
36    /**
37     * This method provides the payment process. It provides the cash-only functionality without any additional
38     * {@link PaymentData}.
39     *
40     * @param bill is the identifier of the {@link BillEto} on which the payment process takes place.
41     * @return the {@link PaymentStatus status code} of the payment process.
42     */
43    PaymentStatus doPayment(BillEto bill);
44  
45    /**
46     * This method provides the payment process.
47     *
48     * @param paymentDataDebitor is the {@link PaymentData} which holds special data for the payment process.
49     * @param bill is the identifier of the {@link BillEto} on which the payment process takes place.
50     * @return the {@link PaymentStatus status code} of the payment process.
51     */
52    PaymentStatus doPayment(BillEto bill, PaymentData paymentDataDebitor);
53  }
