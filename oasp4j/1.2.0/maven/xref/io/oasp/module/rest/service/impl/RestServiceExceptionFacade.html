36  /**
37   * This is an implementation of {@link ExceptionMapper} that acts as generic exception facade for REST services.
38   *
39   * the exception handling class for all upcoming exceptions thrown at REST requests. Each type of possible thrown
40   * exception will be fetched within the method "toResponse".
41   *
42   * @author agreul
43   */
44  @Provider
45  public class RestServiceExceptionFacade implements ExceptionMapper<Throwable> {

48    public static final String KEY_MESSAGE = "message";

52  
53    /** JSON key for {@link NlsRuntimeException#getCode() error code}. */
54    public static final String KEY_CODE = "code";

79     * Registers a {@link Class} as a top-level security {@link Throwable exception}. Instances of this class and all its
80     * subclasses will be handled as security errors. Therefore an according HTTP error code is used and no further
81     * details about the exception is send to the client to prevent <a
82     * href="https://www.owasp.org/index.php/Top_10_2013-A6-Sensitive_Data_Exposure">sensitive data exposure</a>.
83     *
84     * @param securityException is the {@link Class} reflecting the security error.
85     */
86    protected void registerToplevelSecurityException(Class<? extends Throwable> securityException) {
87  
88      this.securityExceptions.add(securityException);
89    }

93     * override it to add additional or other classes.

95    protected void registerToplevelSecurityExceptions() {
96  
97      this.securityExceptions.add(SecurityException.class);
98      this.securityExceptions.add(SecurityErrorUserException.class);
99      registerToplevelSecurityExceptions("org.springframework.security.access.AccessDeniedException");
100     registerToplevelSecurityExceptions("org.springframework.security.authentication.AuthenticationServiceException");
101     registerToplevelSecurityExceptions("org.springframework.security.authentication.AuthenticationCredentialsNotFoundException");
102     registerToplevelSecurityExceptions("org.springframework.security.authentication.BadCredentialsException");
103     registerToplevelSecurityExceptions("org.springframework.security.authentication.AccountExpiredException");
104   }

106   /**
107    * @param className the className to be registered
108    */
109   protected void registerToplevelSecurityExceptions(String className) {
110 
111     try {
112       @SuppressWarnings("unchecked")
113       Class<? extends Throwable> securityException = (Class<? extends Throwable>) Class.forName(className);
114       registerToplevelSecurityException(securityException);
115     } catch (ClassNotFoundException e) {
116       LOG.info(
117           "Spring security was not found on classpath. Spring security exceptions will not be handled as such by {}",
118           getClass().getSimpleName());
119     }
120   }

122   /**
123    * {@inheritDoc}
124    */
125   @Override
126   public Response toResponse(Throwable exception) {
127 
128     // business exceptions
129     if (exception instanceof WebApplicationException) {
130       return createResponse((WebApplicationException) exception);
131     } else if (exception instanceof ValidationException) {
132       Throwable t = exception;
133       Map<String, List<String>> errorsMap = null;
134       if (exception instanceof ConstraintViolationException) {
135         ConstraintViolationException constraintViolationException = (ConstraintViolationException) exception;
136         Set<ConstraintViolation<?>> violations = constraintViolationException.getConstraintViolations();
137         errorsMap = new HashMap<>();
138 
139         for (ConstraintViolation<?> violation : violations) {
140           Iterator<Node> it = violation.getPropertyPath().iterator();
141           String fieldName = null;
142 
143           // Getting fieldname from the exception
144           while (it.hasNext()) {
145             fieldName = it.next().toString();
146           }
147 
148           List<String> errorsList = errorsMap.get(fieldName);
149 
150           if (errorsList == null) {
151             errorsList = new ArrayList<>();
152             errorsMap.put(fieldName, errorsList);
153           }
154 
155           errorsList.add(violation.getMessage());
156 
157         }
158 
159         t = new ValidationException(errorsMap.toString());
160       }
161       ValidationErrorUserException error = new ValidationErrorUserException(t);
162       return createResponse(t, error, errorsMap);
163     } else if (exception instanceof ValidationErrorUserException) {
164       return createResponse(exception, (ValidationErrorUserException) exception, null);
165     } else {
166       Class<?> exceptionClass = exception.getClass();
167       for (Class<?> securityError : this.securityExceptions) {
168         if (securityError.isAssignableFrom(exceptionClass)) {
169           return handleSecurityError(exception);
170         }
171       }
172       return handleGenericError(exception);
173     }
174   }

176   /**
177    * Creates the {@link Response} for the given validation exception.
178    *
179    * @param exception is the original validation exception.
180    * @param error is the wrapped exception or the same as <code>exception</code>.
181    * @param errorsMap is a map with all validation errors
182    * @return the requested {@link Response}.
183    */
184   protected Response createResponse(Throwable exception, ValidationErrorUserException error,
185       Map<String, List<String>> errorsMap) {
186 
187     LOG.warn("Service failed due to validation failure.", error);
188     if (exception == error) {
189       return createResponse(Status.BAD_REQUEST, error, errorsMap);
190     } else {
191       return createResponse(Status.BAD_REQUEST, error, exception.getMessage(), errorsMap);
192     }
193   }

195   /**
196    * Exception handling depending on technical Exception or not.
197    *
198    * @param exception the exception thrown
199    * @return the response build from error status
200    */
201   protected Response handleGenericError(Throwable exception) {
202 
203     NlsRuntimeException userError = TechnicalErrorUserException.getOrCreateUserException(exception);
204     if (userError.isTechnical()) {
205       LOG.error("Service failed on server", exception);
206     } else {
207       LOG.warn("Service failed due to business error: {}", exception.getMessage());
208     }
209     return createResponse(userError);
210   }

212   /**
213    * Exception handling for security exceptions.
214    *
215    * @param exception the exception thrown
216    * @return the response build from error status
217    */
218   protected Response handleSecurityError(Throwable exception) {
219 
220     // *** security error ***
221     NlsRuntimeException error;
222     if (exception instanceof NlsRuntimeException) {
223       error = (NlsRuntimeException) exception;
224     } else {
225       error = new SecurityErrorUserException(exception);
226     }
227     LOG.error("Service failed due to security error", error);
228     // NOTE: for security reasons we do not send any details about the error
229     // to the client!
230     String message;
231     String code = null;
232     if (this.exposeInternalErrorDetails) {
233       message = getExposedErrorDetails(error);
234     } else {
235       message = "forbidden";
236     }
237     return createResponse(Status.FORBIDDEN, message, code, error.getUuid(), null);
238   }
239 
240   /**
241    * @param error is the {@link Throwable} to extract message details from.
242    * @return the exposed message(s).
243    */
244   protected String getExposedErrorDetails(Throwable error) {
245 
246     StringBuilder buffer = new StringBuilder();
247     Throwable e = error;
248     while (e != null) {
249       if (buffer.length() > 0) {
250         buffer.append(StringUtil.LINE_SEPARATOR);
251       }
252       buffer.append(e.getClass().getSimpleName());
253       buffer.append(": ");
254       buffer.append(e.getLocalizedMessage());
255       e = e.getCause();
256     }
257     return buffer.toString();
258   }

262    *
263    * @param error the generic {@link NlsRuntimeException}.
264    * @return the corresponding {@link Response}.
265    */
266   protected Response createResponse(NlsRuntimeException error) {
267 
268     Status status;
269     if (error.isTechnical()) {
270       status = Status.INTERNAL_SERVER_ERROR;
271     } else {
272       status = Status.BAD_REQUEST;
273     }
274     return createResponse(status, error, null);
275   }

327   /**
328    * Create a response message as a JSON-String from the given parts.
329    *
330    * @param status is the HTTP {@link Status}.
331    * @param message is the JSON message attribute.
332    * @param code is the {@link NlsRuntimeException#getCode() error code}.
333    * @param uuid the {@link UUID} of the response message.
334    * @param errorsMap is a map with all validation errors
335    * @return the corresponding {@link Response}.
336    */
337   protected Response createResponse(Status status, String message, String code, UUID uuid,
338       Map<String, List<String>> errorsMap) {
339 
340     String json = createJsonErrorResponseMessage(message, code, uuid, errorsMap);
341     return Response.status(status).entity(json).build();
342   }

344   /**
345    * Create a response message as a JSON-String from the given parts.
346    *
347    * @param message the message of the response message
348    * @param code the code of the response message
349    * @param uuid the uuid of the response message
350    * @param errorsMap is a map with all validation errors
351    * @return the response message as a JSON-String
352    */
353   protected String createJsonErrorResponseMessage(String message, String code, UUID uuid,
354       Map<String, List<String>> errorsMap) {
355 
356     Map<String, Object> jsonMap = new HashMap<>();
357     if (message != null) {
358       jsonMap.put(KEY_MESSAGE, message);
359     }
360     if (code != null) {
361       jsonMap.put(KEY_CODE, code);
362     }
363     if (uuid != null) {
364       jsonMap.put(KEY_UUID, uuid.toString());
365     }
366 
367     if (errorsMap != null) {
368       jsonMap.put(KEY_ERRORS, errorsMap);
369     }
370 
371     String responseMessage = "";
372     try {
373       responseMessage = this.mapper.writeValueAsString(jsonMap);
374     } catch (JsonProcessingException e) {
375       LOG.error("Exception facade failed to create JSON.", e);
376       responseMessage = "{}";
377     }
378     return responseMessage;
379 
380   }

382   /**
383    * Add a response message to an existing response.
384    *
385    * @param exception the {@link WebApplicationException}.
386    * @return the response with the response message added
387    */
388   protected Response createResponse(WebApplicationException exception) {
389 
390     Response response = exception.getResponse();
391     int statusCode = response.getStatus();
392     Status status = Status.fromStatusCode(statusCode);
393     NlsRuntimeException error;
394     if (exception instanceof ServerErrorException) {
395       error = new TechnicalErrorUserException(exception);
396       LOG.error("Service failed on server", error);
397       return createResponse(status, error, null);
398     } else {
399       UUID uuid = UUID.randomUUID();
400       if (exception instanceof ClientErrorException) {
401         LOG.warn("Service failed due to unexpected request. UUDI: {}, reason: {} ", uuid, exception.getMessage());
402       } else {
403         LOG.warn("Service caused redirect or other error. UUID: {}, reason: {}", uuid, exception.getMessage());
404       }
405       return createResponse(status, exception.getMessage(), String.valueOf(statusCode), uuid, null);
406     }

427   /**
428    * @param exposeInternalErrorDetails - <code>true</code> if internal exception details shall be exposed to clients
429    *        (useful for debugging and testing), <code>false</code> if such details are hidden to prevent <a
430    *        href="https://www.owasp.org/index.php/Top_10_2013-A6-Sensitive_Data_Exposure">Sensitive Data Exposure</a>
431    *        (default, has to be used in production environment).
432    */
433   public void setExposeInternalErrorDetails(boolean exposeInternalErrorDetails) {
434 
435     this.exposeInternalErrorDetails = exposeInternalErrorDetails;
436     if (exposeInternalErrorDetails) {
437       String message =
438           "****** Exposing of internal error details is enabled! This violates OWASP A6 (Sensitive Data Exposure) and shall only be used for testing/debugging and never in production. ******";
439       LOG.warn(message);
440       // CHECKSTYLE:OFF (for development only)
441       System.err.println(message);
442       // CHECKSTYLE:ON
443     }
444   }
