23    @SuppressWarnings("rawtypes")
24    private static String dumpSession(HttpSession session, int indent) {
25  
26      String indentString = repeat(INDENDATION, indent);
27      if (session == null)
28        return indentString + "{ }";
29      StringBuilder sb = new StringBuilder();
30      sb.append(indentString).append("{\n");
31      sb.append(indentString).append(INDENDATION).append("'id': '").append(session.getId()).append("', \n");
32      sb.append(indentString).append(INDENDATION).append("'last_accessed_time': ").append(session.getLastAccessedTime())
33          .append(", \n");
34      sb.append(indentString).append(INDENDATION).append("'max_inactive_interval': ")
35          .append(session.getMaxInactiveInterval()).append(", \n");
36      sb.append(indentString).append(INDENDATION).append("'is_new': '").append(session.isNew()).append("', \n");
37      sb.append(indentString).append(INDENDATION).append("'attributes': {\n");
38      Enumeration attributeNames = session.getAttributeNames();
39      while (attributeNames.hasMoreElements()) {
40        String attributeName = (String) attributeNames.nextElement();
41        Object o = session.getAttribute(attributeName);
42        sb.append(indentString).append(INDENDATION).append("'").append(attributeName).append("': ").append("'")
43            .append(o.toString()).append("',\n");
44      }
45      sb.append(indentString).append(INDENDATION).append("}\n");
46      sb.append(indentString).append("}\n");
47      return sb.toString();
48    }

50    private static String dumpParameter(String indentString, String parameterName, String[] parameterValues) {
51  
52      StringBuilder sb = new StringBuilder();
53      sb.append(indentString).append(INDENDATION).append("'").append(parameterName).append("': ");
54      if (parameterValues == null || parameterValues.length == 0) {
55        sb.append("None");
56      } else {
57        if (parameterValues.length > 1)
58          sb.append("[");
59        sb.append(join(parameterValues, ","));
60        if (parameterValues.length > 1)
61          sb.append("]");
62      }
63      return sb.toString();
64    }

82    @SuppressWarnings({ "rawtypes", "unused" })
83    private static String dumpParameters(HttpServletRequest request, int indent) {
84  
85      String indentString = repeat(INDENDATION, indent);
86      StringBuilder sb = new StringBuilder();
87      sb.append(indentString).append("{\n");
88      Enumeration parameterNames = request.getParameterNames();
89      while (parameterNames.hasMoreElements()) {
90        String parameterName = (String) parameterNames.nextElement();
91        String[] parameterValues = request.getParameterValues(parameterName);
92        List<String> headerValuesList = new ArrayList<String>();
93        sb.append(dumpParameter(indentString, parameterName, parameterValues)).append(",\n");
94      }
95      sb.append(indentString).append("}\n");
96      return sb.toString();
97    }

99    private static String dumpCookie(Cookie cookie, String indentString) {
100 
101     if (cookie == null)
102       return "";
103     StringBuilder sb = new StringBuilder();
104     sb.append(indentString).append("{ \n");
105     sb.append(indentString).append(INDENDATION).append("'name': '").append(cookie.getName()).append("', \n");
106     sb.append(indentString).append(INDENDATION).append("'value': '").append(cookie.getValue()).append("', \n");
107     sb.append(indentString).append(INDENDATION).append("'domain': '").append(cookie.getDomain()).append("', \n");
108     sb.append(indentString).append(INDENDATION).append("'path': '").append(cookie.getPath()).append("', \n");
109     sb.append(indentString).append(INDENDATION).append("'max_age': ").append(cookie.getMaxAge()).append(", \n");
110     sb.append(indentString).append(INDENDATION).append("'version': ").append(cookie.getVersion()).append(", \n");
111     sb.append(indentString).append(INDENDATION).append("'comment': '").append(cookie.getComment()).append("', \n");
112     sb.append(indentString).append(INDENDATION).append("'secure': '").append(cookie.getSecure()).append("',\n");
113     sb.append(indentString).append("}");
114     return sb.toString();
115   }

117   private static String dumpCookies(HttpServletRequest request, int indent) {
118 
119     if (request.getCookies() == null)
120       return "";
121     String indentString = repeat(INDENDATION, indent);
122     StringBuilder sb = new StringBuilder();
123     sb.append(indentString).append("[\n");
124     int cookieCount = 0;
125     for (Cookie cookie : request.getCookies()) {
126       sb.append(dumpCookie(cookie, indentString + INDENDATION)).append(",\n");
127       cookieCount++;
128     }
129     if (cookieCount > 0) {
130       sb.delete(sb.length() - ",\n".length(), sb.length());
131     }
132     sb.append("\n").append(indentString).append("]\n");
133     return sb.toString();
134   }

136   @SuppressWarnings("rawtypes")
137   private static String dumpHeaders(HttpServletRequest request, int indent) {
138 
139     String indentString = repeat(INDENDATION, indent);
140     StringBuilder sb = new StringBuilder();
141     sb.append(indentString).append("{\n");
142     Enumeration headerNames = request.getHeaderNames();
143     while (headerNames.hasMoreElements()) {
144       String headerName = (String) headerNames.nextElement();
145       Enumeration headerValues = request.getHeaders(headerName);
146       List<String> headerValuesList = new ArrayList<String>();
147       while (headerValues.hasMoreElements()) {
148         String headerValue = (String) headerValues.nextElement();
149         headerValuesList.add(headerValue);
150       }
151       sb.append(dumpHeader(indentString, headerName, headerValuesList)).append(",\n");
152     }
153     sb.append(indentString).append("}\n");
154     return sb.toString();
155   }

157   public static String repeat(String what, int times) {
158 
159     StringBuilder buffer = new StringBuilder(what.length() * times);
160     for (int i = 0; i < times; i++) {
161       buffer.append(what);
162     }
163     return buffer.toString();
164   }

166   public static String join(List<String> values, String conjuction) {
167 
168     StringBuilder buffer = new StringBuilder();
169     for (String s : values) {
170       if (buffer.length() > 0) {
171         buffer.append(conjuction);
172       }
173       buffer.append(s);
174     }
175     return buffer.toString();
176   }

226   /**
227    * Debug complete request
228    *
229    * @param request Request parameter.
230    * @param printSession Enable session information printing
231    * @return A string with debug information on Request's header
232    */
233   public static String dump(HttpServletRequest request, boolean printSession) {
234 
235     StringBuilder sb = new StringBuilder();
236 
237     // GENERAL INFO
238     sb.append("<strong>PROTOCOL:</strong> ").append(request.getProtocol()).append("\n");
239     sb.append("<strong>METHOD:</strong> ").append(request.getMethod()).append("\n");
240     sb.append("<strong>QUERY STRING:</strong> ").append(request.getQueryString()).append("\n");
241     sb.append("<strong>PATH INFO:</strong> ").append(request.getPathInfo()).append("\n");
242     sb.append("<strong>PATH TRANSLATED:</strong> ").append(request.getPathTranslated()).append("\n");
243 
244     // COOKIES
245     sb.append("<strong>COOKIES:</strong>\n");
246     sb.append(dumpCookies(request, 1));
247 
248     // PARAMETERS
249     sb.append("<strong>PARAMETERS:</strong>\n");
250     sb.append(dumpParameters(request, 1));
251 
252     // HEADERS
253     sb.append("<strong>HEADERS:</strong>\n");
254     sb.append(dumpHeaders(request, 1));
255 
256     // SESSION
257     if (printSession) {
258       sb.append("<strong>SESSION:</strong>\n");
259       HttpSession session = request.getSession(false);
260       if (session != null) {
261         sb.append(dumpSession(session, 1));
262       } else {
263         sb.append("NO SESSION AVAILABLE\n");
264       }
265     }
266 
267     return sb.toString();
268   }
