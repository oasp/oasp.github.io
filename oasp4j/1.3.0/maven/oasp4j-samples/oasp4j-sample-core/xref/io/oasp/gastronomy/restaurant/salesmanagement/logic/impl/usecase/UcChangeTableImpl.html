18  /**
19   * UseCase: The guests can change a table to get a better table.
20   *
21   * @author rjoeris
22   */
23  @Named
24  @UseCase
25  public class UcChangeTableImpl extends AbstractUc implements UcChangeTable {
26  
27    private Tablemanagement tableManagement;
28  
29    private Salesmanagement salesManagement;
30  
31    @Override
32    @RolesAllowed(PermissionConstants.SAVE_TABLE)
33    public void changeTable(long orderId, long newTableId) {
34  
35      OrderEto order = this.salesManagement.findOrder(orderId);
36      // save old table data
37      long oldTableId = order.getTableId();
38      if (oldTableId == newTableId) {
39        return; // nothing to do...
40      }
41      TableEto oldTable = this.tableManagement.findTable(oldTableId);
42      TableState oldTableState = oldTable.getState();
43  
44      // throw exception if the newTableState is occupied
45      TableEto newTable = this.tableManagement.findTable(newTableId);
46      if (newTable.getState().isOccupied()) {
47        throw new ChangeTableIllegalStateCombinationException(order, newTable);
48      }
49  
50      // update order
51      order.setTableId(newTableId);
52  
53      // marks new table with copied status
54      newTable.setState(oldTableState);
55  
56      this.tableManagement.saveTable(newTable);
57  
58      // saves Order for the new table
59      this.salesManagement.saveOrder(order);
60  
61      // change table status:
62      // marks old table as free
63      oldTable.setState(TableState.FREE);
64      this.tableManagement.saveTable(oldTable);
65  
66    }
67

75    }

84    }
