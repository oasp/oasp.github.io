19  /**
20   * Implementation of {@link UcFindOrderPosition}.
21   *
22   * @author jozitz
23   */
24  @Named
25  @UseCase
26  public class UcFindOrderPositionImpl extends AbstractOrderPositionUc implements UcFindOrderPosition {
27  
28    private static final Logger LOG = LoggerFactory.getLogger(UcFindOrderPositionImpl.class);
29  
30    @Override
31    @RolesAllowed(PermissionConstants.FIND_ORDER_POSITION)
32    public OrderPositionEto findOrderPosition(Long orderPositionId) {
33  
34      LOG.debug("Get order position.");
35      OrderPositionEntity orderPositionEntities = getOrderPositionDao().findOne(orderPositionId);
36      OrderPositionEto orderPositionBo = getBeanMapper().map(orderPositionEntities, OrderPositionEto.class);
37      return orderPositionBo;
38    }
39  
40    @Override
41    @RolesAllowed(PermissionConstants.FIND_ORDER_POSITION)
42    public List<OrderPositionEto> findOrderPositionsByOrderId(Long orderId) {
43  
44      List<OrderPositionEntity> positions = getOrderPositionDao().findOrderPositionsByOrder(orderId);
45      return getBeanMapper().mapList(positions, OrderPositionEto.class);
46    }
47  
48    @Override
49    @RolesAllowed(PermissionConstants.FIND_ORDER_POSITION)
50    public List<OrderPositionEto> findOpenOrderPositionsByOrderId(Long orderId) {
51  
52      LOG.debug("Get all open order positions for order id '" + orderId + "'.");
53      return getBeanMapper()
54          .mapList(getOrderPositionDao().findOpenOrderPositionsByOrder(orderId), OrderPositionEto.class);
55    }
56  
57    @Override
58    @RolesAllowed(PermissionConstants.FIND_ORDER_POSITION)
59    public List<OrderPositionEto> findOrderPositions(OrderPositionSearchCriteriaTo criteria) {
60  
61      List<OrderPositionEntity> positions = getOrderPositionDao().findOrderPositions(criteria);
62      return getBeanMapper().mapList(positions, OrderPositionEto.class);
63    }
