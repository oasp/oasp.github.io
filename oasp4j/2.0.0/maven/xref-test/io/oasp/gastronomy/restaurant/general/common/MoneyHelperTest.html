41    /**
42     * Test behavior when passing an empty list.
43     */
44    @Test
45    public void sumTestEmptyList() {
46  
47      List<Money> moneyList = new ArrayList<>();
48      Money summedAmount = MoneyHelper.sumMoneys(moneyList);
49      assertThat(MoneyHelper.ZERO_MONEY).isEqualTo(summedAmount);
50    }

52    /**
53     * Test for summing up {@link Money}s with default currency (EUR).
54     */
55    @Test
56    public void sumTestEURList() {
57  
58      List<Money> moneyList = new ArrayList<>();
59  
60      double[] amounts = { 15.28, 11.47, 4.20 };
61      double sum = 0.0;
62  
63      for (double amount : amounts) {
64        moneyList.add(MoneyHelper.getMoneyWithDefaultCurrency(amount));
65        sum += amount;
66      }
67  
68      Money summedAmount = MoneyHelper.sumMoneys(moneyList);
69  
70      assertThat(MoneyHelper.getMoney(sum, "EUR")).isEqualTo(summedAmount);
71    }
