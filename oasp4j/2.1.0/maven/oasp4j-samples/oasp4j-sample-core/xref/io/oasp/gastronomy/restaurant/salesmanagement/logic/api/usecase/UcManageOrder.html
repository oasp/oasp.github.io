3   import io.oasp.gastronomy.restaurant.salesmanagement.logic.api.to.OrderCto;

6   import javax.validation.Valid;
7   import javax.validation.constraints.NotNull;
8   
9   /**
10   * Interface of UcMangeOrder to centralize documentation and signatures of methods.
11   *
12   * @author mvielsac
13   */
14  public interface UcManageOrder {
15  
16    /**
17     * If no ID is contained creates the {@link OrderCto} for the first time. Else it updates the {@link OrderCto} with
18     * given ID. If no {@link OrderCto} with given ID is present, an exception will be thrown.
19     *
20     * @param order the {@link OrderCto} to persist.
21     * @return the persisted {@link OrderCto}.
22     */
23    OrderCto saveOrder(@NotNull OrderCto order);
24  
25    /**
26     * If no ID is contained creates the {@link OrderEto} for the first time. Else it updates the {@link OrderEto} with
27     * given ID. If no {@link OrderEto} with given ID is present, an exception will be thrown.
28     *
29     * @param order the {@link OrderEto} to persist.
30     * @return the persisted {@link OrderEto}.
31     */
32    OrderEto saveOrder(@NotNull @Valid OrderEto order);
33  
34    /**
35     * @param id is the {@link OrderEto#getId() ID} of the order to delete.
36     */
37    void deleteOrder(long id);
38  
39  }
