21  /**
22   *
23   * The service class for REST calls in order to execute the methods in {@link Staffmanagement}.
24   *
25   * @author agreul, geazzi, jmolinar
26   */
27  @Path("/staffmanagement/v1/staff")
28  @Consumes(MediaType.APPLICATION_JSON)
29  @Produces(MediaType.APPLICATION_JSON)
30  public interface StaffmanagementRestService extends RestService {
31  
32    /**
33     * @return a list of all {@link StaffMemberEto}
34     *
35     */
36    @GET
37    @Path("/")
38    @Deprecated
39    List<StaffMemberEto> getAllStaffMember();
40  
41    /**
42     * @param login the login of a staff member
43     * @return {@link StaffMemberEto}
44     */
45    @GET
46    @Path("/{login}")
47    StaffMemberEto getStaffMember(@PathParam("login") String login);
48  
49    /**
50     * @param staffMemberBo the staffMember to be updated as JSON
51     */
52    @PUT
53    @Path("/{login}")
54    @Deprecated
55    void updateStaffMember(StaffMemberEto staffMemberBo);
56  
57    /**
58     * Calls {@link Staffmanagement#saveStaffMember}.
59     *
60     * @param staffMemberEto the staffMember to be created or updated
61     * @return the saved {@link StaffMemberEto}
62     */
63    @POST
64    @Path("/")
65    StaffMemberEto saveStaffMember(StaffMemberEto staffMemberEto);
66  
67    /**
68     * @param login the login
69     */
70    @DELETE
71    @Path("/{login}")
72    void deleteStaffMember(@PathParam("login") String login);
73  
74    /**
75     * Delegates to {@link Staffmanagement#findStaffMemberEtos}.
76     *
77     * @param searchCriteriaTo the pagination and search criteria to be used for finding staffmembers.
78     * @return the {@link PaginatedListTo list} of matching {@link StaffMemberEto}s.
79     */
80    @Path("/search")
81    @POST
82    PaginatedListTo<StaffMemberEto> findStaffMembersByPost(StaffMemberSearchCriteriaTo searchCriteriaTo);
83  
84  }
