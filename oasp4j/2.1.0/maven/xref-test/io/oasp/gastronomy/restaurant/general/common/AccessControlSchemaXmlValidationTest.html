27  public class AccessControlSchemaXmlValidationTest extends ModuleTest {

29    /**
30     * Tests if the access-control-schema.xml is valid.
31     *
32     * @throws ParserConfigurationException If a DocumentBuilder cannot be created which satisfies the configuration
33     *         requested.
34     * @throws IOException If any IO errors occur.
35     * @throws SAXException If an error occurs during the validation.
36     */
37    @Test
38    public void validateAccessControllSchema() throws ParserConfigurationException, SAXException, IOException {
39  
40      // parse an XML document into a DOM tree
41      DocumentBuilder parser = DocumentBuilderFactory.newInstance().newDocumentBuilder();
42      String xmlPath = getClass().getResource("/config/app/security/access-control-schema.xml").getPath();
43      Document document = parser.parse(new File(xmlPath));
44  
45      // create a SchemaFactory capable of understanding WXS schemas
46      SchemaFactory factory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
47  
48      // load a WXS schema, represented by a Schema instance
49      URL schemaPath = getClass().getResource("/io/oasp/module/security/access-control-schema.xsd");
50      Schema schema = factory.newSchema(schemaPath);
51  
52      // create a Validator instance, which can be used to validate an instance document
53      Validator validator = schema.newValidator();
54  
55      // validate the DOM tree
56      validator.validate(new DOMSource(document));
57    }
58  }
