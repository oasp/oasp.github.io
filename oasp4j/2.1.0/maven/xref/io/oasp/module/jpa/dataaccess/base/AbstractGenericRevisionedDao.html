3   package io.oasp.module.jpa.dataaccess.base;
4   
5   import io.oasp.module.jpa.dataaccess.api.GenericRevisionedDao;
6   import io.oasp.module.jpa.dataaccess.api.RevisionMetadata;
7   import io.oasp.module.jpa.dataaccess.api.MutablePersistenceEntity;
8   import io.oasp.module.jpa.dataaccess.impl.LazyRevisionMetadata;
9   
10  import java.util.ArrayList;
11  import java.util.List;
12  
13  import net.sf.mmm.util.exception.api.ObjectNotFoundException;
14  
15  import org.hibernate.envers.AuditReader;
16  import org.hibernate.envers.AuditReaderFactory;
17  
18  /**
19   * This is the abstract base-implementation of a {@link AbstractGenericDao} using {@link org.hibernate.envers
20   * Hibernate-Envers} to manage the revision-control.
21   *
22   * @param <ID> is the type of the {@link MutablePersistenceEntity#getId() primary key} of the managed
23   *        {@link MutablePersistenceEntity entity}.
24   * @param <ENTITY> is the {@link #getEntityClass() type} of the managed entity.
25   *
26   * @author hohwille
27   */
28  public abstract class AbstractGenericRevisionedDao<ID, ENTITY extends MutablePersistenceEntity<ID>> extends
29      AbstractGenericDao<ID, ENTITY> implements GenericRevisionedDao<ID, ENTITY> {
30  
31    /**
32     * The constructor.
33     */
34    public AbstractGenericRevisionedDao() {
35  
36      super();
37    }
38  
39    /**
40     * @return the auditReader
41     */
42    protected AuditReader getAuditReader() {
43  
44      return AuditReaderFactory.get(getEntityManager());
45    }
46  
47    @Override
48    public ENTITY load(ID id, Number revision) throws ObjectNotFoundException {
49  
50      if (revision == MutablePersistenceEntity.LATEST_REVISION) {
51        return find(id);
52      } else {
53        return loadRevision(id, revision);
54      }
55    }
56  
57    /**
58     * This method gets a historic revision of the {@link net.sf.mmm.util.entity.api.GenericEntity} with the given
59     * <code>id</code>.
60     *
61     * @param id is the {@link net.sf.mmm.util.entity.api.GenericEntity#getId() ID} of the requested
62     *        {@link net.sf.mmm.util.entity.api.GenericEntity entity}.
63     * @param revision is the {@link MutablePersistenceEntity#getRevision() revision}
64     * @return the requested {@link net.sf.mmm.util.entity.api.GenericEntity entity}.
65     */
66    protected ENTITY loadRevision(Object id, Number revision) {
67  
68      Class<? extends ENTITY> entityClassImplementation = getEntityClass();
69      ENTITY entity = getAuditReader().find(entityClassImplementation, id, revision);
70      if (entity != null) {
71        entity.setRevision(revision);
72      }
73      return entity;
74    }
75  
76    @Override
77    public List<Number> getRevisionHistory(ID id) {
78  
79      return getAuditReader().getRevisions(getEntityClass(), id);
80    }
81  
82    @Override
83    public List<RevisionMetadata> getRevisionHistoryMetadata(Object id) {
84  
85      AuditReader auditReader = getAuditReader();
86      List<Number> revisionList = auditReader.getRevisions(getEntityClass(), id);
87      List<RevisionMetadata> result = new ArrayList<>();
88      for (Number revision : revisionList) {
89        Long revisionLong = Long.valueOf(revision.longValue());
90        result.add(new LazyRevisionMetadata(getEntityManager(), revisionLong));
91      }
92      return result;
93    }
94  }
