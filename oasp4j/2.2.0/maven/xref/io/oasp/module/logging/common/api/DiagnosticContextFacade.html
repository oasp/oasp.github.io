3   /**
4    * This is the interface for a simple facade to write data into the {@link org.slf4j.MDC mapped diagnostic context}. As
5    * additional value you can easily hook in custom extensions without interfering the logger implementation. A use case
6    * may be to provide diagnostic informations also to additional components such as a performance monitoring module.
7    * Therefore setting diagnostic information from OASP code is always indirected via this interface so the implementation
8    * can be extended or replaced (what is not as easy for {@link org.slf4j.MDC#put(String, String) static methods}).
9    *
10   */
11  public interface DiagnosticContextFacade {
12  
13    /**
14     * @return the current {@link LoggingConstants#CORRELATION_ID correlation ID} or {@code null} if not
15     *         {@link #setCorrelationId(String) set}.
16     */
17    String getCorrelationId();
18  
19    /**
20     * Sets the {@link LoggingConstants#CORRELATION_ID correlation ID} for the current processing and thread.
21     *
22     * @param correlationId is the {@link LoggingConstants#CORRELATION_ID correlation ID} as unique identifier for the
23     *        current processing task.
24     */
25    void setCorrelationId(String correlationId);
26  
27    /**
28     * Removes the {@link #setCorrelationId(String) correlation ID} from the diagnostic context.
29     */
30    void removeCorrelationId();
31  
32  }
