128     JSONObject postRequest = new JSONObject();

221   /**
222    * This test method creates a sample instance of {@link OrderCto} and saves it into the database. In addition some
223    * sample instances of {@link OrderPositionEto} are created, linked to the sample {@link OrderCto} object by the
224    * id/orderId and saved into the database. Thereafter a HTTP GET request is sent demanding all saved
225    * {@link OrderPositionEto} objects in the database. Finally the body of the response {@link JSONObject} is tested
226    * concerning the number of loaded orderPositionEtos and the correctness attributes of the sample instances.
227    */
228   @Test
229   public void getAllOrderPositions() {
230 
231     // given
232     OrderCto sampleOrderCto = this.helper.createSampleOrderCto(SAMPLE_TABLE_ID);
233     OrderCto responseOrderCto = this.service.saveOrder(sampleOrderCto);
234     assertThat(responseOrderCto).isNotNull();
235 
236     int oldNumberOfOrderPositions = getNumberOfOrderPositions();
237     int numberOfOrderPositionsToSave = NUMBER_OF_SAMPLE_ORDER_POSITIONS;
238 
239     OrderPositionEto sampleOrderPositionEto;
240     ArrayList<OrderPositionEto> savedOrderPositionEtos = new ArrayList<>();
241     for (int i = 0; i < numberOfOrderPositionsToSave; ++i) {
242       sampleOrderPositionEto = new OrderPositionEtoBuilder().orderId(responseOrderCto.getOrder().getId())
243           .offerId(SAMPLE_OFFER_ID).createNew();
244       savedOrderPositionEtos.add(this.service.saveOrderPosition(sampleOrderPositionEto));
245     }
246 
247     int newNumberOfOrderPositions = getNumberOfOrderPositions();
248 
249     HttpEntity<String> getRequest = new HttpEntity<>(this.AUTHENTIFICATED_HEADERS);
250 
251     // when
252     ResponseEntity<String> getResponse =
253         this.template.exchange(generateBaseUrl() + "orderposition/", HttpMethod.GET, getRequest, String.class);
254 
255     // then
256     assertThat(getResponse).isNotNull();
257 
258     JSONArray responseOrderPositions = new JSONArray(getResponse.getBody());
259     assertThat(responseOrderPositions).isNotNull();
260 
261     assertThat(oldNumberOfOrderPositions + numberOfOrderPositionsToSave).isEqualTo(newNumberOfOrderPositions);
262     assertThat(responseOrderPositions.length()).isEqualTo(newNumberOfOrderPositions);
263 
264     int responseOrderPositionId = 0;
265     int countNumberOfSavedOrderPositions = 0;
266 
267     for (int i = 0; i < responseOrderPositions.length(); ++i) {
268       responseOrderPositionId = responseOrderPositions.getJSONObject(i).getInt("id");
269       for (OrderPositionEto orderPositionEto : savedOrderPositionEtos) {
270         if (responseOrderPositionId == orderPositionEto.getId()) {
271 
272           JSONAssert.assertEquals("{orderId:" + Long.toString(orderPositionEto.getOrderId()) + "}",
273               responseOrderPositions.getJSONObject(i), false);
274           JSONAssert.assertEquals("{offerId:" + Long.toString(SAMPLE_OFFER_ID) + "}",
275               responseOrderPositions.getJSONObject(i), false);
276           JSONAssert.assertEquals("{offerName:" + SAMPLE_OFFER_NAME + "}", responseOrderPositions.getJSONObject(i),
277               false);
278           JSONAssert.assertEquals("{price:" + "\"" + SAMPLE_PRICE.getValue() + "\"" + "}",
279               responseOrderPositions.getJSONObject(i).toString(), false);
280 
281           countNumberOfSavedOrderPositions++;
282         }
283       }
284     }
285 
286     assertThat(countNumberOfSavedOrderPositions).isEqualTo(numberOfOrderPositionsToSave);
287   }

343   /**
344    * This method builds an encoded authentication header and returns it
345    */
346   private HttpHeaders getAuthentificatedHeaders() {
347 
348     String plainCreds = ROLE + ":" + ROLE;
349     byte[] plainCredsBytes = plainCreds.getBytes();
350     byte[] base64CredsBytes = Base64.encodeBase64(plainCredsBytes);
351     String base64Creds = new String(base64CredsBytes);
352     HttpHeaders headers = new HttpHeaders();
353     headers.add("Authorization", "Basic " + base64Creds);
354 
355     return headers;
356   }
