3   package io.oasp.module.jpa.dataaccess.api;
4   
5   import net.sf.mmm.util.entity.api.MutableGenericEntity;
6   import net.sf.mmm.util.entity.api.PersistenceEntity;
7   
8   /**
9    * This is the interface for a {@link PersistenceEntity} in OASP.
10   *
11   * @param <ID> is the type of the {@link #getId() primary key}.
12   *
13   * @see AbstractPersistenceEntity
14   *
15   */
16  public interface MutablePersistenceEntity<ID> extends PersistenceEntity<ID>, MutableGenericEntity<ID> {
17  
18    /**
19     * This method sets the {@link #getRevision() revision} of this entity. <br>
20     * <b>ATTENTION:</b><br>
21     * This operation should only be used in specific cases and if you are aware of what you are doing as this attribute
22     * is managed by the persistence. However, for final freedom we decided to add this method to the API (e.g. to copy
23     * from transfer-object to persistent-entity and vice-versa).
24     *
25     * @param revision is the new value of {@link #getRevision()}.
26     */
27    void setRevision(Number revision);
28  
29  }
