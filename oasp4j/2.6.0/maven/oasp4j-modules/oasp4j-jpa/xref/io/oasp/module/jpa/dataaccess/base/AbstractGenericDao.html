34  /**
35   * This is the abstract base-implementation of the {@link GenericDao} interface.
36   *
37   * @param <ID> is the generic type if the {@link PersistenceEntity#getId() primary key}.
38   * @param <E> is the generic type of the managed {@link PersistenceEntity}.
39   *
40   */
41  // @Repository
42  public abstract class AbstractGenericDao<ID, E extends PersistenceEntity<ID>> implements GenericDao<ID, E> {

83  
84      return getEntityClass().getSimpleName();

87    @Override
88    public E save(E entity) {
89  
90      if (isNew(entity)) {
91        getEntityManager().persist(entity);
92        LOG.debug("Saved new {} with id {}.", getEntityName(), entity.getId());
93        return entity;
94      } else {
95        if (getEntityManager().find(entity.getClass(), entity.getId()) != null) {
96          E update = getEntityManager().merge(entity);
97          LOG.debug("Updated {} with id {}.", getEntityName(), entity.getId());
98          return update;
99        } else {
100         throw new EntityNotFoundException("Entity not found");
101       }
102     }
103   }

105   /**
106    * Determines if the given {@link PersistenceEntity} is {@link PersistenceEntity#STATE_NEW new}.
107    *
108    * @param entity is the {@link PersistenceEntity} to check.
109    * @return {@code true} if {@link PersistenceEntity#STATE_NEW new}, {@code false} otherwise (e.g.
110    *         {@link PersistenceEntity#STATE_DETACHED detached} or {@link PersistenceEntity#STATE_MANAGED managed}.
111    */
112   protected boolean isNew(E entity) {
113 
114     return entity.getId() == null;
115   }

138   @Override
139   public E find(ID id) throws ObjectNotFoundUserException {
140 
141     E entity = findOne(id);
142     if (entity == null) {
143       throw new ObjectNotFoundUserException(getEntityClass().getSimpleName(), id);
144     }
145     return entity;

151     // pointless...

155   /**
156    * @return an {@link Iterable} to find ALL {@link #getEntityClass() managed entities} from the persistent store. Not
157    *         exposed to API by default as this might not make sense for all kind of entities.
158    */
159   public List<E> findAll() {
160 
161     CriteriaQuery<E> query = getEntityManager().getCriteriaBuilder().createQuery(getEntityClass());
162     Root<E> root = query.from(getEntityClass());
163     query.select(root);
164     TypedQuery<E> typedQuery = getEntityManager().createQuery(query);
165     List<E> resultList = typedQuery.getResultList();
166     LOG.debug("Query for all {} objects returned {} hit(s).", getEntityName(), resultList.size());
167     return resultList;
168   }
169 
170   @Override
171   public List<E> findAll(Iterable<ID> ids) {
172 
173     CriteriaBuilder builder = getEntityManager().getCriteriaBuilder();
174     CriteriaQuery<E> query = builder.createQuery(getEntityClass());
175     Root<E> root = query.from(getEntityClass());
176     query.select(root);
177     query.where(root.get("id").in(toCollection(ids)));
178     TypedQuery<E> typedQuery = getEntityManager().createQuery(query);
179     List<E> resultList = typedQuery.getResultList();
180     LOG.debug("Query for selection of {} objects returned {} hit(s).", getEntityName(), resultList.size());
181     return resultList;
182   }
183 
184   /**
185    * @param ids sequence of id
186    * @return a collection of these ids to use {@link Predicate#in(Collection)} for instance
187    */
188   protected Collection<ID> toCollection(Iterable<ID> ids) {
189 
190     if (ids instanceof Collection) {
191       return (Collection<ID>) ids;
192     }
193 
194     final Collection<ID> idsList = new ArrayList<>();
195     for (final ID id : ids) {
196       idsList.add(id);
197     }
198     return idsList;
199   }

201   @Override
202   public void delete(ID id) {
203 
204     E entity = getEntityManager().getReference(getEntityClass(), id);
205     getEntityManager().remove(entity);
206     LOG.debug("Deleted {} with ID {}.", getEntityName(), id);
207   }

209   @Override
210   public void delete(E entity) {
211 
212     // entity might be detached and could cause trouble in entityManager on remove
213     if (getEntityManager().contains(entity)) {
214       getEntityManager().remove(entity);
215       LOG.debug("Deleted {} with ID {}.", getEntityName(), entity.getId());
216     } else {
217       delete(entity.getId());
218     }
219 
220   }
221 
222   @Override
223   public void delete(Iterable<? extends E> entities) {
224 
225     for (E entity : entities) {
226       delete(entity);
227     }
228   }

236   /**
237    * Returns a paginated list of entities according to the supplied {@link SearchCriteriaTo criteria}.
238    * <p>
239    * Applies {@code limit} and {@code offset} values to the supplied {@code query} according to the supplied
240    * {@link PaginationTo pagination} information inside {@code criteria}.
241    * <p>
242    * If a {@link PaginationTo#isTotal() total count} of available entities is requested, will also execute a second
243    * query, without pagination parameters applied, to obtain said count.
244    * <p>
245    * Will install a query timeout if {@link SearchCriteriaTo#getSearchTimeout()} is not null.
246    *
247    * @param criteria contains information about the requested page.
248    * @param query is a query which is preconfigured with the desired conditions for the search.
249    * @param expr is used for the final mapping from the SQL result to the entities.
250    * @return a paginated list.
251    */
252   protected PaginatedListTo<E> findPaginated(SearchCriteriaTo criteria, JPAQuery query, Expression<E> expr) {
253 
254     applyCriteria(criteria, query);
255 
256     PaginationTo pagination = criteria.getPagination();
257 
258     PaginationResultTo paginationResult = createPaginationResult(pagination, query);
259 
260     applyPagination(pagination, query);
261     List<E> paginatedList = query.list(expr);
262 
263     return new PaginatedListTo<>(paginatedList, paginationResult);
264   }

266   /**
267    * Creates a {@link PaginationResultTo pagination result} for the given {@code pagination} and {@code query}.
268    * <p>
269    * Needs to be called before pagination is applied to the {@code query}.
270    *
271    * @param pagination contains information about the requested page.
272    * @param query is a query preconfigured with the desired conditions for the search.
273    * @return information about the applied pagination.
274    */
275   protected PaginationResultTo createPaginationResult(PaginationTo pagination, JPAQuery query) {
276 
277     Long total = calculateTotalBeforePagination(pagination, query);
278 
279     return new PaginationResultTo(pagination, total);
280   }

282   /**
283    * Calculates the total number of entities the given {@link JPAQuery query} would return without pagination applied.
284    * <p>
285    * Needs to be called before pagination is applied to the {@code query}.
286    *
287    * @param pagination is the pagination information as requested by the client.
288    * @param query is the {@link JPAQuery query} for which to calculate the total.
289    * @return the total count, or {@literal null} if {@link PaginationTo#isTotal()} is {@literal false}.
290    */
291   protected Long calculateTotalBeforePagination(PaginationTo pagination, JPAQuery query) {
292 
293     Long total = null;
294     if (pagination.isTotal()) {
295       total = query.clone().count();
296     }
297 
298     return total;
299   }

301   /**
302    * Applies the {@link PaginationTo pagination criteria} to the given {@link JPAQuery}.
303    *
304    * @param pagination is the {@link PaginationTo pagination criteria} to apply.
305    * @param query is the {@link JPAQuery} to apply to.
306    */
307   protected void applyPagination(PaginationTo pagination, JPAQuery query) {
308 
309     if (pagination == PaginationTo.NO_PAGINATION) {
310       return;
311     }
312 
313     Integer limit = pagination.getSize();
314     if (limit != null) {
315       query.limit(limit);
316 
317       int page = pagination.getPage();
318       if (page > 0) {
319         query.offset((page - 1) * limit);
320       }
321     }
322   }

324   /**
325    * Applies the meta-data of the given {@link AbstractSearchCriteria search criteria} to the given {@link JPAQuery}.
326    *
327    * @param criteria is the {@link AbstractSearchCriteria search criteria} to apply.
328    * @param query is the {@link JPAQuery} to apply to.
329    */
330   protected void applyCriteria(AbstractSearchCriteria criteria, JPAQuery query) {
331 
332     Integer limit = criteria.getMaximumHitCount();
333     if (limit != null) {
334       query.limit(limit);
335     }
336     int offset = criteria.getHitOffset();
337     if (offset > 0) {
338       query.offset(offset);
339     }
340     Long timeout = criteria.getSearchTimeout();
341     if (timeout != null) {
342       query.setHint("javax.persistence.query.timeout", timeout.intValue());
343     }
344   }

346   /**
347    * Applies the meta-data of the given {@link AbstractSearchCriteria search criteria} to the given {@link Query}.
348    *
349    * @param criteria is the {@link AbstractSearchCriteria search criteria} to apply.
350    * @param query is the {@link Query} to apply to.
351    */
352   protected void applyCriteria(AbstractSearchCriteria criteria, Query query) {
353 
354     Integer limit = criteria.getMaximumHitCount();
355     if (limit != null) {
356       query.setMaxResults(limit);
357     }
358     int offset = criteria.getHitOffset();
359     if (offset > 0) {
360       query.setFirstResult(offset);
361     }
362     Long timeout = criteria.getSearchTimeout();
363     if (timeout != null) {
364       query.setHint("javax.persistence.query.timeout", timeout.intValue());
365     }
366   }
