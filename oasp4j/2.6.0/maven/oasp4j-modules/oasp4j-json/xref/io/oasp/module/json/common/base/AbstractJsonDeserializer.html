17  /**
18   * Helper class to simplify implementation of {@link JsonDeserializer}.
19   *
20   * @param <T> the class to be deserialized
21   *
22   * @since 3.0.0
23   */
24  public abstract class AbstractJsonDeserializer<T> extends JsonDeserializer<T> {
25  
26    /**
27     * @param <V> the type to convert the requested value to.
28     * @param node parent node to deserialize
29     * @param fieldName the name of the JSON property to get the value from.
30     * @param type the {@link Class} reflecting the type to convert the requested value to.
31     * @return requested value converted to the given {@code type}. May not be {@code null}.
32     * @throws RuntimeException if the requested value is not present or if the conversion failed.
33     */
34    protected <V> V getRequiredValue(JsonNode node, String fieldName, Class<V> type) throws RuntimeException {
35  
36      return getValue(node, fieldName, type, true);
37    }
38  
39    /**
40     * @param <V> the type to convert the requested value to.
41     * @param node parent node to deserialize
42     * @param fieldName the name of the JSON property to get the value from.
43     * @param type the {@link Class} reflecting the type to convert the requested value to.
44     * @param fallback is the default returned if the requested value is undefined.
45     * @return requested value converted to the given {@code type} or {@code fallback} if undefined.
46     * @throws RuntimeException if the conversion failed.
47     */
48    protected <V> V getOptionalValue(JsonNode node, String fieldName, Class<V> type, V fallback) throws RuntimeException {
49  
50      V result = getValue(node, fieldName, type, false);
51      if (result == null) {
52        result = fallback;
53      }
54      return result;
55    }
56  
57    /**
58     * @param <V> the type to convert the requested value to.
59     * @param node parent node to deserialize
60     * @param fieldName the name of the JSON property to get the value from.
61     * @param type the {@link Class} reflecting the type to convert the requested value to.
62     * @param required {@code true} if the requested value has to be present, {@code false} otherwise (if optional).
63     * @return requested value converted to the given {@code type} or {@code null} if undefined and {@code required} is
64     *         {@code true}.
65     * @throws RuntimeException if {@code required} is {@code true} and the requested value is not present or if the
66     *         conversion failed.
67     */
68    @SuppressWarnings("unchecked")
69    protected <V> V getValue(JsonNode node, String fieldName, Class<V> type, boolean required) throws RuntimeException {
70  
71      V value = null;
72      JsonNode childNode = node.get(fieldName);
73      if (childNode != null) {
74        if (!childNode.isNull()) {
75          try {
76            if (type == String.class) {
77              value = type.cast(childNode.asText());
78            } else if (type == BigDecimal.class) {
79              value = type.cast(new BigDecimal(childNode.asText()));
80            } else if (type == BigInteger.class) {
81              value = type.cast(new BigInteger(childNode.asText()));
82            } else if (type == Date.class) {
83              value = type.cast(Iso8601UtilImpl.getInstance().parseDate(childNode.asText()));
84            } else if (type == Calendar.class) {
85              value = type.cast(Iso8601UtilImpl.getInstance().parseCalendar(childNode.asText()));
86            } else if (type == Boolean.class) {
87              // types that may be primitive shall be casted explicitly as Class.cast does not work for primitive types.
88              value = (V) Boolean.valueOf(childNode.booleanValue());
89            } else if (type == Integer.class) {
90              value = (V) Integer.valueOf(childNode.asText());
91            } else if (type == Long.class) {
92              value = (V) Long.valueOf(childNode.asText());
93            } else if (type == Double.class) {
94              value = (V) Double.valueOf(childNode.asText());
95            } else if (type == Float.class) {
96              value = (V) Float.valueOf(childNode.asText());
97            } else if (type == Short.class) {
98              value = (V) Short.valueOf(childNode.asText());
99            } else if (type == Byte.class) {
100             value = (V) Byte.valueOf(childNode.asText());
101           } else {
102             throw new IllegalArgumentException("Unsupported value type " + type.getName());
103           }
104         } catch (NumberFormatException e) {
105           throw new IllegalArgumentException("Failed to convert value to type " + type.getName(), e);
106         }
107       }
108     }
109     if ((value == null) && (required)) {
110       throw new IllegalStateException(
111           "Deserialization failed due to missing " + type.getSimpleName() + " field " + fieldName + "!");
112     }
113     return value;
114   }
115 
116   @Override
117   public T deserialize(JsonParser jp, DeserializationContext ctxt) throws IOException, JsonProcessingException {
118 
119     JsonNode node = jp.getCodec().readTree(jp);
120     return deserializeNode(node);
121   }
122 
123   /**
124    * @param node is the {@link JsonNode} with the value content to be deserialized
125    * @return the deserialized java object
126    */
127   protected abstract T deserializeNode(JsonNode node);
128 }
