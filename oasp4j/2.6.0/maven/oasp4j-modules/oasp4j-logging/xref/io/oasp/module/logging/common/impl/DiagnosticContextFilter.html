21  /**
22   * Request logging filter that adds the request log message to the SLF4j mapped diagnostic context (MDC) before the
23   * request is processed, removing it again after the request is processed.
24   *
25   */
26  public class DiagnosticContextFilter implements Filter {
27  
28    private static final Logger LOG = LoggerFactory.getLogger(DiagnosticContextFilter.class);
29  
30    /**
31     * The name of the {@link FilterConfig#getInitParameter(String) init parameter} for
32     * {@link #setCorrelationIdHttpHeaderName(String)}.
33     */
34    private static final String CORRELATION_ID_HEADER_NAME_PARAM = "correlationIdHeaderName";
35  
36    /** The default value for {@link #setCorrelationIdHttpHeaderName(String)}. */
37    public static final String CORRELATION_ID_HEADER_NAME_DEFAULT = "X-Correlation-Id";
38  
39    /** @see #setCorrelationIdHttpHeaderName(String) */
40    private String correlationIdHttpHeaderName;
41  
42    private DiagnosticContextFacade diagnosticContextFacade;
43  
44    /**
45     * The constructor.
46     */
47    public DiagnosticContextFilter() {
48  
49      super();
50      this.correlationIdHttpHeaderName = CORRELATION_ID_HEADER_NAME_DEFAULT;
51    }
52  
53    /**
54     * @param correlationIdHttpHeaderName is the name of the {@link HttpServletRequest#getHeader(String) HTTP header} for
55     *        the {@link io.oasp.module.logging.common.api.LoggingConstants#CORRELATION_ID correlation ID}.
56     */
57    public void setCorrelationIdHttpHeaderName(String correlationIdHttpHeaderName) {
58  
59      this.correlationIdHttpHeaderName = correlationIdHttpHeaderName;
60    }
61  
62    @Override
63    public void destroy() {
64  
65    }
66  
67    private static String normalizeValue(String value) {
68  
69      if (value != null) {
70        String result = value.trim();
71        if (!result.isEmpty()) {
72          return result;
73        }
74      }
75      return null;
76    }
77  
78    @Override
79    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
80        throws IOException, ServletException {
81  
82      setCorrelationId(request);
83      try {
84        chain.doFilter(request, response);
85      } finally {
86        this.diagnosticContextFacade.removeCorrelationId();
87      }
88    }
89  
90    private void setCorrelationId(ServletRequest request) {
91  
92      String correlationId = null;
93      if (request instanceof HttpServletRequest && this.correlationIdHttpHeaderName != null) {
94        correlationId = normalizeValue(((HttpServletRequest) request).getHeader(this.correlationIdHttpHeaderName));
95        if (correlationId == null) {
96          LOG.debug("No correlation ID found for HTTP header {}.", this.correlationIdHttpHeaderName);
97        } else {
98          this.diagnosticContextFacade.setCorrelationId(correlationId);
99          LOG.debug("Using correlation ID {} from HTTP header {}.", correlationId, this.correlationIdHttpHeaderName);
100         return;
101       }
102     }
103     if (correlationId == null) {
104       // potential fallback if initialized before this filter...
105       correlationId = normalizeValue(this.diagnosticContextFacade.getCorrelationId());
106       if (correlationId != null) {
107         LOG.debug("Correlation ID was already set to {} before DiagnosticContextFilter has been invoked.",
108             correlationId);
109       } else {
110         // no correlation ID present, create a unique ID
111         correlationId = UUID.randomUUID().toString();
112         this.diagnosticContextFacade.setCorrelationId(correlationId);
113         LOG.debug("Created unique correlation ID {}.", correlationId);
114       }
115     }
116   }
117 
118   /**
119    * @param diagnosticContextFacade the diagnosticContextFacade to set
120    */
121   @Inject
122   public void setDiagnosticContextFacade(DiagnosticContextFacade diagnosticContextFacade) {
123 
124     this.diagnosticContextFacade = diagnosticContextFacade;
125   }
126 
127   @Override
128   public void init(FilterConfig config) throws ServletException {
129 
130     String headerName = config.getInitParameter(CORRELATION_ID_HEADER_NAME_PARAM);
131     if (headerName == null) {
132       LOG.debug("Parameter {} not configured via filter config.", CORRELATION_ID_HEADER_NAME_PARAM);
133     } else {
134       this.correlationIdHttpHeaderName = headerName;
135     }
136     LOG.info("Correlation ID header initialized to: {}", this.correlationIdHttpHeaderName);
137     if (this.diagnosticContextFacade == null) {
138       try {
139         // ATTENTION: We do not import these classes as we keep spring as an optional dependency.
140         // If spring is not available in your classpath (e.g. some real JEE context) then this will produce a
141         // ClassNotFoundException and use the fallback in the catch statement.
142         ServletContext servletContext = config.getServletContext();
143         org.springframework.web.context.WebApplicationContext springContext;
144         springContext =
145             org.springframework.web.context.support.WebApplicationContextUtils.getWebApplicationContext(servletContext);
146         this.diagnosticContextFacade = springContext.getBean(DiagnosticContextFacade.class);
147       } catch (Throwable e) {
148         LOG.warn("DiagnosticContextFacade not defined in spring. Falling back to default", e);
149         this.diagnosticContextFacade = new DiagnosticContextFacadeImpl();
150       }
151     }
152   }
