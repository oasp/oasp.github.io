39  /**
40   * This is an implementation of {@link ExceptionMapper} that acts as generic exception facade for REST services. It
41   * {@link #toResponse(Throwable) maps} {@link Throwable exceptions} to an according HTTP status code and JSON result as
42   * defined by <a href="https://github.com/oasp-forge/oasp4j-wiki/wiki/guide-rest#error-results">OASP REST error
43   * specification</a>.
44   *
45   * @since 2.0.0
46   */
47  @Provider
48  public class RestServiceExceptionFacade implements ExceptionMapper<Throwable> {

108    * Registers a {@link Class} as a top-level security {@link Throwable exception}. Instances of this class and all its
109    * subclasses will be handled as security errors. Therefore an according HTTP error code is used and no further
110    * details about the exception is send to the client to prevent
111    * <a href="https://www.owasp.org/index.php/Top_10_2013-A6-Sensitive_Data_Exposure">sensitive data exposure</a>.

117     this.securityExceptions.add(securityException);

131         "org.springframework.security.authentication.AuthenticationCredentialsNotFoundException");

133     registerToplevelSecurityExceptions("org.springframework.security.authentication.AccountExpiredException");

147   private Class<? extends Throwable> loadException(String className) {
148 
149     try {
150       @SuppressWarnings("unchecked")
151       Class<? extends Throwable> exception = (Class<? extends Throwable>) Class.forName(className);
152       return exception;
153     } catch (ClassNotFoundException e) {
154       LOG.info("Exception {} was not found on classpath and can not be handled by this {}.", className,
155           getClass().getSimpleName());
156     } catch (Exception e) {
157       LOG.error("Exception {} is invalid and can not be handled by this {}.", className, getClass().getSimpleName(), e);
158     }
159     return null;
160   }

162   @Override
163   public Response toResponse(Throwable exception) {
164 
165     if (exception instanceof WebApplicationException) {
166       return createResponse((WebApplicationException) exception);
167     } else if (exception instanceof NlsRuntimeException) {
168       return toResponse(exception, exception);
169     } else {
170       Throwable error = exception;
171       Throwable catched = exception;
172       error = getRollbackCause(exception);
173       if (error == null) {
174         error = unwrapNlsUserError(exception);
175       }
176       if (error == null) {
177         error = exception;
178       }
179       return toResponse(error, catched);
180     }
181   }
182 
183   /**
184    * Unwraps potential NLS user error from a wrapper exception such as {@code JsonMappingException} or
185    * {@code PersistenceException}.
186    *
187    * @param exception the exception to unwrap.
188    * @return the unwrapped {@link NlsRuntimeException} exception or {@code null} if no
189    *         {@link NlsRuntimeException#isForUser() use error}.
190    */
191   private NlsRuntimeException unwrapNlsUserError(Throwable exception) {
192 
193     Throwable cause = exception.getCause();
194     if (cause instanceof NlsRuntimeException) {
195       NlsRuntimeException nlsError = (NlsRuntimeException) cause;
196       if (nlsError.isForUser()) {
197         return nlsError;
198       }
199     }
200     return null;
201   }
202 
203   private Throwable getRollbackCause(Throwable exception) {
204 
205     Class<?> exceptionClass = exception.getClass();
206     if (exceptionClass == this.transactionSystemException) {
207       Throwable cause = exception.getCause();
208       if (cause != null) {
209         exceptionClass = cause.getClass();
210         if (exceptionClass == this.rollbackException) {
211           return cause.getCause();
212         }
213       }
214     }
215     return null;
216   }

218   /**
219    * @see #toResponse(Throwable)
220    *
221    * @param exception the exception to handle
222    * @param catched the original exception that was cached. Either same as {@code error} or a (child-)
223    *        {@link Throwable#getCause() cause} of it.
224    * @return the response build from the exception.
225    */
226   protected Response toResponse(Throwable exception, Throwable catched) {
227 
228     if (exception instanceof ValidationException) {
229       return handleValidationException(exception, catched);
230     } else if (exception instanceof ValidationErrorUserException) {
231       return createResponse(exception, (ValidationErrorUserException) exception, null);
232     } else {
233       Class<?> exceptionClass = exception.getClass();
234       for (Class<?> securityError : this.securityExceptions) {
235         if (securityError.isAssignableFrom(exceptionClass)) {
236           return handleSecurityError(exception, catched);
237         }
238       }
239       return handleGenericError(exception, catched);
240     }
241   }

243   /**
244    * Creates the {@link Response} for the given validation exception.
245    *
246    * @param exception is the original validation exception.
247    * @param error is the wrapped exception or the same as <code>exception</code>.
248    * @param errorsMap is a map with all validation errors
249    * @return the requested {@link Response}.
250    */
251   protected Response createResponse(Throwable exception, ValidationErrorUserException error,
252       Map<String, List<String>> errorsMap) {
253 
254     LOG.warn("Service failed due to validation failure.", error);
255     if (exception == error) {
256       return createResponse(Status.BAD_REQUEST, error, errorsMap);
257     } else {
258       return createResponse(Status.BAD_REQUEST, error, exception.getMessage(), errorsMap);
259     }
260   }

262   /**
263    * Exception handling for generic exception (fallback).
264    *
265    * @param exception the exception to handle
266    * @param catched the original exception that was cached. Either same as {@code error} or a (child-)
267    *        {@link Throwable#getCause() cause} of it.
268    * @return the response build from the exception
269    */
270   protected Response handleGenericError(Throwable exception, Throwable catched) {
271 
272     NlsRuntimeException userError;
273     boolean logged = false;
274     if (exception instanceof NlsThrowable) {
275       NlsThrowable nlsError = (NlsThrowable) exception;
276       if (!nlsError.isTechnical()) {
277         LOG.warn("Service failed due to business error: {}", nlsError.getMessage());
278         logged = true;
279       }
280       userError = TechnicalErrorUserException.getOrCreateUserException(exception);
281     } else {
282       userError = TechnicalErrorUserException.getOrCreateUserException(catched);
283     }
284     if (!logged) {
285       LOG.error("Service failed on server", userError);
286     }
287     return createResponse(userError);
288   }

290   /**
291    * Exception handling for security exception.
292    *
293    * @param exception the exception to handle
294    * @param catched the original exception that was cached. Either same as {@code error} or a (child-)
295    *        {@link Throwable#getCause() cause} of it.
296    * @return the response build from exception
297    */
298   protected Response handleSecurityError(Throwable exception, Throwable catched) {
299 
300     NlsRuntimeException error;
301     if ((exception == catched) && (exception instanceof NlsRuntimeException)) {
302       error = (NlsRuntimeException) exception;
303     } else {
304       error = new SecurityErrorUserException(catched);
305     }
306     LOG.error("Service failed due to security error", error);
307     // NOTE: for security reasons we do not send any details about the error to the client!
308     String message;
309     String code = null;
310     if (this.exposeInternalErrorDetails) {
311       message = getExposedErrorDetails(error);
312     } else {
313       message = "forbidden";
314     }
315     return createResponse(Status.FORBIDDEN, message, code, error.getUuid(), null);
316   }

318   /**
319    * Exception handling for validation exception.
320    *
321    * @param exception the exception to handle
322    * @param catched the original exception that was cached. Either same as {@code error} or a (child-)
323    *        {@link Throwable#getCause() cause} of it.
324    * @return the response build from the exception.
325    */
326   protected Response handleValidationException(Throwable exception, Throwable catched) {
327 
328     Throwable t = catched;
329     Map<String, List<String>> errorsMap = null;
330     if (exception instanceof ConstraintViolationException) {
331       ConstraintViolationException constraintViolationException = (ConstraintViolationException) exception;
332       Set<ConstraintViolation<?>> violations = constraintViolationException.getConstraintViolations();
333       errorsMap = new HashMap<>();
334 
335       for (ConstraintViolation<?> violation : violations) {
336         Iterator<Node> it = violation.getPropertyPath().iterator();
337         String fieldName = null;
338 
339         // Getting fieldname from the exception
340         while (it.hasNext()) {
341           fieldName = it.next().toString();
342         }
343 
344         List<String> errorsList = errorsMap.get(fieldName);
345 
346         if (errorsList == null) {
347           errorsList = new ArrayList<>();
348           errorsMap.put(fieldName, errorsList);
349         }
350 
351         errorsList.add(violation.getMessage());
352 
353       }
354 
355       t = new ValidationException(errorsMap.toString(), catched);
356     }
357     ValidationErrorUserException error = new ValidationErrorUserException(t);
358     return createResponse(t, error, errorsMap);
359   }

361   /**
362    * @param error is the {@link Throwable} to extract message details from.
363    * @return the exposed message(s).
364    */
365   protected String getExposedErrorDetails(Throwable error) {
366 
367     StringBuilder buffer = new StringBuilder();
368     Throwable e = error;
369     while (e != null) {
370       if (buffer.length() > 0) {
371         buffer.append(StringUtil.LINE_SEPARATOR);
372       }
373       buffer.append(e.getClass().getSimpleName());
374       buffer.append(": ");
375       buffer.append(e.getLocalizedMessage());
376       e = e.getCause();
377     }
378     return buffer.toString();
379   }

453    * @param code is the {@link NlsRuntimeException#getCode() error code}.

458   protected Response createResponse(Status status, String message, String code, UUID uuid,
459       Map<String, List<String>> errorsMap) {
460 
461     String json = createJsonErrorResponseMessage(message, code, uuid, errorsMap);
462     return Response.status(status).entity(json).build();
463   }

465   /**
466    * Create a response message as a JSON-String from the given parts.
467    *
468    * @param message the message of the response message
469    * @param code the code of the response message
470    * @param uuid the uuid of the response message
471    * @param errorsMap is a map with all validation errors
472    * @return the response message as a JSON-String
473    */
474   protected String createJsonErrorResponseMessage(String message, String code, UUID uuid,
475       Map<String, List<String>> errorsMap) {
476 
477     Map<String, Object> jsonMap = new HashMap<>();
478     if (message != null) {
479       jsonMap.put(ServiceConstants.KEY_MESSAGE, message);
480     }
481     if (code != null) {
482       jsonMap.put(ServiceConstants.KEY_CODE, code);
483     }
484     if (uuid != null) {
485       jsonMap.put(ServiceConstants.KEY_UUID, uuid.toString());
486     }
487 
488     if (errorsMap != null) {
489       jsonMap.put(ServiceConstants.KEY_ERRORS, errorsMap);
490     }
491 
492     String responseMessage = "";
493     try {
494       responseMessage = this.mapper.writeValueAsString(jsonMap);
495     } catch (JsonProcessingException e) {
496       LOG.error("Exception facade failed to create JSON.", e);
497       responseMessage = "{}";
498     }
499     return responseMessage;
500 
501   }

503   /**
504    * Add a response message to an existing response.
505    *
506    * @param exception the {@link WebApplicationException}.
507    * @return the response with the response message added
508    */
509   protected Response createResponse(WebApplicationException exception) {
510 
511     Response response = exception.getResponse();
512     int statusCode = response.getStatus();
513     Status status = Status.fromStatusCode(statusCode);
514     NlsRuntimeException error;
515     if (exception instanceof ServerErrorException) {
516       error = new TechnicalErrorUserException(exception);
517       LOG.error("Service failed on server", error);
518       return createResponse(status, error, null);
519     } else {
520       UUID uuid = UUID.randomUUID();
521       if (exception instanceof ClientErrorException) {
522         LOG.warn("Service failed due to unexpected request. UUDI: {}, reason: {} ", uuid, exception.getMessage());
523       } else {
524         LOG.warn("Service caused redirect or other error. UUID: {}, reason: {}", uuid, exception.getMessage());
525       }
526       return createResponse(status, exception.getMessage(), String.valueOf(statusCode), uuid, null);
527     }

548   /**
549    * @param exposeInternalErrorDetails - {@code true} if internal exception details shall be exposed to clients (useful
550    *        for debugging and testing), {@code false} if such details are hidden to prevent
551    *        <a href="https://www.owasp.org/index.php/Top_10_2013-A6-Sensitive_Data_Exposure">Sensitive Data Exposure</a>
552    *        (default, has to be used in production environment).
553    */
554   public void setExposeInternalErrorDetails(boolean exposeInternalErrorDetails) {
555 
556     this.exposeInternalErrorDetails = exposeInternalErrorDetails;
557     if (exposeInternalErrorDetails) {
558       String message =
559           "****** Exposing of internal error details is enabled! This violates OWASP A6 (Sensitive Data Exposure) and shall only be used for testing/debugging and never in production. ******";
560       LOG.warn(message);
561       // CHECKSTYLE:OFF (for development only)
562       System.err.println(message);
563       // CHECKSTYLE:ON
564     }
565   }
