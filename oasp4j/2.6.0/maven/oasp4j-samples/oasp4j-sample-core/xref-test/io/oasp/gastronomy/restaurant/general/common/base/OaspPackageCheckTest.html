23    /**
24     * Scans all the packages of this application root pacakge namespace. Will verify that these are
25     * {@link OaspPackage#isValid() valid OASP packages}.
26     */
27    @Test
28    public void testPackages() {
29  
30      OaspPackage pkg = OaspPackage.of(OaspPackageCheckTest.class);
31      assertThat(pkg.isValid()).isTrue();
32  
33      ReflectionUtil ru = ReflectionUtilImpl.getInstance();
34      Set<String> classNames = ru.findClassNames(getRootPackage2Scan(pkg), true);
35      String appPackage = pkg.getRoot() + "." + pkg.getApplication();
36      Set<String> packages = new HashSet<>(128);
37      packages.add(appPackage); // allow SpringBootApp, etc. in application package
38      SoftAssertions assertion = new SoftAssertions();
39      for (String className : classNames) {
40        int lastDot = className.lastIndexOf('.');
41        if (lastDot > 0) {
42          String packageName = className.substring(0, lastDot);
43          boolean added = packages.add(packageName);
44          if (added) {
45            pkg = OaspPackage.of(packageName);
46            if (!pkg.isValid()) {
47              assertion.assertThat(pkg.isValid())
48                  .as("package " + packageName + " is invalid (component: " + pkg.getComponent() + ", layer: "
49                      + pkg.getLayer() + ", scope: " + pkg.getScope() + "). Hint contains e.g. "
50                      + className.substring(lastDot + 1))
51                  .isTrue();
52            }
53          }
54        }
55      }
56      assertion.assertAll();
57    }
58  
59    /**
60     * @param pkg the {@link OaspPackage} of this test.
61     * @return the root package to scan for {@link Class}es to get the actual packages to check.
62     */
63    protected String getRootPackage2Scan(OaspPackage pkg) {
64  
65      String root = pkg.getRoot();
66      if (root.startsWith("io.oasp")) {
67        return "io.oasp";
68      }
69      return root;
70    }
