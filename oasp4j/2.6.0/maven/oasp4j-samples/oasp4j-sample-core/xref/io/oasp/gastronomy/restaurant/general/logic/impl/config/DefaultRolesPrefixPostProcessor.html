1   package io.oasp.gastronomy.restaurant.general.logic.impl.config;
2   
3   import org.springframework.beans.BeansException;
4   import org.springframework.beans.factory.config.BeanPostProcessor;
5   import org.springframework.core.PriorityOrdered;
6   import org.springframework.security.access.annotation.Jsr250MethodSecurityMetadataSource;
7   import org.springframework.security.access.expression.method.DefaultMethodSecurityExpressionHandler;
8   import org.springframework.security.web.access.expression.DefaultWebSecurityExpressionHandler;
9   import org.springframework.security.web.servletapi.SecurityContextHolderAwareRequestFilter;
10  
11  /**
12   * This is an implementation of {@link BeanPostProcessor} that allows to change the role prefix of spring-security. By
13   * default spring-security is magically adding a strange prefix called "ROLE_" to your granted authorities. In order to
14   * prevent this we use this class with an empty prefix.
15   */
16  public class DefaultRolesPrefixPostProcessor implements BeanPostProcessor, PriorityOrdered {
17  
18    private final String rolePrefix;
19  
20    /**
21     * Der Konstruktor.
22     *
23     * @param rolePrefix das gewünschte Rollen-Präfix (z.B. der leere String).
24     */
25    public DefaultRolesPrefixPostProcessor(String rolePrefix) {
26      super();
27      this.rolePrefix = rolePrefix;
28    }
29  
30    @Override
31    public Object postProcessAfterInitialization(Object bean, String beanName) throws BeansException {
32  
33      // remove this if you are not using JSR-250
34      if (bean instanceof Jsr250MethodSecurityMetadataSource) {
35        ((Jsr250MethodSecurityMetadataSource) bean).setDefaultRolePrefix(this.rolePrefix);
36      }
37  
38      if (bean instanceof DefaultMethodSecurityExpressionHandler) {
39        ((DefaultMethodSecurityExpressionHandler) bean).setDefaultRolePrefix(this.rolePrefix);
40      }
41      if (bean instanceof DefaultWebSecurityExpressionHandler) {
42        ((DefaultWebSecurityExpressionHandler) bean).setDefaultRolePrefix(this.rolePrefix);
43      }
44      if (bean instanceof SecurityContextHolderAwareRequestFilter) {
45        ((SecurityContextHolderAwareRequestFilter) bean).setRolePrefix(this.rolePrefix);
46      }
47      return bean;
48    }
49  
50    @Override
51    public Object postProcessBeforeInitialization(Object bean, String beanName) throws BeansException {
52  
53      return bean;
54    }
55  
56    @Override
57    public int getOrder() {
58  
59      return PriorityOrdered.HIGHEST_PRECEDENCE;
60    }
61  }
