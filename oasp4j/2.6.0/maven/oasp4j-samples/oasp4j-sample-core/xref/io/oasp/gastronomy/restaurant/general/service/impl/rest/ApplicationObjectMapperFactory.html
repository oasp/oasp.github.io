17  /**
18   * The MappingFactory class to resolve polymorphic conflicts within the restaurant application.
19   */
20  @Named("ApplicationObjectMapperFactory")
21  public class ApplicationObjectMapperFactory extends ObjectMapperFactory {
22  
23    /**
24     * The constructor.
25     */
26    public ApplicationObjectMapperFactory() {
27  
28      super();
29      // register polymorphic base classes
30      // BEGIN ARCHETYPE SKIP
31      setBaseClasses(ProductEto.class);
32      // END ARCHETYPE SKIP
33  
34      NamedType[] subtypes;
35      // register mapping for polymorphic sub-classes
36      // BEGIN ARCHETYPE SKIP
37      subtypes = new NamedType[] { new NamedType(MealEto.class, "Meal"), new NamedType(DrinkEto.class, "Drink"),
38      new NamedType(SideDishEto.class, "SideDish") };
39      setSubtypes(subtypes);
40  
41      // register (de)serializers for custom datatypes
42      SimpleModule module = getExtensionModule();
43      module.addDeserializer(Money.class, new MoneyJsonDeserializer());
44      module.addSerializer(Money.class, new MoneyJsonSerializer());
45      // END ARCHETYPE SKIP
46    }
47  }
