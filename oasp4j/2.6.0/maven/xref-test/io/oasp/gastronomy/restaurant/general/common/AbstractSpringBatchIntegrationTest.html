28  /**
29   * Base class for all spring batch integration tests. It helps to do End-to-End job tests.
30   *
31   */
32  // @DirtiesContext(classMode = ClassMode.AFTER_CLASS)
33  public abstract class AbstractSpringBatchIntegrationTest extends ComponentTest {
34    private static final Logger LOG = LoggerFactory.getLogger(AbstractSpringBatchIntegrationTest.class);
35  
36    /** directory for temporary test files */
37    private static final String TMP_DIR = "./tmp";
38  
39    /** scripts for all tests db setup */
40    private static final String ALL_TESTS_DB_SETUP_DIR = "classpath:AllTests/setup/db";
41  
42    protected static void login(String login, String password, String... permissions) {
43    
44      Set<String> groups = new HashSet<>(Arrays.asList(permissions));
45    
46      Set<GrantedAuthority> authorities = new HashSet<>();
47      for (String permission : groups) {
48        authorities.add(new AccessControlGrantedAuthority(new AccessControlPermission(permission)));
49      }
50      SecurityContextHolder.getContext().setAuthentication(
51          new UsernamePasswordAuthenticationToken(new UserData(login, password, authorities), password));
52    }
53  
54    public static void logout() {
55    
56      SecurityContextHolder.getContext().setAuthentication(null);
57    }
58  
59    /**
60     * database migration helper
61     */
62    @Inject
63    protected DatabaseMigrator flyway;
64  
65    @Inject
66    private JobLauncher jobLauncher;
67  
68    /**
69     * @param job job to configure
70     * @return jobLauncherTestUtils
71     */
72    public JobLauncherTestUtils getJobLauncherTestUtils(Job job) {
73  
74      JobLauncherTestUtils jobLauncherTestUtils = new JobLauncherTestUtils();
75      jobLauncherTestUtils.setJob(job);
76      jobLauncherTestUtils.setJobLauncher(this.jobLauncher);
77  
78      return jobLauncherTestUtils;
79    }
80  
81    /**
82     * setup run before all tests
83     *
84     * @throws Exception throw by FileUtils
85     */
86    @Before
87    public void setup() throws Exception {
88  
89      // setup test data
90      this.flyway.importTestData(ALL_TESTS_DB_SETUP_DIR);
91  
92      LOG.debug("Delete tmp directory");
93      FileUtils.deleteDirectory(new File(TMP_DIR));
94    }
95  }
