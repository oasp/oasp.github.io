40    /**
41     * Test behavior when passing an empty list.
42     */
43    @Test
44    public void sumTestEmptyList() {
45  
46      List<Money> moneyList = new ArrayList<>();
47      Money summedAmount = MoneyHelper.sumMoneys(moneyList);
48      assertThat(MoneyHelper.ZERO_MONEY).isEqualTo(summedAmount);
49    }

51    /**
52     * Test for summing up {@link Money}s with default currency (EUR).
53     */
54    @Test
55    public void sumTestEURList() {
56  
57      List<Money> moneyList = new ArrayList<>();
58  
59      double[] amounts = { 15.28, 11.47, 4.20 };
60      double sum = 0.0;
61  
62      for (double amount : amounts) {
63        moneyList.add(MoneyHelper.getMoneyWithDefaultCurrency(amount));
64        sum += amount;
65      }
66  
67      Money summedAmount = MoneyHelper.sumMoneys(moneyList);
68  
69      assertThat(MoneyHelper.getMoney(sum, "EUR")).isEqualTo(summedAmount);
70    }
