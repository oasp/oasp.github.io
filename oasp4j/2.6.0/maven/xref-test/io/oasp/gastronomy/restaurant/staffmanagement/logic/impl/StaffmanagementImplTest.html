18  /**
19   * This class provides a basic implementation of a {@code ModuleTest} which employs the Mockito framework to provide
20   * fake objects to the SUT. For a limited number of test cases per class it can be convenient to use the annotation
21   * {@code @InjectMocks} to inject all needed mocks in the class under test.
22   *
23   */
24  
25  public class StaffmanagementImplTest extends ModuleTest {
26  
27    /**
28     * Initializes mocks before each test method.
29     */
30    @Rule
31    public MockitoRule rule = MockitoJUnit.rule();
32  
33    @Mock
34    private BeanMapper beanMapper;
35  
36    @Mock
37    private StaffMemberEntity staffMemberEntity;
38  
39    @Mock
40    private StaffMemberEto staffMemberEto;
41  
42    @Mock
43    private StaffMemberDao staffMemberDao;
44  
45    @InjectMocks
46    StaffmanagementImpl staffmanagementImpl = new StaffmanagementImpl();
47  
48    /**
49     * This test method tests the {@code findStaffMember} method of {@link StaffmanagementImpl} by mocking needed
50     * dependencies.
51     */
52    @Test
53    public void testFindStaffMember() {
54  
55      // given
56      long id = 1L;
57      Class<StaffMemberEto> targetClass = StaffMemberEto.class;
58  
59      when(this.staffMemberDao.find(id)).thenReturn(this.staffMemberEntity);
60      when(this.beanMapper.map(this.staffMemberEntity, targetClass)).thenReturn(this.staffMemberEto);
61  
62      // when
63      StaffMemberEto resultEto = this.staffmanagementImpl.findStaffMember(id);
64  
65      // then
66      assertThat(resultEto).isNotNull();
67      assertThat(resultEto).isEqualTo(this.staffMemberEto);
68    }
69  
70    /**
71     * This test method updates a saved staffMemberEntity. It demonstrates the approach when there is a method call in the
72     * method under test. To solve this in the given case you need to override the {@code @Inject} of the tested method by
73     * using the appropriate setter {@code setStaffMemberDao()}.
74     */
75    @Test
76    public void testSaveStaffMember() {
77  
78      // given
79      long id = 1L;
80  
81      when(this.staffMemberEto.getId()).thenReturn(id);
82      when(this.staffMemberDao.find(id)).thenReturn(this.staffMemberEntity);
83      when(this.staffMemberEto.getName()).thenReturn("new Name");
84      when(this.staffMemberEntity.getName()).thenReturn("old Name");
85      when(this.beanMapper.map(this.staffMemberEto, StaffMemberEntity.class)).thenReturn(this.staffMemberEntity);
86      when(this.staffMemberDao.save(this.staffMemberEntity)).thenReturn(this.staffMemberEntity);
87      when(this.beanMapper.map(this.staffMemberEntity, StaffMemberEto.class)).thenReturn(this.staffMemberEto);
88  
89      // when
90      this.staffmanagementImpl.setStaffMemberDao(this.staffMemberDao);
91      StaffMemberEto resultEto = this.staffmanagementImpl.saveStaffMember(this.staffMemberEto);
92  
93      // then
94      assertThat(resultEto.getId()).isEqualTo(this.staffMemberEto.getId());
95    }
