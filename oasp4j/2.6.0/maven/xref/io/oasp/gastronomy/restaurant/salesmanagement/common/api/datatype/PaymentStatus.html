6   public enum PaymentStatus {
7   
8       /**
9        * This status describes a successful payment transaction.
10       */
11      SUCCESS,
12  
13      /**
14       * This status describes a unsuccessful payment transaction. The input data is not correct.
15       */
16      INPUT_DATA_ERROR,
17  
18      /**
19       * This status describes a unsuccessful payment transaction. The connection timed out.
20       */
21      TIME_OUT_ERROR;
22  
23      /**
24       * @return {@code true}, if the {@link PaymentStatus} equals {@link PaymentStatus#SUCCESS}.
25       *         {@code false} otherwise.
26       */
27      public boolean isSuccessful() {
28          return this == SUCCESS;
29      }
30  
31      /**
32       * 
33       * @return {@code true}, if the {@link PaymentStatus} equals {@link PaymentStatus#INPUT_DATA_ERROR}
34       *         or {@link PaymentStatus#TIME_OUT_ERROR}. {@code false} otherwise.
35       */
36      public boolean isUnsuccessful() {
37          return (this == INPUT_DATA_ERROR || this == TIME_OUT_ERROR);
38      }
39  }
