21  /**
22   *
23   * The service class for REST calls in order to execute the methods in {@link Staffmanagement}.
24   *
25   */
26  @Path("/staffmanagement/v1/staff")
27  @Consumes(MediaType.APPLICATION_JSON)
28  @Produces(MediaType.APPLICATION_JSON)
29  public interface StaffmanagementRestService extends RestService {
30  
31    /**
32     * @return a list of all {@link StaffMemberEto}
33     *
34     */
35    @GET
36    @Path("/")
37    @Deprecated
38    List<StaffMemberEto> getAllStaffMember();
39  
40    /**
41     * @param login the login of a staff member
42     * @return {@link StaffMemberEto}
43     */
44    @GET
45    @Path("/{login}")
46    StaffMemberEto getStaffMember(@PathParam("login") String login);
47  
48    /**
49     * @param staffMemberBo the staffMember to be updated as JSON
50     */
51    @PUT
52    @Path("/{login}")
53    @Deprecated
54    void updateStaffMember(StaffMemberEto staffMemberBo);
55  
56    /**
57     * Calls {@link Staffmanagement#saveStaffMember}.
58     *
59     * @param staffMemberEto the staffMember to be created or updated
60     * @return the saved {@link StaffMemberEto}
61     */
62    @POST
63    @Path("/")
64    StaffMemberEto saveStaffMember(StaffMemberEto staffMemberEto);
65  
66    /**
67     * @param login the login
68     */
69    @DELETE
70    @Path("/{login}")
71    void deleteStaffMember(@PathParam("login") String login);
72  
73    /**
74     * Delegates to {@link Staffmanagement#findStaffMemberEtos}.
75     *
76     * @param searchCriteriaTo the pagination and search criteria to be used for finding staffmembers.
77     * @return the {@link PaginatedListTo list} of matching {@link StaffMemberEto}s.
78     */
79    @Path("/search")
80    @POST
81    PaginatedListTo<StaffMemberEto> findStaffMembersByPost(StaffMemberSearchCriteriaTo searchCriteriaTo);

83  }
