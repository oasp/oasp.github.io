3   import java.math.BigDecimal;
4   import java.math.BigInteger;
5   import java.util.Currency;

7   /**
8    * Utility that helps to deal with configuration values.
9    */
10  public final class ConfigValueUtil {
11  
12    private ConfigValueUtil() {
13    }
14  
15    /**
16     * @param <T> the generic {@code type}.
17     * @param object the {@link Object} to convert. Will not be {@code null}.
18     * @param type the {@link Class} reflecting the requested type.
19     * @return the given {@link Object} converted to the given {@code type}.
20     */
21    @SuppressWarnings({ "rawtypes", "unchecked" })
22    public static <T> T convertValue(Object object, Class<T> type) {
23  
24      try {
25        Object result;
26        if (type.isInstance(object)) {
27          result = object;
28        } else if (type.isEnum()) {
29          result = Enum.valueOf((Class<? extends Enum>) type, object.toString());
30        } else if (type.isAssignableFrom(String.class)) {
31          result = object.toString();
32        } else if ((type == boolean.class) || (type == Boolean.class)) {
33          result = Boolean.valueOf(object.toString());
34        } else if ((type == int.class) || (type == Integer.class)) {
35          result = Integer.valueOf(object.toString());
36        } else if ((type == long.class) || (type == Long.class)) {
37          result = Long.valueOf(object.toString());
38        } else if ((type == double.class) || (type == Double.class)) {
39          result = Double.valueOf(object.toString());
40        } else if (type == Class.class) {
41          result = Class.forName(object.toString());
42        } else if ((type == float.class) || (type == Float.class)) {
43          result = Float.valueOf(object.toString());
44        } else if ((type == short.class) || (type == Short.class)) {
45          result = Short.valueOf(object.toString());
46        } else if ((type == byte.class) || (type == Byte.class)) {
47          result = Byte.valueOf(object.toString());
48        } else if (type == BigDecimal.class) {
49          result = new BigDecimal(object.toString());
50        } else if (type == BigInteger.class) {
51          result = new BigInteger(object.toString());
52        } else if (type == Number.class) {
53          result = Double.parseDouble(object.toString());
54        } else if ((type == Character.class) || ((type == char.class))) {
55          String value = object.toString();
56          if (value.length() == 1) {
57            result = Character.valueOf(value.charAt(0));
58          } else {
59            throw new IllegalArgumentException(value);
60          }
61        } else if (type == Currency.class) {
62          result = Currency.getInstance(object.toString());
63        } else {
64          throw new IllegalArgumentException(object.toString());
65        }
66        return (T) result;
67      } catch (NumberFormatException | ClassNotFoundException e) {
68        throw new IllegalArgumentException(object.toString(), e);
69      }
70    }
