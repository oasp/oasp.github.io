1   package io.oasp.module.cxf.common.impl.server;
2   
3   import org.apache.cxf.transport.servlet.CXFServlet;
4   import org.springframework.boot.web.servlet.ServletRegistrationBean;
5   import org.springframework.context.annotation.Bean;
6   import org.springframework.context.annotation.Configuration;
7   import org.springframework.context.annotation.ImportResource;
8   
9   import io.oasp.module.service.common.api.constants.ServiceConstants;
10  
11  /**
12   * Basic {@link Configuration} for Apache CXF server.
13   *
14   * @since 3.0.0
15   */
16  @Configuration
17  @ImportResource({ "classpath:META-INF/cxf/cxf.xml" })
18  public class CxfServerAutoConfiguration {
19  
20    /**
21     * @return the {@link ServletRegistrationBean} to register {@link CXFServlet} according to OASP conventions at
22     *         {@link ServiceConstants#URL_PATH_SERVICES}.
23     */
24    @Bean
25    public ServletRegistrationBean servletRegistrationBean() {
26  
27      CXFServlet cxfServlet = new CXFServlet();
28      ServletRegistrationBean servletRegistration =
29          new ServletRegistrationBean(cxfServlet, ServiceConstants.URL_PATH_SERVICES + "/*");
30      return servletRegistration;
31    }
32  
33  }
