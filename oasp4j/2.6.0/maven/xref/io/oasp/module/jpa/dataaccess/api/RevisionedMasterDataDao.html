7   /**
8    * This is the interface for a {@link Dao} responsible for a {@link PersistenceEntity} that represents master-data. In
9    * that case you typically have a limited number of entities in your persistent store and need operations like
10   * {@link #findAll()}.<br>
11   * <b>ATTENTION:</b><br>
12   * Such operations are not part of {@link GenericDao} or {@link Dao} as invoking them (accidently) could cause that an
13   * extraordinary large number of entities are loaded into main memory and could cause serious performance and stability
14   * disasters. So only extend this interface in case you are aware of what you are doing.
15   *
16   * @param <E> is the generic type of the {@link PersistenceEntity}.
17   *
18   */
19  public interface RevisionedMasterDataDao<E extends MutablePersistenceEntity<Long>> extends GenericRevisionedDao<Long, E> {
20  
21    /**
22     * @return an {@link Iterable} with ALL managed entities from the persistent store. Not exposed to API by default as
23     *         this might not make sense for all kind of entities.
24     */
25    List<E> findAll();
26  
27  }
