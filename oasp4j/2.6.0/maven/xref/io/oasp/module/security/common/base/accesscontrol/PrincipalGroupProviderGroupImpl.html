5   import java.security.Principal;
6   import java.security.acl.Group;
7   import java.util.Collection;
8   import java.util.Enumeration;
9   import java.util.HashSet;
10  import java.util.Set;
11  
12  /**
13   * This is an implementation of {@link PrincipalAccessControlProvider} based on {@link Group}. Due to the confusing API of
14   * {@link Group} that mixes a {@link Principal} with permissions and permission groups it is not commonly used even
15   * though it is available in the Java standard edition.
16   *
17   */
18  public class PrincipalGroupProviderGroupImpl implements PrincipalAccessControlProvider<Group> {
19  
20    /**
21     * The constructor.
22     */
23    public PrincipalGroupProviderGroupImpl() {
24  
25      super();
26    }
27  
28    @Override
29    public Collection<String> getAccessControlIds(Group principal) {
30  
31      Set<String> groupSet = new HashSet<>();
32      collectGroups(principal, groupSet);
33      return groupSet;
34    }
35  
36    /**
37     * Called from {@link #getAccessControlIds(Group)} to recursively collect the groups.
38     *
39     * @param group is the {@link Group} to traverse.
40     * @param groupSet is the {@link Set} where to add the principal names.
41     */
42    protected void collectGroups(Group group, Set<String> groupSet) {
43  
44      Enumeration<? extends Principal> members = group.members();
45      while (members.hasMoreElements()) {
46        Principal member = members.nextElement();
47        String name = member.getName();
48        boolean added = groupSet.add(name);
49        if (added && (member instanceof Group)) {
50          collectGroups((Group) member, groupSet);
51        }
52      }
53    }
54  }
