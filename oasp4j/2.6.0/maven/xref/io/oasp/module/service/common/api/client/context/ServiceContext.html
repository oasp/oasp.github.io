3   import java.util.Collection;

5   import io.oasp.module.basic.common.api.config.ConfigProperties;
6   
7   /**
8    * This interface gives read access to contextual information of a {@link io.oasp.module.service.common.api.Service}.
9    *
10   * @param <S> the generic type of the {@link #getApi() service API}.
11   *
12   * @since 3.0.0
13   */
14  public interface ServiceContext<S> {
15  
16    /**
17     * @return the {@link Class} reflecting the API of the {@link io.oasp.module.service.common.api.Service}.
18     */
19    Class<S> getApi();
20  
21    /**
22     * @return the URL (or URI) of the remote service.
23     */
24    String getUrl();
25  
26    /**
27     * @return a {@link Collection} with the available {@link #getHeader(String) header} names (keys).
28     */
29    Collection<String> getHeaderNames();
30  
31    /**
32     * @param name the name (key) of the header to get.
33     * @return the value of the requested header or {@code null} if undefined.
34     */
35    String getHeader(String name);
36  
37    /**
38     * @return the {@link ConfigProperties} with configuration metadata.
39     * @see io.oasp.module.service.common.api.client.ServiceClientFactory#create(Class, java.util.Map)
40     */
41    ConfigProperties getConfig();
42  
43  }
