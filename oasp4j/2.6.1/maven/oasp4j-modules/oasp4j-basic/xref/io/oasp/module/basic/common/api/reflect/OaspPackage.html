8   /**
9    * This class represents a {@link Package} following the
10   * <a href="https://github.com/oasp/oasp4j/wiki/coding-conventions#packages">OASP coding convetion</a>. <br>
11   * After parsing a {@link Package} as {@link OaspPackage} you can get individual parts/segments such as
12   * {@link #getComponent() compoent}, {@link #getLayer() layer}, {@link #getScope() scope}, etc.<br>
13   * This is useful for advanced features and tools such as service clients and exception facades, code-generators,
14   * static-code-analyzers (SonarQube plugin), etc.
15   *
16   * @see #of(String)
17   * @see #of(Package)
18   * @see #of(Class)
19   *
20   * @author hohwille
21   */
22  public final class OaspPackage {

79    /**
80     * Der Konstruktor.
81     *
82     * @param segments - see {@link #getSegment(int)}.
83     * @param scope - see {@link #getScope()}.
84     */
85    private OaspPackage(String pkg, String[] segments, String root, String detail, int scope) {
86      super();
87      Objects.requireNonNull(segments, "segments");
88      this.pkg = pkg;
89      for (int i = 0; i < segments.length; i++) {
90        if (!isValidSegment(segments[i])) {
91          throw new IllegalArgumentException("segments[" + i + "] = " + segments[i]);
92        }
93      }
94      this.root = root;
95      this.detail = detail;
96      this.segments = segments;
97      this.scopeIndex = scope;
98    }

100   private static boolean isValidSegment(String segment) {
101 
102     if (segment == null) {
103       return false;
104     }
105     if (segment.isEmpty()) {
106       return false;
107     }
108     if (segment.indexOf('.') >= 0) {
109       return false;
110     }
111     return true;
112   }

122   /**
123    * @param index the position of the requested segment. A valid index is in the range from {@code 0} to <code>

125    * @return the {@link Package} segment at the given index or {@code null} if the given index is invalid.
126    */
127   public String getSegment(int index) {
128 
129     if ((index >= 0) && (index < this.segments.length)) {
130       return this.segments[index];
131     }
132     return null;
133   }

135   /**
136    * @return {@code true} if this {@link OaspPackage} is a valid according to OASP
137    *         <a href="https://github.com/oasp/oasp4j/wiki/coding-conventions#packages">package conventions"</a>,
138    *         {@code false} otherwise.
139    */
140   public boolean isValid() {
141 
142     if (this.valid == null) {
143       this.valid = Boolean.valueOf(isValidInternal());
144     }
145     return this.valid;
146   }

148   private boolean isValidInternal() {
149 
150     if (this.segments.length < 4) {
151       return false;
152     }
153     if (!isValidLayer()) {
154       return false;
155     }
156     if (!isValidScope()) {
157       return false;
158     }
159     return true;
160   }

162   /**
163    * @return {@code true} if the {@link #getScope() scope} is valid (one of the predefined scopes {@link #isScopeApi()
164    *         api}, {@link #isScopeBase() base}, or {@link #isScopeImpl() impl}).
165    */
166   public boolean isValidScope() {
167 
168     return SCOPES.contains(getScope());
169   }

171   /**
172    * @return {@code true} if the {@link #getLayer() layer} is valid (one of the predefined scopes {@link #isLayerBatch()
173    *         batch}, {@link #isLayerClient() client}, {@link #isLayerCommon() common}, {@link #isLayerDataAccess()
174    *         dataaccess}, {@link #isLayerLogic() logic}, or {@link #isLayerService() service}).
175    */
176   public boolean isValidLayer() {
177 
178     return LAYERS.contains(getLayer());
179   }

181   /**
182    * @return the root-{@link Package} of the organization or IT project owning the code.
183    */
184   public String getRoot() {
185 
186     if (this.root == null) {
187       if (this.scopeIndex == -1) {
188         return this.pkg;
189       }
190       int appIndex = this.scopeIndex - 3;
191       if (appIndex <= 0) {
192         return null;
193       }
194       StringBuilder sb = new StringBuilder();
195       for (int i = 0; i < appIndex; i++) {
196         if (i > 0) {
197           sb.append('.');
198         }
199         sb.append(this.segments[i]);
200       }
201       this.root = sb.toString();
202     }
203     return this.root;
204   }

310   /**
311    * @return the optional detail. Can be a single segment or multiple segments separated with dot. May be {@code null}.
312    */
313   public String getDetail() {
314 
315     if (this.detail == null) {
316       if (this.scopeIndex < 3) {
317         return null;
318       }
319       this.detail = joinPackage(this.scopeIndex + 1);
320     }
321     return this.detail;
322   }

330   @Override
331   public boolean equals(Object obj) {
332 
333     if (obj == this) {
334       return true;
335     }
336     if ((obj == null) || (obj.getClass() != OaspPackage.class)) {
337       return false;
338     }
339     OaspPackage other = (OaspPackage) obj;
340     if (!Arrays.deepEquals(this.segments, other.segments)) {
341       return false;
342     }
343     if (this.scopeIndex != other.scopeIndex) {
344       return false;
345     }
346     return true;
347   }

358   private String joinPackage(int start) {
359 
360     return joinPackage(start, this.segments.length);
361   }
362 
363   private String joinPackage(int start, int end) {
364 
365     if (start >= end) {
366       return null;
367     }
368     StringBuilder sb = new StringBuilder();
369     for (int i = start; i < end; i++) {
370       if (i > start) {
371         sb.append('.');
372       }
373       sb.append(this.segments[i]);
374     }
375     return sb.toString();
376   }

378   /**
379    * @param root - see {@link #getRoot()}.
380    * @param application - see {@link #getApplication()}.
381    * @param component - see {@link #getComponent()}.
382    * @param layer - see {@link #getLayer()}.
383    * @param scope - see {@link #getScope()}.
384    * @param detail - see {@link #getDetail()}.
385    * @return the {@link OaspPackage} for the given parameters.
386    */
387   public static OaspPackage of(String root, String application, String component, String layer, String scope,
388       String detail) {
389 
390     String[] roots;
391     if (root == null) {
392       roots = new String[0];
393     } else {
394       roots = root.split(REGEX_PKG_SEPARATOR);
395     }
396     String[] details;
397     if (detail == null) {
398       details = new String[0];
399     } else {
400       details = detail.split(REGEX_PKG_SEPARATOR);
401     }
402     String[] segments = new String[roots.length + details.length + 4];
403     System.arraycopy(roots, 0, segments, 0, roots.length);
404     int i = roots.length;
405     segments[i++] = application;
406     segments[i++] = component;
407     segments[i++] = layer;
408     segments[i++] = scope;
409     System.arraycopy(details, 0, segments, i, details.length);
410     return new OaspPackage(null, segments, root, detail, (i - 1));
411   }

413   /**
414    * @param packageName the {@link Package#getName() package name} to parse.
415    * @return the parsed {@link OaspPackage} corresponding to the given package.
416    */
417   public static OaspPackage of(String packageName) {
418 
419     String[] segments = packageName.split(REGEX_PKG_SEPARATOR);
420     int scopeIndex = -1;
421     for (int i = 2; i < segments.length; i++) {
422       if (SCOPES.contains(segments[i])) {
423         scopeIndex = i;
424         break;
425       }
426       if (LAYERS.contains(segments[i])) {
427         scopeIndex = i + 1;
428       }
429     }
430     return new OaspPackage(packageName, segments, null, null, scopeIndex);
431   }
