3   package io.oasp.module.jpa.dataaccess.base;
4   
5   import java.util.ArrayList;
6   import java.util.List;
7   
8   import net.sf.mmm.util.exception.api.ObjectNotFoundException;
9   
10  import org.hibernate.envers.AuditReader;
11  import org.hibernate.envers.AuditReaderFactory;
12  
13  import io.oasp.module.jpa.dataaccess.api.GenericRevisionedDao;
14  import io.oasp.module.jpa.dataaccess.api.MutablePersistenceEntity;
15  import io.oasp.module.jpa.dataaccess.api.RevisionMetadata;
16  import io.oasp.module.jpa.dataaccess.impl.LazyRevisionMetadata;
17  
18  /**
19   * This is the abstract base-implementation of a {@link AbstractGenericDao} using to manage the revision-control.
20   *
21   * @param <ID> is the type of the {@link MutablePersistenceEntity#getId() primary key} of the managed
22   *        {@link MutablePersistenceEntity entity}.
23   * @param <ENTITY> is the {@link #getEntityClass() type} of the managed entity.
24   *
25   */
26  public abstract class AbstractGenericRevisionedDao<ID, ENTITY extends MutablePersistenceEntity<ID>>
27      extends AbstractGenericDao<ID, ENTITY> implements GenericRevisionedDao<ID, ENTITY> {
28  
29    /**
30     * The constructor.
31     */
32    public AbstractGenericRevisionedDao() {
33  
34      super();
35    }
36  
37    /**
38     * @return the auditReader
39     */
40    protected AuditReader getAuditReader() {
41  
42      return AuditReaderFactory.get(getEntityManager());
43    }
44  
45    @Override
46    public ENTITY load(ID id, Number revision) throws ObjectNotFoundException {
47  
48      if (revision == MutablePersistenceEntity.LATEST_REVISION) {
49        return find(id);
50      } else {
51        return loadRevision(id, revision);
52      }
53    }
54  
55    /**
56     * This method gets a historic revision of the {@link net.sf.mmm.util.entity.api.GenericEntity} with the given
57     * <code>id</code>.
58     *
59     * @param id is the {@link net.sf.mmm.util.entity.api.GenericEntity#getId() ID} of the requested
60     *        {@link net.sf.mmm.util.entity.api.GenericEntity entity}.
61     * @param revision is the {@link MutablePersistenceEntity#getRevision() revision}
62     * @return the requested {@link net.sf.mmm.util.entity.api.GenericEntity entity}.
63     */
64    protected ENTITY loadRevision(Object id, Number revision) {
65  
66      Class<? extends ENTITY> entityClassImplementation = getEntityClass();
67      ENTITY entity = getAuditReader().find(entityClassImplementation, id, revision);
68      if (entity != null) {
69        entity.setRevision(revision);
70      }
71      return entity;
72    }
73  
74    @Override
75    public List<Number> getRevisionHistory(ID id) {
76  
77      return getAuditReader().getRevisions(getEntityClass(), id);
78    }
79  
80    @Override
81    public List<RevisionMetadata> getRevisionHistoryMetadata(Object id) {
82  
83      AuditReader auditReader = getAuditReader();
84      List<Number> revisionList = auditReader.getRevisions(getEntityClass(), id);
85      List<RevisionMetadata> result = new ArrayList<>();
86      for (Number revision : revisionList) {
87        Long revisionLong = Long.valueOf(revision.longValue());
88        result.add(new LazyRevisionMetadata(getEntityManager(), revisionLong));
89      }
90      return result;
91    }
92  }
