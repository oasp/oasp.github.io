8   /**
9    * A {@link SimpleModule} to extend Jackson to mixin annotations for polymorphic types.
10   *
11   * @since 3.0.0
12   */
13  public class MixInAnnotationsModule extends SimpleModule {
14  
15    private static final long serialVersionUID = 1L;
16  
17    private final Class<?>[] polymorphicClasses;
18  
19    /**
20     * @param polymorphicClasses the classes reflecting JSON transfer-objects that are polymorphic.
21     */
22    public MixInAnnotationsModule(Class<?>... polymorphicClasses) {
23  
24      super("oasp.PolymorphyModule",
25          new Version(1, 0, 0, null, ObjectMapperFactory.GROUP_ID, ObjectMapperFactory.ARTIFACT_ID));
26      this.polymorphicClasses = polymorphicClasses;
27    }
28  
29    @Override
30    public void setupModule(SetupContext context) {
31  
32      for (Class<?> type : this.polymorphicClasses) {
33        context.setMixInAnnotations(type, JacksonPolymorphicAnnotation.class);
34      }
35    }
36

38     * The blueprint class for the following JSON-annotation allowing to convert from JSON to POJO and vice versa
39     *
40     */
41    @JsonTypeInfo(use = JsonTypeInfo.Id.NAME, include = As.PROPERTY, property = "@type")
42    public static class JacksonPolymorphicAnnotation {
43  
44    }
