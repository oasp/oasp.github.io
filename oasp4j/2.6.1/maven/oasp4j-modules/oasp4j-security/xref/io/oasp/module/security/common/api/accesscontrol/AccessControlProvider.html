1   package io.oasp.module.security.common.api.accesscontrol;
2   
3   import java.util.Set;
4   
5   /**
6    * This is the interface for a provider of {@link AccessControl}s. It allows to
7    * {@link #collectAccessControlIds(String, Set) collect} all {@link AccessControl}s for an ID of a {@link AccessControl}
8    * (typically a {@link AccessControlGroup} or role). This is used to expand the groups provided by the access-manager
9    * (authentication and identity-management) to the full set of {@link AccessControlPermission permissions} of the
10   * {@link java.security.Principal user}.<br/>
11   * The actual authorization can then check individual permissions of the user by simply checking for
12   * {@link Set#contains(Object) contains} in the collected {@link Set}, what is very fast as security checks happen
13   * frequently.
14   *
15   * @see PrincipalAccessControlProvider
16   *
17   */
18  public interface AccessControlProvider {
19  
20    /**
21     * @param id is the {@link AccessControl#getId() ID} of the requested {@link AccessControl}.
22     * @return the requested {@link AccessControl} or {@code null} if not found.
23     */
24    AccessControl getAccessControl(String id);
25  
26    /**
27     * This method collects the {@link AccessControl#getId() IDs} of all {@link AccessControlPermission}s (or more
28     * precisely of all {@link AccessControl}s) contained in the {@link AccessControl} {@link AccessControl#getId()
29     * identified} by the given <code>groupId</code>.
30     *
31     * @see #collectAccessControls(String, Set)
32     *
33     * @param id is the {@link AccessControl#getId() ID} of the {@link AccessControl} (typically an
34     *        {@link AccessControlGroup}) to collect.
35     * @param permissions is the {@link Set} where to {@link Set#add(Object) add} the collected
36     *        {@link AccessControl#getId() IDs}. This will include the given <code>groupId</code>.
37     * @return {@code true} if the given <code>groupId</code> has been found, {@code false} otherwise.
38     */
39    boolean collectAccessControlIds(String id, Set<String> permissions);
40  
41    /**
42     * This method collects the {@link AccessControl}s contained in the {@link AccessControl} 
43     * {@link AccessControl#getId() identified} by the given <code>groupId</code>.
44     *
45     * @param id is the {@link AccessControl#getId() ID} of the {@link AccessControl} (typically an
46     *        {@link AccessControlGroup}) to collect.
47     * @param permissions is the {@link Set} where to {@link Set#add(Object) add} the collected {@link AccessControl}s.
48     *        This will include the {@link AccessControl} {@link AccessControl#getId() identified} by the given
49     *        <code>groupId</code>.
50     * @return {@code true} if the given <code>groupId</code> has been found, {@code false} otherwise.
51     */
52    boolean collectAccessControls(String id, Set<AccessControl> permissions);
53  
54  }
