26  public class AccessControlSchemaXmlValidationTest extends ModuleTest {

28    /**
29     * Tests if the access-control-schema.xml is valid.
30     *
31     * @throws ParserConfigurationException If a DocumentBuilder cannot be created which satisfies the configuration
32     *         requested.
33     * @throws IOException If any IO errors occur.
34     * @throws SAXException If an error occurs during the validation.
35     */
36    @Test
37    public void validateAccessControllSchema() throws ParserConfigurationException, SAXException, IOException {
38  
39      // parse an XML document into a DOM tree
40      DocumentBuilder parser = DocumentBuilderFactory.newInstance().newDocumentBuilder();
41      String xmlPath = getClass().getResource("/config/app/security/access-control-schema.xml").getPath();
42      Document document = parser.parse(new File(xmlPath));
43  
44      // create a SchemaFactory capable of understanding WXS schemas
45      SchemaFactory factory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
46  
47      // load a WXS schema, represented by a Schema instance
48      URL schemaPath = getClass().getResource("/io/oasp/module/security/access-control-schema.xsd");
49      Schema schema = factory.newSchema(schemaPath);
50  
51      // create a Validator instance, which can be used to validate an instance document
52      Validator validator = schema.newValidator();
53  
54      // validate the DOM tree
55      validator.validate(new DOMSource(document));
56    }
57  }
