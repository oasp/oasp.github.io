11  /**
12   * The implementation of {@link PrincipalAccessControlProvider} for this sample application.<br/>
13   * ATTENTION:<br/>
14   * In reality you would typically receive the user-profile from the central identity-management (via LDAP) and the roles
15   * (and groups) from a central access manager (that might also implement the identify-management). This design was only
16   * chosen to keep our sample application simple. Otherwise one would have to start a separate external server
17   * application to make everything work what would be too complicated to get things running easily.
18   *
19   */
20  @Named
21  public class PrincipalAccessControlProviderImpl implements PrincipalAccessControlProvider<UserProfile> {
22  
23    /**
24     * The constructor.
25     */
26    public PrincipalAccessControlProviderImpl() {
27  
28      super();
29    }
30  
31    @Override
32    public Collection<String> getAccessControlIds(UserProfile principal) {
33  
34      return Arrays.asList(principal.getRole().getName());
35    }
36  
37  }
