10  /**
11   * {@link ApplicationPersistenceEntity Entity} representing a {@link Table} of the restaurant. A table has a unique
12   * {@link #getNumber() number} can be {@link TableState#isReserved() reserved}, {@link TableState#isOccupied() occupied}
13   * and may have a {@link io.oasp.gastronomy.restaurant.staffmanagement.dataaccess.api.StaffMemberEntity waiter}
14   * assigned.
15   *
16   */
17  @Entity
18  // Table is a reserved word in SQL/RDBMS and can not be used as table name
19  @javax.persistence.Table(name = "RestaurantTable")
20  public class TableEntity extends ApplicationPersistenceEntity implements Table {
21  
22    private static final long serialVersionUID = 1L;
23  
24    private Long number;
25  
26    private Long waiterId;
27  
28    private TableState state;
29  
30    @Override
31    /*
32     * Uncomment the following Column annotation if the database used is Oracle 11g and comment the Column annotation just
33     * before @Override annotation
34     */
35    // @Column(name = "\"number\"", unique = true)
36    @Column(unique = true)
37    public Long getNumber() {
38  
39      return this.number;
40    }
41  
42    @Override
43    public void setNumber(Long number) {
44  
45      this.number = number;
46    }
47  
48    @Override
49    @Column(name = "waiterId")
50    public Long getWaiterId() {
51  
52      return this.waiterId;
53    }
54  
55    @Override
56    public void setWaiterId(Long waiterId) {
57  
58      this.waiterId = waiterId;
59    }
60  
61    @Override
62    public TableState getState() {
63  
64      return this.state;
65    }
66  
67    @Override
68    public void setState(TableState state) {
69  
70      this.state = state;
71    }
72  
73  }
