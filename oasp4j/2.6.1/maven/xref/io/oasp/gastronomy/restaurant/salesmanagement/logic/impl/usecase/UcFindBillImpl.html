23   * Implementation of {@link UcFindBill}.

25   */
26  @Named
27  @UseCase
28  public class UcFindBillImpl extends AbstractBillUc implements UcFindBill {
29  
30    private static final Logger LOG = LoggerFactory.getLogger(UcFindBillImpl.class);
31  
32    private Salesmanagement salesmanagement;
33  
34    @Override
35    @RolesAllowed(PermissionConstants.FIND_BILL)
36    public BillCto findBill(long id) {
37  
38      LOG.debug("Get Bill with id '" + id + "' from database.");
39      BillCto billCto = new BillCto();
40      BillEto bill = getBeanMapper().map(getBillDao().findOne(id), BillEto.class);
41      if (bill == null) {
42        return null;
43      }
44      billCto.setBill(bill);
45  
46      List<OrderPositionEto> orderPositions = new ArrayList<>();
47      for (Long orderPositionId : bill.getOrderPositionIds()) {
48        OrderPositionEto position = this.salesmanagement.findOrderPosition(orderPositionId);
49        orderPositions.add(position);
50      }
51      billCto.setPositions(orderPositions);
52  
53      return billCto;
54    }
55  
56    /**
57     * @param salesmanagement the salesmanagement to set
58     */
59    @Inject
60    public void setSalesmanagement(Salesmanagement salesmanagement) {
61  
62      this.salesmanagement = salesmanagement;
63    }

65  }
