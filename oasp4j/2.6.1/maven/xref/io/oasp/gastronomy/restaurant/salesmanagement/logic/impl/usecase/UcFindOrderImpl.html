25  /**
26   * Use Case to find an order.
27   *
28   */
29  @Named
30  @UseCase
31  public class UcFindOrderImpl extends AbstractOrderUc implements UcFindOrder {
32  
33    private static final Logger LOG = LoggerFactory.getLogger(UcFindOrderImpl.class);
34  
35    private Salesmanagement salesManagement;
36  
37    @Override
38    @RolesAllowed(PermissionConstants.FIND_ORDER)
39    public OrderEto findOrder(long orderId) {
40  
41      LOG.debug("Get order.");
42      return getBeanMapper().map(getOrderDao().findOne(orderId), OrderEto.class);
43    }
44  
45    @Override
46    @RolesAllowed(PermissionConstants.FIND_ORDER)
47    public PaginatedListTo<OrderEto> findOrderEtos(OrderSearchCriteriaTo criteria) {
48  
49      criteria.limitMaximumPageSize(MAXIMUM_HIT_LIMIT);
50      PaginatedListTo<OrderEntity> orders = getOrderDao().findOrders(criteria);
51  
52      return mapPaginatedEntityList(orders, OrderEto.class);
53    }
54  
55    @Override
56    @RolesAllowed(PermissionConstants.FIND_ORDER)
57    public PaginatedListTo<OrderCto> findOrderCtos(OrderSearchCriteriaTo criteria) {
58  
59      PaginatedListTo<OrderEto> orderEtos = findOrderEtos(criteria);
60  
61      List<OrderCto> page = new ArrayList<>(orderEtos.getResult().size());
62      for (OrderEto order : orderEtos.getResult()) {
63        // REVIEW <who> (hohwille) N+1 problem. Find a better and more efficient way to load the order positions.
64        page.add(findOrderCto(order));
65      }
66  
67      PaginatedListTo<OrderCto> result = new PaginatedListTo<>(page, orderEtos.getPagination());
68      return result;
69    }
70  
71    @Override
72    @RolesAllowed(PermissionConstants.FIND_ORDER)
73    public OrderCto findOrderCto(OrderEto order) {
74  
75      OrderCto result = new OrderCto();
76      result.setOrder(order);
77      List<OrderPositionEto> positions = this.salesManagement.findOrderPositionsByOrderId(order.getId());
78      result.setPositions(positions);
79      return result;
80    }
81  
82    @Override
83    @RolesAllowed(PermissionConstants.FIND_ORDER)
84    public OrderEto findOpenOrderForTable(long tableId) {
85  
86      OrderEntity order = getOrderDao().findOpenOrderByTable(tableId);
87      return getBeanMapper().map(order, OrderEto.class);
88    }
89  
90    /**
91     * @param salesManagement the {@link Salesmanagement} to {@link Inject}.
92     */
93    @Inject
94    public void setSalesManagement(Salesmanagement salesManagement) {
95  
96      this.salesManagement = salesManagement;
97    }
