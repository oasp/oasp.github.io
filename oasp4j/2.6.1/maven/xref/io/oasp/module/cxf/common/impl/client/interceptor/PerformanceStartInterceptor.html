10  /**
11   * Implementation of {@link AbstractPhaseInterceptor} that logs the duration time of a service client invocation.
12   *
13   * @since 3.0.0
14   */
15  public class PerformanceStartInterceptor extends AbstractPhaseInterceptor<Message> {
16  
17    /**
18     * The constructor.
19     */
20    public PerformanceStartInterceptor() {
21      super(Phase.SETUP);
22    }
23  
24    @Override
25    public void handleMessage(Message message) throws Fault {
26  
27      message.getExchange().put(TimeMeasure.class, new TimeMeasure());
28    }

30  }
