7   import javax.inject.Inject;
8   import javax.ws.rs.Path;
9   
10  import org.apache.cxf.jaxrs.client.Client;
11  import org.apache.cxf.jaxrs.client.ClientConfiguration;
12  import org.apache.cxf.jaxrs.client.JAXRSClientFactory;
13  import org.apache.cxf.jaxrs.client.WebClient;

15  import com.fasterxml.jackson.jaxrs.json.JacksonJsonProvider;

17  import io.oasp.module.cxf.common.impl.client.SyncServiceClientFactoryCxf;
18  import io.oasp.module.service.common.api.client.context.ServiceContext;
19  import io.oasp.module.service.common.api.constants.ServiceConstants;
20  import io.oasp.module.service.common.api.sync.SyncServiceClientFactory;
21  
22  /**
23   * Implementation of {@link SyncServiceClientFactory} for JAX-RS REST service clients using Apache CXF.
24   *
25   * @since 3.0.0
26   */
27  public class SyncServiceClientFactoryCxfRest extends SyncServiceClientFactoryCxf {
28  
29    private JacksonJsonProvider jsonProvider;
30  
31    /**
32     * The constructor.
33     */
34    public SyncServiceClientFactoryCxfRest() {
35      super();
36    }
37  
38    /**
39     * @param jsonProvider the {@link JacksonJsonProvider} to {@link Inject}.
40     */
41    @Inject
42    public void setJsonProvider(JacksonJsonProvider jsonProvider) {
43  
44      this.jsonProvider = jsonProvider;
45    }
46  
47    @Override
48    protected <S> void applyAspects(ServiceContext<S> context, S serviceClient, String serviceName) {
49  
50      ClientConfiguration clientConfig = WebClient.getConfig(serviceClient);
51      applyInterceptors(context, clientConfig, serviceName);
52      applyClientPolicy(context, clientConfig.getHttpConduit());
53      applyHeaders(context, serviceClient);
54    }
55  
56    @Override
57    protected <S> S createService(ServiceContext<S> context, String url, String serviceName) {
58  
59      List<Object> providers = createProviderList(context, serviceName);
60      return JAXRSClientFactory.create(url, context.getApi(), providers);
61    }
62  
63    @Override
64    protected String getServiceTypeFolderName() {
65  
66      return ServiceConstants.URL_FOLDER_REST;
67    }
68  
69    @Override
70    protected void applyHeaders(ServiceContext<?> context, Object serviceClient) {
71  
72      Collection<String> headerNames = context.getHeaderNames();
73      if (!headerNames.isEmpty()) {
74        Client webClient = WebClient.client(serviceClient);
75        for (String headerName : headerNames) {
76          webClient.header(headerName, context.getHeader(headerName));
77        }
78      }
79    }
80  
81    @Override
82    protected boolean isResponsibleForService(ServiceContext<?> context) {
83  
84      return context.getApi().isAnnotationPresent(Path.class);
85    }
86  
87    /**
88     * @param context the {@link ServiceContext}.
89     * @param serviceName the {@link #createServiceName(ServiceContext) service name}.
90     * @return the {@link List} of {@link javax.ws.rs.ext.Provider}s.
91     */
92    protected List<Object> createProviderList(ServiceContext<?> context, String serviceName) {
93  
94      List<Object> providers = new ArrayList<>();
95      if (this.jsonProvider != null) {
96        providers.add(this.jsonProvider);
97      }
98      providers.add(new RestServiceExceptionMapper(serviceName));
99      return providers;
100   }

102 }
