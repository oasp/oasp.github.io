1   package io.oasp.module.logging.common.impl;
2   
3   import java.util.regex.Pattern;
4   
5   import ch.qos.logback.classic.PatternLayout;
6   import ch.qos.logback.classic.spi.ILoggingEvent;
7   import ch.qos.logback.classic.spi.IThrowableProxy;
8   import ch.qos.logback.classic.spi.StackTraceElementProxy;
9   import ch.qos.logback.core.CoreConstants;
10  
11  /**
12   * Custom {@link PatternLayout} for logging entries.
13   * 
14   */
15  public class SingleLinePatternLayout extends PatternLayout {
16  
17    // --------------- FIELDS ---------------
18  
19    /** The separator used as replacement for newlines. */
20    private static final String LINE_SEP = " | ";
21  
22    /** OS specific line separator. */
23    private static final String NEWLINE = CoreConstants.LINE_SEPARATOR;
24  
25    /** Regular expression for line breaks. */
26    private static final Pattern LINEBREAK_PATTERN = Pattern.compile("[\\r\\n|\\n]");
27  
28    /** Average buffer size per line of stack trace. */
29    private static final int BUFFER_PER_LINE = 50;
30  
31    // --------------- CONSTRUCTORS ---------------
32  
33    /**
34     * Default constructor.
35     */
36    public SingleLinePatternLayout() {
37  
38      super();
39    }
40  
41    // --------------- METHODS ---------------
42  
43    /**
44     * Creates formatted String, using conversion pattern.
45     * 
46     * @param event ILoggingEvent
47     * @return String Formatted event as string in one line
48     */
49    @Override
50    public String doLayout(ILoggingEvent event) {
51  
52      // Format message
53      String msg = super.doLayout(event).trim();
54      // prevent log forging
55      msg = preventLogForging(msg);
56  
57      // Formatting of exception, remove line breaks
58      IThrowableProxy throwableProxy = event.getThrowableProxy();
59      if (throwableProxy != null) {
60        StackTraceElementProxy[] s = throwableProxy.getStackTraceElementProxyArray();
61        if (s != null && s.length > 0) {
62          // Performance: Initialize StringBuilder with (number of StackTrace-Elements + 1)*50 (+1 for Message)
63          StringBuilder sb = new StringBuilder(s.length * BUFFER_PER_LINE);
64          sb.append(msg);
65  
66          int len = s.length;
67          for (int i = 0; i < len; i++) {
68            sb.append(LINE_SEP).append(s[i]);
69          }
70          msg = sb.toString();
71        }
72      }
73      return msg + NEWLINE;
74    }
75  
76    /**
77     * Method to prevent log forging.
78     * 
79     * @param logMsg
80     * @return Encoded message
81     */
82    private String preventLogForging(String logMsg) {
83  
84      String result = logMsg;
85      // use precompiled pattern for performance reasons
86      result = LINEBREAK_PATTERN.matcher(logMsg).replaceAll(SingleLinePatternLayout.LINE_SEP);
87      return result;
88    }
89  
90  }
