6   /**
7    * Configuration for {@link io.oasp.module.service.common.api.client.ServiceClientFactory} and all related
8    * implementations.
9    *
10   * @see ServiceConfigProperties
11   *
12   * @since 3.0.0
13   */
14  public interface ServiceConfig {
15  
16    /** The key segment for the client specific configuration sub-tree. */
17    String KEY_SEGMENT_CLIENT = "client";
18  
19    /** The key segment for the URL of a {@link io.oasp.module.service.common.api.Service}. */
20    String KEY_SEGMENT_URL = "url";
21  
22    /** The key segment for the port of a {@link io.oasp.module.service.common.api.Service}. */
23    String KEY_SEGMENT_PORT = "port";
24  
25    /** The key segment for the host of a {@link io.oasp.module.service.common.api.Service}. */
26    String KEY_SEGMENT_HOST = "host";
27  
28    /** The key segment for the protocol of a {@link io.oasp.module.service.common.api.Service}. */
29    String KEY_SEGMENT_PROTOCOL = "protocol";
30  
31    /** The key segment for the WSDL settings of a SOAP {@link javax.jws.WebService}. */
32    String KEY_SEGMENT_WSDL = "wsdl";
33  
34    /** The key segment for the boolean property to disable download (e.g. of {@link #KEY_SEGMENT_WSDL WSDL}). */
35    String KEY_SEGMENT_DISABLE_DOWNLOAD = "disable-download";
36  
37    /** The key segment for the application specific configuration sub-tree. */
38    String KEY_SEGMENT_APP = "app";
39  
40    /** The key segment for the default configuration sub-tree. */
41    String KEY_SEGMENT_DEFAULT = "default";
42  
43    /** The key segment for the authentication mechanism (values are "basic", "oauth", etc.). */
44    String KEY_SEGMENT_AUTH = "auth";
45  
46    /**
47     * The key segment for the user configuration sub-tree.
48     *
49     * @see #KEY_SEGMENT_USER_LOGIN
50     * @see #KEY_SEGMENT_USER_PASSWORD
51     */
52    String KEY_SEGMENT_USER = "user";
53  
54    /** The key segment for the {@link #KEY_SEGMENT_USER user} login name. */
55    String KEY_SEGMENT_USER_LOGIN = "login";
56  
57    /** The key segment for the {@link #KEY_SEGMENT_USER user} password. */
58    String KEY_SEGMENT_USER_PASSWORD = "password";
59  
60    /**
61     * The key segment for the timeout configuration sub-tree.
62     *
63     * @see #KEY_SEGMENT_TIMEOUT_CONNECTION
64     * @see #KEY_SEGMENT_TIMEOUT_RESPONSE
65     */
66    String KEY_SEGMENT_TIMEOUT = "timeout";
67  
68    /** The key segment for the {@link #KEY_SEGMENT_TIMEOUT timeout} to establish a connection in seconds. */
69    String KEY_SEGMENT_TIMEOUT_CONNECTION = "connection";
70  
71    /** The key segment for the {@link #KEY_SEGMENT_TIMEOUT timeout} to wait for a response in seconds. */
72    String KEY_SEGMENT_TIMEOUT_RESPONSE = "response";
73  
74    /** The value of {@link #KEY_SEGMENT_AUTH authentication} for basic auth. */
75    String VALUE_AUTH_BASIC = "basic";
76  
77    /** The value of {@link #KEY_SEGMENT_AUTH authentication} for OAuth. */
78    String VALUE_AUTH_OAUTH = "oauth";
79  
80    /**
81     * The value of {@link #KEY_SEGMENT_AUTH authentication} for Basic, Oauth,JWT.
82     **/
83  
84    String VALUE_AUTH_FORWARD = "authForward";
85  
86    /**
87     * @return the root {@link ConfigProperties}-node with the configuration for services.
88     */
89    ConfigProperties asConfig();
90  
91    /**
92     * @return the client {@link ConfigProperties}-node with the configuration for
93     *         {@link io.oasp.module.service.common.api.client.ServiceClientFactory} and all related implementations.
94     */
95    ConfigProperties asClientConfig();
96  
97  }
