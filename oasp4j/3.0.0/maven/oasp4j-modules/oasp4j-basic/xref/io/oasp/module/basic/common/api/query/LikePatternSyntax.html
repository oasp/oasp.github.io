3   /**
4    * Enum defining available syntaxes for a match pattern in a LIKE-clause. While databases typically require {@link #SQL}
5    * syntax, human user expect {@link #GLOB} syntax in search forms. Therefore this enum also supports
6    * {@link #convert(String, LikePatternSyntax, boolean) conversion} from one syntax to another.
7    *
8    * @since 3.0.0
9    */
10  public enum LikePatternSyntax {
11  
12    /**
13     * Glob syntax that is typically expected by end-users and supported by typical search forms. It uses asterisk ('*')
14     * for {@link #getAny() any wildcard} and question-mark ('?') for {@link #getSingle() single wildcard}.
15     */
16    GLOB('*', '?'),
17  
18    /**
19     * SQL syntax that is typically required by databases. It uses percent ('%') for {@link #getAny() any wildcard} and
20     * underscore ('_') for {@link #getSingle() single wildcard}.
21     */
22    SQL('%', '_');
23  
24    /** The escape character. */
25    public static final char ESCAPE = '\\';
26  
27    private final char any;
28  
29    private final char single;
30  
31    private LikePatternSyntax(char any, char single) {
32  
33      this.any = any;
34      this.single = single;
35    }
36  
37    /**
38     * @return the wildcard character that matches any string including the {@link String#isEmpty() empty} string.
39     */
40    public char getAny() {
41  
42      return this.any;
43    }
44  
45    /**
46     * @return the wildcard character that matches exactly one single character.
47     */
48    public char getSingle() {
49  
50      return this.single;
51    }
52  
53    /**
54     * @param pattern the LIKE pattern in the given {@link LikePatternSyntax}.
55     * @param syntax the {@link LikePatternSyntax} of the given {@code pattern}.
56     * @return the given {@code pattern} converted to this {@link LikePatternSyntax}.
57     */
58    public String convert(String pattern, LikePatternSyntax syntax) {
59  
60      return convert(pattern, syntax, false);
61    }
62  
63    /**
64     * @param pattern the LIKE pattern in the given {@link LikePatternSyntax}.
65     * @param syntax the {@link LikePatternSyntax} of the given {@code pattern}.
66     * @param matchSubstring - {@code true} if the given {@code pattern} shall also match substrings, {@code false}
67     *        otherwise.
68     * @return the given {@code pattern} converted to this {@link LikePatternSyntax}.
69     */
70    public String convert(String pattern, LikePatternSyntax syntax, boolean matchSubstring) {
71  
72      if ((pattern == null) || pattern.isEmpty()) {
73        if (matchSubstring) {
74          return Character.toString(this.any);
75        } else {
76          return pattern;
77        }
78      }
79      if (this == syntax) {
80        String result = pattern;
81        if (matchSubstring) {
82          if (pattern.charAt(0) != this.any) {
83            result = this.any + result;
84          }
85          int lastIndex = pattern.length() - 1;
86          if ((pattern.charAt(lastIndex) != this.any) || ((lastIndex > 0) && (pattern.charAt(lastIndex - 1) == ESCAPE))) {
87            result = result + this.any;
88          }
89        }
90        return result;
91      }
92      int length = pattern.length();
93      StringBuilder sb = new StringBuilder(length + 8);
94      boolean lastWildcardAny = false;
95      for (int i = 0; i < length; i++) {
96        lastWildcardAny = false;
97        char c = pattern.charAt(i);
98        if (c == syntax.any) {
99          c = this.any;
100         lastWildcardAny = true;
101       } else if (c == syntax.single) {
102         c = this.single;
103       } else if ((c == this.any) || (c == this.single) || (c == ESCAPE)) {
104         if ((i == 0) && matchSubstring) {
105           sb.append(this.any);
106         }
107         sb.append(ESCAPE);
108       }
109       if (matchSubstring && (i == 0) && !lastWildcardAny) {
110         sb.append(this.any);
111       }
112       sb.append(c);
113     }
114     if (matchSubstring && !lastWildcardAny) {
115       sb.append(this.any);
116     }
117     return sb.toString();
118   }

120   /**
121    * @param pattern the string value that may be a pattern.
122    * @return the {@link LikePatternSyntax} for the given {@code pattern} or {@code null} if the given {@code pattern}
123    *         does not contain any wildcards.
124    */
125   public static LikePatternSyntax autoDetect(String pattern) {
126 
127     if ((pattern == null) || pattern.isEmpty()) {
128       return null;
129     }
130     for (LikePatternSyntax syntax : values()) {
131       if (pattern.indexOf(syntax.any) > 0) {
132         return syntax;
133       } else if (pattern.indexOf(syntax.single) > 0) {
134         return syntax;
135       }
136     }
137     return null;
138   }
