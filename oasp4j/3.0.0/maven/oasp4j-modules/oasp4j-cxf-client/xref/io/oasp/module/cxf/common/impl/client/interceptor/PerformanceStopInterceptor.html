14  /**
15   * Implementation of {@link AbstractPhaseInterceptor} that logs the duration time of a service client invocation.
16   *
17   * @since 3.0.0
18   */
19  public class PerformanceStopInterceptor extends AbstractPhaseInterceptor<Message> {
20  
21    private static final Logger LOG = LoggerFactory.getLogger(PerformanceStopInterceptor.class);
22  
23    private boolean errorLogged;
24  
25    /**
26     * The constructor.
27     */
28    public PerformanceStopInterceptor() {
29      super(Phase.POST_INVOKE);
30    }
31  
32    @Override
33    public void handleMessage(Message message) throws Fault {
34  
35      TimeMeasure timeMeasure = message.getExchange().get(TimeMeasure.class);
36      if (timeMeasure == null) {
37        if (!this.errorLogged) {
38          LOG.warn("Invalid setup - no TimeMeasure present!");
39          this.errorLogged = true;
40        }
41        return;
42      }
43      Throwable exception = message.getContent(Exception.class);
44      if (exception == null) {
45        timeMeasure.succeed();
46      }
47      Conduit conduit = message.get(Conduit.class);
48      EndpointReferenceType target = conduit.getTarget();
49      String url = target.getAddress().getValue();
50      timeMeasure.log(LOG, url);
51    }
52  
53  }
