1   package io.oasp.module.jpa.dataaccess.api;
2   
3   import javax.persistence.EntityManager;
4   
5   import org.slf4j.Logger;
6   import org.slf4j.LoggerFactory;
7   
8   /**
9    * Internal access to {@link EntityManager}.
10   */
11  class JpaEntityManagerAccess {
12  
13    private static final Logger LOG = LoggerFactory.getLogger(JpaEntityManagerAccess.class);
14  
15    private static EntityManager entityManager;
16  
17    static void setEntityManager(EntityManager entityManager, boolean check) {
18  
19      if ((JpaEntityManagerAccess.entityManager != null) && (JpaEntityManagerAccess.entityManager != entityManager)) {
20        if (check) {
21          throw new IllegalStateException("EntityManager has already been initialized!");
22        } else {
23          LOG.debug("EntityManager conflict: {} has been replaced with {}. This may only happen during tests.",
24              JpaEntityManagerAccess.entityManager, entityManager);
25        }
26      }
27      JpaEntityManagerAccess.entityManager = entityManager;
28    }
29  
30    static boolean hasEntityManager() {
31  
32      return (entityManager != null);
33    }
34  
35    static EntityManager getEntityManager() {
36  
37      if (entityManager == null) {
38        throw new IllegalStateException("EntityManager has not yet been initialized!");
39      }
40      return entityManager;
41    }
42  
43  }
