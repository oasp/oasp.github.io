128   /**
129    * @param expression the {@link StringExpression} to {@link StringExpression#like(String) create the LIKE-clause}
130    *        from.
131    * @param pattern the pattern for the LIKE-clause to create.
132    * @param syntax the {@link LikePatternSyntax} of the given {@code pattern}.
133    * @param ignoreCase - {@code true} to ignore the case, {@code false} otherwise (to search case-sensitive).
134    * @param matchSubstring - {@code true} to match also if the given {@code pattern} shall also match substrings on the
135    *        given {@link StringExpression}.
136    * @param negate - {@code true} for {@link StringExpression#notLike(String) NOT LIKE}, {@code false} for
137    *        {@link StringExpression#like(String) LIKE}.
138    * @return the LIKE-clause as {@link BooleanExpression}.
139    */
140   protected BooleanExpression newLikeClause(StringExpression expression, String pattern, LikePatternSyntax syntax,
141       boolean ignoreCase, boolean matchSubstring, boolean negate) {
142 
143     if (syntax == null) {
144       syntax = LikePatternSyntax.autoDetect(pattern);
145       if (syntax == null) {
146         syntax = LikePatternSyntax.SQL;
147       }
148     }
149     String likePattern = LikePatternSyntax.SQL.convert(pattern, syntax, matchSubstring);
150     StringExpression exp = expression;
151     if (ignoreCase) {
152       likePattern = likePattern.toUpperCase(Locale.US);
153       exp = exp.upper();
154     }
155     BooleanExpression clause;
156     if (syntax != LikePatternSyntax.SQL) {
157       clause = exp.like(likePattern, LikePatternSyntax.ESCAPE);
158     } else {
159       clause = exp.like(likePattern);
160     }
161     if (negate) {
162       clause = clause.not();
163     }
164     return clause;
165   }

167   /**
168    * @param expression the {@link StringExpression} to search on.
169    * @param value the string value or pattern to search for.
170    * @param config the {@link StringSearchConfigTo} to configure the search. May be {@code null} for regular equals
171    *        search as default fallback.
172    * @return the new {@link BooleanExpression} for the specified string comparison clause.
173    */
174   protected BooleanExpression newStringClause(StringExpression expression, String value, StringSearchConfigTo config) {
175 
176     StringSearchOperator operator = StringSearchOperator.EQ;
177     LikePatternSyntax syntax = null;
178     boolean ignoreCase = false;
179     boolean matchSubstring = false;
180     if (config != null) {
181       operator = config.getOperator();
182       syntax = config.getLikeSyntax();
183       ignoreCase = config.isIgnoreCase();
184       matchSubstring = config.isMatchSubstring();
185     }
186     return newStringClause(expression, value, operator, syntax, ignoreCase, matchSubstring);
187   }

189   /**
190    * @param expression the {@link StringExpression} to search on.
191    * @param value the string value or pattern to search for.
192    * @param syntax the {@link LikePatternSyntax} of the given {@code pattern}.
193    * @param operator the {@link StringSearchOperator} used to compare the search string {@code value}.
194    * @param ignoreCase - {@code true} to ignore the case, {@code false} otherwise (to search case-sensitive).
195    * @param matchSubstring - {@code true} to match also if the given {@code pattern} shall also match substrings on the
196    *        given {@link StringExpression}.
197    * @return the new {@link BooleanExpression} for the specified string comparison clause.
198    */
199   protected BooleanExpression newStringClause(StringExpression expression, String value, StringSearchOperator operator,
200       LikePatternSyntax syntax, boolean ignoreCase, boolean matchSubstring) {
201 
202     if (operator == null) {
203       if (syntax == null) {
204         syntax = LikePatternSyntax.autoDetect(value);
205         if (syntax == null) {
206           operator = StringSearchOperator.EQ;
207         } else {
208           operator = StringSearchOperator.LIKE;
209         }
210       } else {
211         operator = StringSearchOperator.LIKE;
212       }
213     }
214     if (matchSubstring && ((operator == StringSearchOperator.EQ) || (operator == StringSearchOperator.NE))) {
215       if (syntax == null) {
216         syntax = LikePatternSyntax.SQL;
217       }
218       if (operator == StringSearchOperator.EQ) {
219         operator = StringSearchOperator.LIKE;
220       } else {
221         operator = StringSearchOperator.NOT_LIKE;
222       }
223     }
224     String v = value;
225     if (v == null) {
226       switch (operator) {
227         case LIKE:
228         case EQ:
229           return expression.isNull();
230         case NE:
231           return expression.isNotNull();
232         default:
233           throw new IllegalArgumentException("Operator " + operator + " does not accept null!");
234       }
235     } else if (v.isEmpty()) {
236       switch (operator) {
237         case LIKE:
238         case EQ:
239           return expression.isEmpty();
240         case NOT_LIKE:
241         case NE:
242           return expression.isNotEmpty();
243         default:
244           // continue;
245       }
246     }
247     StringExpression exp = expression;
248     if (ignoreCase) {
249       v = v.toUpperCase(Locale.US);
250       exp = exp.upper();
251     }
252     switch (operator) {
253       case LIKE:
254         return newLikeClause(exp, v, syntax, false, matchSubstring, false);
255       case NOT_LIKE:
256         return newLikeClause(exp, v, syntax, false, matchSubstring, true);
257       case EQ:
258         return exp.eq(v);
259       case NE:
260         return exp.ne(v);
261       case LT:
262         return exp.lt(v);
263       case LE:
264         return exp.loe(v);
265       case GT:
266         return exp.gt(v);
267       case GE:
268         return exp.goe(v);
269       default:
270         throw new IllegalCaseException(StringSearchOperator.class, operator);
271     }
272   }

290   /**
291    * @param <T> generic type of the {@link Collection} values for the {@link SimpleExpression#in(Collection)
292    *        IN-clause}(s).
293    * @param expression the {@link SimpleExpression} used to create the {@link SimpleExpression#in(Collection)
294    *        IN-clause}(s).
295    * @param inValues the {@link Collection} of values for the {@link SimpleExpression#in(Collection) IN-clause}(s).
296    * @return the {@link BooleanExpression} for the {@link SimpleExpression#in(Collection) IN-clause}(s) oder
297    *         {@code null} if {@code inValues} is {@code null} or {@link Collection#isEmpty() empty}.
298    */
299   protected <T> BooleanExpression newInClause(SimpleExpression<T> expression, Collection<T> inValues) {
300 
301     if ((inValues == null) || (inValues.isEmpty())) {
302       return null;
303     }
304     int size = inValues.size();
305     if (size == 1) {
306       return expression.eq(inValues.iterator().next());
307     }
308     int maxSizeOfInClause = DatabaseConfigProperties.getInstance().getMaxSizeOfInClause();
309     if (size <= maxSizeOfInClause) {
310       return expression.in(inValues);
311     }
312     LOG.warn("Creating workaround for IN-clause with {} items - this can cause performance problems.", size);
313     List<T> values;
314     if (inValues instanceof List) {
315       values = (List<T>) inValues;
316     } else {
317       values = new ArrayList<>(inValues);
318     }
319     BooleanExpression predicate = null;
320     int start = 0;
321     while (start < size) {
322       int end = start + maxSizeOfInClause;
323       if (end > size) {
324         end = size;
325       }
326       List<T> partition = values.subList(start, end);
327       if (predicate == null) {
328         predicate = expression.in(partition);
329       } else {
330         predicate = predicate.or(expression.in(partition));
331       }
332       start = end;
333     }
334     return predicate;
335   }

355   /**
356    * Returns a {@link Page} of entities according to the supplied {@link Pageable} and {@link JPAQuery}.
357    *
358    * @param <E> generic type of the entity.
359    * @param pageable contains information about the requested page and sorting.
360    * @param query is a query which is pre-configured with the desired conditions for the search.
361    * @param determineTotal - {@code true} to determine the {@link Page#getTotalElements() total number of hits},
362    *        {@code false} otherwise.
363    * @return a paginated list.
364    */
365   protected <E> Page<E> findPaginatedGeneric(Pageable pageable, JPAQuery<E> query, boolean determineTotal) {
366 
367     long total = -1;
368     if (determineTotal) {
369       total = query.clone().fetchCount();
370     }
371     long offset = 0;
372     if (pageable != null) {
373       offset = pageable.getOffset();
374       query.offset(offset);
375       query.limit(pageable.getPageSize());
376       applySort(query, pageable.getSort());
377     }
378     List<E> hits = query.fetch();
379     if (total == -1) {
380       total = offset + hits.size();
381     }
382     return new PageImpl<>(hits, pageable, total);
383   }

385   /**
386    * @param query the {@link JPAQuery} to apply the {@link Sort} to.
387    * @param sort the {@link Sort} to apply as ORDER BY to the given {@link JPAQuery}.
388    */
389   @SuppressWarnings("rawtypes")
390   protected void applySort(JPAQuery<?> query, Sort sort) {
391 
392     if (sort == null) {
393       return;
394     }
395     PathBuilder<?> alias = findAlias(query);
396     for (Order order : sort) {
397       String property = order.getProperty();
398       Direction direction = order.getDirection();
399       ComparablePath<Comparable> path = alias.getComparable(property, Comparable.class);
400       OrderSpecifier<Comparable> orderSpecifier;
401       if (direction == Direction.ASC) {
402         orderSpecifier = path.asc();
403       } else {
404         orderSpecifier = path.desc();
405       }
406       query.orderBy(orderSpecifier);
407     }
408   }

410   private <E> PathBuilder<E> findAlias(JPAQuery<E> query) {
411 
412     String alias = null;
413     List<JoinExpression> joins = query.getMetadata().getJoins();
414     if ((joins != null) && !joins.isEmpty()) {
415       JoinExpression join = joins.get(0);
416       Expression<?> target = join.getTarget();
417       if (target instanceof EntityPath) {
418         alias = target.toString(); // no safe API
419       }
420     }
421     Class<E> type = query.getType();
422     if (alias == null) {
423       // should actually never happen, but fallback is provided as buest guess
424       alias = StringUtils.uncapitalize(type.getSimpleName());
425     }
426     return new PathBuilder<>(type, alias);
427   }
