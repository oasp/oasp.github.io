3   package io.oasp.module.jpa.dataaccess.impl;
4   
5   import java.util.Date;
6   
7   import javax.persistence.EntityManager;
8   
9   import io.oasp.module.jpa.dataaccess.api.AdvancedRevisionEntity;
10  import io.oasp.module.jpa.dataaccess.api.RevisionMetadata;
11  
12  /**
13   * This is a lazy implementation of the {@link RevisionMetadata} interface.
14   *
15   */
16  public class LazyRevisionMetadata implements RevisionMetadata {
17  
18    /** The {@link EntityManager} used to read the metadata. */
19    private final EntityManager entityManager;
20  
21    /** @see #getRevision() */
22    private final Long revision;
23  
24    /** @see #getRevisionEntity() */
25    private AdvancedRevisionEntity revisionEntity;
26  
27    /**
28     * The constructor.
29     *
30     * @param entityManager is the {@link EntityManager} used to fetch metadata.
31     * @param revision is the {@link #getRevision() revision}.
32     */
33    public LazyRevisionMetadata(EntityManager entityManager, Long revision) {
34  
35      super();
36      this.entityManager = entityManager;
37      this.revision = revision;
38    }
39  
40    /**
41     * @return the revisionEntity
42     */
43    public AdvancedRevisionEntity getRevisionEntity() {
44  
45      if (this.revisionEntity == null) {
46        this.revisionEntity = this.entityManager.find(AdvancedRevisionEntity.class, this.revision);
47        assert (this.revisionEntity != null);
48      }
49      return this.revisionEntity;
50    }
51  
52    @Override
53    public String getAuthor() {
54  
55      return getRevisionEntity().getUserLogin();
56    }
57  
58    @Override
59    public Date getDate() {
60  
61      return getRevisionEntity().getDate();
62    }
63  
64    @Override
65    public Number getRevision() {
66  
67      return this.revision;
68    }
69  
70  }
