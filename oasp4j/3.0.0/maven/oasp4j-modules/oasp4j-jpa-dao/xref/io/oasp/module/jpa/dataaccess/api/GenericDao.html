12  /**
13   * This is the interface for a <em>Data Access Object</em> (DAO). It acts as a manager responsible for the persistence
14   * operations on a specific {@link PersistenceEntity entity} {@literal <E>}.<br/>
15   * This is base interface contains the CRUD operations:
16   * <ul>
17   * <li>Create: call {@link #save(PersistenceEntity)} on a new entity.</li>
18   * <li>Retrieve: use <code>find*</code> methods such as {@link #findOne(Object)}. More specific queries will be added in
19   * dedicated DAO interfaces.</li>
20   * <li>Update: done automatically by JPA vendor (hibernate) on commit or call {@link #save(PersistenceEntity)} to
21   * {@link javax.persistence.EntityManager#merge(Object) merge} an entity.</li>
22   * <li>Delete: call {@link #delete(PersistenceEntity)} or {@link #delete(Object)}.</li>
23   * </ul>
24   * For each (non-abstract) implementation of {@link PersistenceEntity entity} <code>MyEntity</code> you should create an
25   * interface interface <code>MyEntityDao</code> that inherits from this {@link GenericDao} interface. Also you create an
26   * implementation of that interface <code>MyEntityDaoImpl</code> that you derive from
27   * {@link io.oasp.module.jpa.dataaccess.base.AbstractGenericDao}.
28   *
29   * @param <ID> is the generic type if the {@link PersistenceEntity#getId() primary key}.
30   * @param <E> is the generic type of the {@link PersistenceEntity}.
31   *
32   */
33  public interface GenericDao<ID, E extends PersistenceEntity<ID>> extends FeatureForceIncrementModificationCounter<E> {
34  
35    /**
36     * Saves a given entity. Use the returned instance for further operations as the save operation might have changed the
37     * entity instance completely.
38     *
39     * @param entity the {@link PersistenceEntity entity} to save
40     * @return the saved entity
41     */
42    E save(E entity);
43  
44    /**
45     * Saves all given entities.
46     *
47     * @param entities the {@link PersistenceEntity entities} to save
48     */
49    void save(Iterable<? extends E> entities);
50  
51    /**
52     * Retrieves an entity by its id.
53     *
54     * @param id must not be {@literal null}.
55     * @return the entity with the given id or {@literal null} if none found
56     * @throws ObjectNotFoundUserException if the requested entity does not exists (use {@link #findOne(Object)} to
57     *         prevent).
58     */
59    E find(ID id) throws ObjectNotFoundUserException;
60  
61    /**
62     * Retrieves an entity by its id.
63     *
64     * @param id must not be {@literal null}.
65     * @return the entity with the given id or {@literal null} if none found
66     * @throws IllegalArgumentException if {@code id} is {@code null}
67     */
68    E findOne(ID id) throws IllegalArgumentException;
69  
70    /**
71     * @param reference the {@link Ref} to the {@link PersistenceEntity} to get. Typically an instance of
72     *        {@link io.oasp.module.basic.common.api.reference.IdRef}.
73     * @return the {@link PersistenceEntity} as {@link javax.persistence.EntityManager#getReference(Class, Object)
74     *         reference} for the given {@link Ref}. Will be {@code null} if the given {@link Ref} was {@code null}.
75     */
76    E get(Ref<ID, E> reference);
77  
78    /**
79     * Returns whether an entity with the given id exists.
80     *
81     * @param id must not be {@literal null}.
82     * @return true if an entity with the given id exists, {@literal false} otherwise
83     */
84    boolean exists(ID id);
85  
86    /**
87     * Returns all instances of the type with the given IDs.
88     *
89     * @param ids are the IDs of all entities to retrieve e.g. as {@link java.util.List}.
90     * @return an {@link Iterable} with all {@link PersistenceEntity entites} for the given <code>ids</code>.
91     */
92    List<E> findAll(Iterable<ID> ids);
93  
94    /**
95     * Deletes the entity with the given id.
96     *
97     * @param id must not be {@literal null}.
98     * @throws IllegalArgumentException in case the given {@code id} is {@code null}
99     */
100   void delete(ID id) throws IllegalArgumentException;
101 
102   /**
103    * Deletes a given entity.
104    *
105    * @param entity the {@link PersistenceEntity entity} to delete
106    */
107   void delete(E entity);
108 
109   /**
110    * Deletes the given entities.
111    *
112    * @param entities the {@link PersistenceEntity entities} to delete
113    */
114   void delete(Iterable<? extends E> entities);
115 
116 }
