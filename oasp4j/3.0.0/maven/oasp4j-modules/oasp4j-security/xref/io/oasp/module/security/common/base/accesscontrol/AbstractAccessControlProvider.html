19  /**
20   * This is the abstract base implementation of {@link AccessControlProvider}.<br/>
21   * ATTENTION:<br/>
22   * You need to call {@link #initialize(AccessControlSchema)} from the derived implementation.
23   *
24   */
25  public abstract class AbstractAccessControlProvider implements AccessControlProvider {
26  
27    /** Logger instance. */
28    private static final Logger LOG = LoggerFactory.getLogger(AbstractAccessControlProvider.class);
29  
30    /** @see #getAccessControl(String) */
31    private final Map<String, AccessControl> id2nodeMap;
32  
33    /**
34     * The constructor.
35     */
36    public AbstractAccessControlProvider() {
37  
38      super();
39      this.id2nodeMap = new HashMap<>();
40    }
41  
42    /**
43     * Performs the required initialization of this class.
44     *
45     * @param config is the {@link AccessControlSchema}.
46     */
47    protected void initialize(AccessControlSchema config) {
48  
49      LOG.debug("Initializing.");
50      List<AccessControlGroup> groups = config.getGroups();
51      if (groups.size() == 0) {
52        throw new IllegalStateException("AccessControlSchema is empty - please configure at least one group!");
53      }
54      Set<AccessControlGroup> toplevelGroups = new HashSet<>(groups);
55      for (AccessControlGroup group : groups) {
56        collectAccessControls(group, toplevelGroups);
57        List<AccessControlGroup> groupList = new ArrayList<>();
58        groupList.add(group);
59        checkForCyclicDependencies(group, groupList);
60      }
61    }
62  
63    /**
64     * Checks that the given {@link AccessControlGroup} has no cyclic {@link AccessControlGroup#getInherits() inheritance
65     * graph}.
66     *
67     * @param group is the {@link AccessControlGroup} to check.
68     * @param groupList the {@link List} of visited {@link AccessControlGroup}s used to detect cycles.
69     */
70    protected void checkForCyclicDependencies(AccessControlGroup group, List<AccessControlGroup> groupList) {
71  
72      for (AccessControlGroup inheritedGroup : group.getInherits()) {
73        if (groupList.contains(inheritedGroup)) {
74          StringBuilder sb = new StringBuilder("A cyclic dependency of access control groups has been detected: \n");
75          for (int i = groupList.size() - 1; i >= 0; i--) {
76            AccessControlGroup node = groupList.get(i);
77            sb.append(node);
78            if (i > 0) {
79              sb.append("-->");
80            }
81          }
82          throw new IllegalStateException(sb.toString());
83        }
84        groupList.add(inheritedGroup);
85        checkForCyclicDependencies(inheritedGroup, groupList);
86        AccessControlGroup removed = groupList.remove(groupList.size() - 1);
87        assert (removed == inheritedGroup);
88      }
89    }
90  
91    /**
92     * Called from {@link #initialize(AccessControlSchema)} to collect all {@link AccessControl}s recursively.
93     *
94     * @param group the {@link AccessControlGroup} to traverse.
95     * @param toplevelGroups is the {@link Set} of all {@link AccessControlGroup}s from
96     *        {@link AccessControlSchema#getGroups()}.
97     */
98    protected void collectAccessControls(AccessControlGroup group, Set<AccessControlGroup> toplevelGroups) {
99  
100     if (!toplevelGroups.contains(group)) {
101       throw new IllegalStateException("Invalid group not declared as top-level group in schema: " + group);
102     }
103     AccessControl old = this.id2nodeMap.put(group.getId(), group);
104     if (old != null) {
105       LOG.debug("Already visited access control group {}", group);
106       if (old != group) {
107         throw new IllegalStateException(
108             "Invalid security configuration: duplicate groups with id " + group.getId() + "!");
109       }
110       // group has already been visited, stop recursion...
111       return;
112     } else {
113       LOG.debug("Registered access control group {}", group);
114     }
115     for (AccessControlPermission permission : group.getPermissions()) {
116       old = this.id2nodeMap.put(permission.getId(), permission);
117       if (old != null) {
118         // throw new IllegalStateException("Invalid security configuration: duplicate permission with id "
119         // + permission.getId() + "!");
120         LOG.warn("Security configuration contains duplicate permission with id {}.", permission.getId());
121       } else {
122         LOG.debug("Registered access control permission {}", permission);
123       }
124     }
125     for (AccessControlGroup inheritedGroup : group.getInherits()) {
126       collectAccessControls(inheritedGroup, toplevelGroups);
127     }
128   }
129 
130   @Override
131   public AccessControl getAccessControl(String nodeId) {
132 
133     return this.id2nodeMap.get(nodeId);
134   }
135 
136   /**
137    * Registers the given {@link AccessControl} and may be used for configuration of access controls during
138    * bootstrapping. This method should not be used after the application startup (bootstrapping) has completed.
139    *
140    * @param accessControl the {@link AccessControl} to register.
141    */
142   protected void addAccessControl(AccessControl accessControl) {
143 
144     String id = accessControl.getId();
145     AccessControl existing = this.id2nodeMap.get(id);
146     if (existing == null) {
147       this.id2nodeMap.put(id, accessControl);
148       LOG.debug("Registered access control {}", accessControl);
149     } else if (existing == accessControl) {
150       LOG.debug("Access control {} was already registered.", accessControl);
151     } else {
152       throw new IllegalStateException("Duplicate access control with ID '" + id + "'.");
153     }
154   }
155 
156   @Override
157   public boolean collectAccessControlIds(String groupId, Set<String> permissions) {
158 
159     AccessControl node = getAccessControl(groupId);
160     if (node instanceof AccessControlGroup) {
161       collectPermissionIds((AccessControlGroup) node, permissions);
162     } else {
163       // node does not exist or is a flat AccessControlPermission
164       permissions.add(groupId);
165     }
166     return (node != null);
167   }
168 
169   /**
170    * Recursive implementation of {@link #collectAccessControlIds(String, Set)} for {@link AccessControlGroup}s.
171    *
172    * @param group is the {@link AccessControlGroup} to traverse.
173    * @param permissions is the {@link Set} used to collect.
174    */
175   public void collectPermissionIds(AccessControlGroup group, Set<String> permissions) {
176 
177     boolean added = permissions.add(group.getId());
178     if (!added) {
179       // we have already visited this node, stop recursion...
180       return;
181     }
182     for (AccessControlPermission permission : group.getPermissions()) {
183       permissions.add(permission.getId());
184     }
185     for (AccessControlGroup inheritedGroup : group.getInherits()) {
186       collectPermissionIds(inheritedGroup, permissions);
187     }
188   }
189 
190   @Override
191   public boolean collectAccessControls(String groupId, Set<AccessControl> permissions) {
192 
193     AccessControl node = getAccessControl(groupId);
194     if (node == null) {
195       return false;
196     }
197     if (node instanceof AccessControlGroup) {
198       collectPermissionNodes((AccessControlGroup) node, permissions);
199     } else {
200       // node is a flat AccessControlPermission
201       permissions.add(node);
202     }
203     return true;
204   }
205 
206   /**
207    * Recursive implementation of {@link #collectAccessControls(String, Set)} for {@link AccessControlGroup}s.
208    *
209    * @param group is the {@link AccessControlGroup} to traverse.
210    * @param permissions is the {@link Set} used to collect.
211    */
212   public void collectPermissionNodes(AccessControlGroup group, Set<AccessControl> permissions) {
213 
214     boolean added = permissions.add(group);
215     if (!added) {
216       // we have already visited this node, stop recursion...
217       return;
218     }
219     for (AccessControlPermission permission : group.getPermissions()) {
220       permissions.add(permission);
221     }
222     for (AccessControlGroup inheritedGroup : group.getInherits()) {
223       collectPermissionNodes(inheritedGroup, permissions);
224     }
225   }
