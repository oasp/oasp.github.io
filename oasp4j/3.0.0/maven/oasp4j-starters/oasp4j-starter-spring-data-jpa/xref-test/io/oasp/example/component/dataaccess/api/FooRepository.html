12  import io.oasp.example.component.common.api.Foo;

17  /**
18   * {@link DefaultRepository} for {@link FooEntity}.
19   */
20  public interface FooRepository extends DefaultRepository<FooEntity> {
21  
22    /**
23     * @param message the {@link Foo#getMessage() message} to match.
24     * @param pageable the {@link Pageable} to configure the paging.
25     * @return the {@link Page} of {@link FooEntity} objects that matched the search.
26     */
27    @Query("SELECT foo FROM FooEntity foo" //
28        + " WHERE foo.message = :message")
29    Page<FooEntity> findByMessage(@Param("message") String message, Pageable pageable);
30  
31    /**
32     * @param criteria the {@link FooSearchCriteriaTo} with the criteria to search.
33     * @return the {@link Page} of the {@link FooEntity} objects that matched the search.
34     */
35    default Page<FooEntity> findByCriteria(FooSearchCriteriaTo criteria) {
36  
37      FooEntity alias = newDslAlias();
38      JPAQuery<FooEntity> query = newDslQuery(alias);
39      String message = criteria.getMessage();
40      if ((message != null) && !message.isEmpty()) {
41        QueryUtil.get().whereString(query, $(alias.getMessage()), message, criteria.getMessageOption());
42      }
43      return QueryUtil.get().findPaginated(criteria.getPageable(), query, false);
44    }
45  
46  }
