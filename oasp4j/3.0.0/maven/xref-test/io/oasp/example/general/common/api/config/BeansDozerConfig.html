3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import org.dozer.DozerBeanMapper;
7   import org.dozer.Mapper;
8   import org.springframework.context.annotation.Bean;
9   import org.springframework.context.annotation.ComponentScan;
10  import org.springframework.context.annotation.Configuration;
11  
12  /**
13   * Java bean configuration for Dozer
14   */
15  @Configuration
16  @ComponentScan(basePackages = { "io.oasp.module.beanmapping" })
17  public class BeansDozerConfig {
18  
19    private static final String DOZER_MAPPING_XML = "config/app/common/dozer-mapping.xml";
20  
21    /**
22     * @return the {@link DozerBeanMapper}.
23     */
24    @Bean
25    public Mapper getDozer() {
26  
27      List<String> beanMappings = new ArrayList<>();
28      beanMappings.add(DOZER_MAPPING_XML);
29      return new DozerBeanMapper(beanMappings);
30  
31    }
32  }
