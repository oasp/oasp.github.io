1   package io.oasp.module.basic.common.api.query;

3   /**
4    * Enum defining available operators for a string search or string comparison.
5    *
6    * @since 3.0.0
7    */
8   public enum StringSearchOperator {
9   
10    /** Matches if strings are {@link String#equals(Object) equal}. */
11    EQ("=="),
12  
13    /** Matches if strings are NOT {@link String#equals(Object) equal}. */
14    NE("!="),
15  
16    /** Matches if search value is less than search hit(s) in {@link String#compareTo(String) lexicographical order}. */
17    LT("<"),
18  
19    /**
20     * Matches if search value is less or equal to search hit(s) in {@link String#compareTo(String) lexicographical
21     * order}.
22     */
23    LE("<="),
24  
25    /**
26     * Matches if search value is greater than search hit(s) in {@link String#compareTo(String) lexicographical order}.
27     */
28    GT(">"),
29  
30    /**
31     * Matches if search value is greater or equal to search hit(s) in {@link String#compareTo(String) lexicographical
32     * order}.
33     */
34    GE(">="),
35  
36    /**
37     * Matches if search value as pattern matches search hit(s) in <em>LIKE</em> search.
38     *
39     * @see LikePatternSyntax
40     */
41    LIKE("LIKE"),
42  
43    /**
44     * Matches if search value as pattern does not match search hit(s) in <em>LIKE</em> search.
45     *
46     * @see LikePatternSyntax
47     */
48    NOT_LIKE("NOT LIKE");
49  
50    private final String operator;
51  
52    private StringSearchOperator(String operator) {
53  
54      this.operator = operator;
55    }
56  
57    @Override
58    public String toString() {
59  
60      return this.operator;
61    }
62  
63  }
