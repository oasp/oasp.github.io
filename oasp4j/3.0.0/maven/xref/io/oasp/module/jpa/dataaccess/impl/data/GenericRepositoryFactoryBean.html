14  /**
15   * {@link org.springframework.data.repository.core.support.RepositoryFactoryBeanSupport} for
16   * {@link GenericRepositoryImpl}. In order to use {@link GenericRepository} you need to annotate your spring-boot
17   * application as following:
18   *
19   * <pre>
20   * &#64;{@link org.springframework.data.jpa.repository.config.EnableJpaRepositories}(repositoryFactoryBeanClass = {@link GenericRepositoryFactoryBean}.class)
21   * </pre>
22   *
23   * @param <R> generic type of the {@link GenericRepository} interface.
24   * @param <E> generic type of the managed {@link net.sf.mmm.util.entity.api.PersistenceEntity entity}.
25   * @param <ID> generic type of the {@link net.sf.mmm.util.entity.api.PersistenceEntity#getId() primary key} of the
26   *        entity.
27   *
28   * @since 3.0.0
29   */
30  public class GenericRepositoryFactoryBean<R extends GenericRepository<E, ID>, E, ID extends Serializable>
31      extends JpaRepositoryFactoryBean<R, E, ID> {
32  
33    /**
34     * Creates a new {@link JpaRepositoryFactoryBean} for the given repository interface.
35     *
36     * @param repositoryInterface must not be {@literal null}.
37     */
38    public GenericRepositoryFactoryBean(Class<? extends R> repositoryInterface) {
39  
40      super(repositoryInterface);
41    }
42  
43    @Override
44    protected RepositoryFactorySupport createRepositoryFactory(EntityManager entityManager) {
45  
46      return new GenericRepositoryFactory(entityManager);
47    }
48  
49    private static class GenericRepositoryFactory extends JpaRepositoryFactory {
50  
51      /**
52       * The constructor.
53       *
54       * @param entityManager the {@link EntityManager}.
55       */
56      public GenericRepositoryFactory(EntityManager entityManager) {
57  
58        super(entityManager);
59      }
60  
61      @Override
62      protected Class<?> getRepositoryBaseClass(RepositoryMetadata metadata) {
63  
64        return GenericRepositoryImpl.class;
65      }
66  
67    }
68  
69  }
