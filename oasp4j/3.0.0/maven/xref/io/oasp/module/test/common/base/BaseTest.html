1   package io.oasp.module.test.common.base;

3   import org.assertj.core.api.Assertions;
4   import org.junit.After;
5   import org.junit.Before;
6   
7   /**
8    * This is the {@code abstract} base class for all tests. In most cases it will be convenient to extend this class. <br>
9    * <br/>
10   * Although it does not contain abstract methods, the class itself is declared {@code abstract} to clarify its purpose.
11   * <br/>
12   * <br/>
13   * This class provides {@code final} methods {@link #setUp()} and {@link #tearDown()} which call {@code protected}
14   * methods {@link #doSetUp()} and {@link #doTearDown()} internally. <br/>
15   * Both methods {@link #doSetUp()} and {@link #doTearDown()} are left empty here. If some default behaviour is desired
16   * during test setup or teardown these methods should be overridden by the subclass. <br/>
17   * Implementations <b>must</b> call the super implementation. This call should always happen at the beginning of the
18   * implementation. This helps to avoid confusion of call orders. <br/>
19   * <br/>
20   * The following listing should clarify the intention:
21   *
22   * <pre>
23   * public class MyTest extends BaseTest {
24   *
25   *   &#64;Override
26   *   protected void doSetUp() {
27   *
28   *     super.doSetUp();
29   *     // ... my code
30   *   }
31   * }
32   * </pre>
33   *
34   * Additional value is provided by {@link #isInitialSetup()} that may be helpful for specific implementations of
35   * {@link #doSetUp()} where you want to do some cleanup only once for the test-class rather than for every test method.
36   * Unlike {@link org.junit.BeforeClass} this can be used in non-static method code so you have access to injected
37   * dependencies.
38   *
39   * @author shuber, jmolinar
40   */
41  public abstract class BaseTest extends Assertions {
42    /**
43     * Indicates if the test class is to be set up for the first time. {@code true} indicates that the class has already
44     * been set up (e.g., database setup) for the execution of an preceding test method.
45     */
46    protected static boolean INITIALIZED = false;
47  
48    /**
49     * Suggests to use {@link #doSetUp()} method before each tests.
50     */
51    @Before
52    public final void setUp() {
53  
54      // Simply sets INITIALIZED to true when setUp is called for the first time.
55      doSetUp();
56      if (!INITIALIZED) {
57        INITIALIZED = true;
58      }
59    }
60  
61    /**
62     * Suggests to use {@link #doTearDown()} method before each tests.
63     */
64    @After
65    public final void tearDown() {
66  
67      doTearDown();
68    }
69  
70    /**
71     * @return {@code true} if this JUnit class is invoked for the first time (first test method is called), {@code false}
72     *         otherwise (if this is a subsequent invocation).
73     */
74    protected boolean isInitialSetup() {
75  
76      return INITIALIZED;
77    }
78  
79    /**
80     * Provides initialization previous to the creation of the text fixture.
81     */
82    protected void doSetUp() {
83  
84    }
85  
86    /**
87     * Provides clean up after tests.
88     */
89    protected void doTearDown() {
90  
91    }
92  }
