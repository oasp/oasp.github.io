28  /**
29   * This is an implementation of {@link AbstractUserDetailsAuthenticationProvider} based on
30   * {@link PrincipalAccessControlProvider} and {@link AccessControlProvider}.
31   *
32   * @param <U> is the generic type of the {@link UserDetails} implementation used to bridge with spring-security.
33   * @param <P> is the generic type of the {@link Principal} for internal user representation to bridge with
34   *        {@link PrincipalAccessControlProvider}.
35   *
36   * @author hohwille
37   */
38  public abstract class AbstractAccessControlBasedAuthenticationProvider<U extends UserDetails, P extends Principal>
39      extends AbstractUserDetailsAuthenticationProvider {
40  
41    /** The {@link Logger} instance. */
42    private static final Logger LOG = LoggerFactory.getLogger(AbstractAccessControlBasedAuthenticationProvider.class);
43  
44    private PrincipalAccessControlProvider<P> principalAccessControlProvider;
45  
46    private AccessControlProvider accessControlProvider;
47  
48    /**
49     * The constructor.
50     */
51    public AbstractAccessControlBasedAuthenticationProvider() {
52  
53    }
54  
55    /**
56     * @param principalAccessControlProvider the {@link PrincipalAccessControlProvider} to {@link Inject}.
57     */
58    @Inject
59    public void setPrincipalAccessControlProvider(PrincipalAccessControlProvider<P> principalAccessControlProvider) {
60  
61      this.principalAccessControlProvider = principalAccessControlProvider;
62    }
63  
64    /**
65     * @param accessControlProvider the {@link AccessControlProvider} to {@link Inject}.
66     */
67    @Inject
68    public void setAccessControlProvider(AccessControlProvider accessControlProvider) {
69  
70      this.accessControlProvider = accessControlProvider;
71    }
72  
73    /**
74     * Here the actual authentication has to be implemented.<br/>
75     * <br/>
76     *
77     * {@inheritDoc}
78     */
79    @Override
80    protected void additionalAuthenticationChecks(UserDetails userDetails,
81        UsernamePasswordAuthenticationToken authentication) throws AuthenticationException {
82  
83      // default implementation authentications via servlet API (container managed)
84      ServletRequestAttributes currentRequestAttributes =
85          (ServletRequestAttributes) RequestContextHolder.currentRequestAttributes();
86  
87      HttpServletRequest request = currentRequestAttributes.getRequest();
88      String login = authentication.getName();
89      String password = null;
90      Object credentials = authentication.getCredentials();
91      if (credentials != null) {
92        password = credentials.toString();
93      }
94      try {
95        request.login(login, password);
96      } catch (ServletException e) {
97        LOG.warn("Authentication failed: {}", e.toString());
98        throw new BadCredentialsException("Authentication failed.", e);
99      }
100     authentication.setDetails(userDetails);
101   }
102 
103   /**
104    * Creates an instance of {@link UserDetails} that represent the user with the given <code>username</code>.
105    *
106    * @param username is the login of the user to create.
107    * @param password the password of the user.
108    * @param principal is the internal {@link Principal} that has been provided by
109    *        {@link #retrievePrincipal(String, UsernamePasswordAuthenticationToken)}.
110    * @param authorities are the {@link GrantedAuthority granted authorities} or in other words the permissions of the
111    *        user.
112    * @return the new user object.
113    */
114   protected abstract U createUser(String username, String password, P principal, Set<GrantedAuthority> authorities);
115 
116   /**
117    * Retrieves the internal {@link Principal} object representing the user. This can be any object implementing
118    * {@link Principal} and can contain additional user details such as profile data. This object is used to
119    * {@link PrincipalAccessControlProvider#getAccessControlIds(Principal) retrieve} the (top-level)
120    * {@link AccessControl}s that have been granted to the user.
121    *
122    * @param username is the login of the user.
123    * @param authentication is the {@link UsernamePasswordAuthenticationToken}.
124    * @return the {@link Principal}.
125    */
126   protected abstract P retrievePrincipal(String username, UsernamePasswordAuthenticationToken authentication);
127 
128   /**
129    * {@inheritDoc}
130    */
131   @Override
132   protected UserDetails retrieveUser(String username, UsernamePasswordAuthenticationToken authentication)
133       throws AuthenticationException {
134 
135     P principal = retrievePrincipal(username, authentication);
136     if (principal == null) {
137       LOG.warn("Failed to retrieve user for login {}.", username);
138       throw new UsernameNotFoundException(username);
139     }
140 
141     // determine granted authorities for spring-security...
142     Set<GrantedAuthority> authorities = new HashSet<>();
143     Collection<String> accessControlIds = this.principalAccessControlProvider.getAccessControlIds(principal);
144     Set<AccessControl> accessControlSet = new HashSet<>();
145     for (String id : accessControlIds) {
146       boolean success = this.accessControlProvider.collectAccessControls(id, accessControlSet);
147       if (!success) {
148         LOG.warn("Undefined access control {}.", id);
149         // authorities.add(new SimpleGrantedAuthority(id));
150       }
151     }
152     for (AccessControl accessControl : accessControlSet) {
153       authorities.add(new AccessControlGrantedAuthority(accessControl));
154     }
155 
156     String password = null;
157     Object credentials = authentication.getCredentials();
158     if (credentials != null) {
159       password = credentials.toString();
160     }
161     return createUser(username, password, principal, authorities);
162   }
