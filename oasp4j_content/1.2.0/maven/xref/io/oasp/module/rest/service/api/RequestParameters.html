17  /**
18   * This class helps to deal with {@link UriInfo} and {@link MultivaluedMap} from the JAX-RS API. E.g. if you have a REST
19   * query operation for a collection URI you can use {@link UriInfo} in case you want to support a mixture of optional
20   * and required parameters. The methods provided here throw according exceptions such as {@link BadRequestException} and
21   * already support conversion of values.
22   *
23   * @author hohwille
24   */
25  public class RequestParameters {
26  
27    private final MultivaluedMap<String, String> parameters;
28  
29    /**
30     * The constructor.
31     *
32     * @param parameters is the {@link MultivaluedMap} containing the parameters to wrap.
33     */
34    public RequestParameters(MultivaluedMap<String, String> parameters) {
35  
36      super();
37      this.parameters = parameters;
38    }
39  
40    /**
41     * Gets the single parameter in a generic and flexible way.
42     *
43     * @param <T> is the generic type of <code>targetType</code>.
44     * @param key is the {@link java.util.Map#get(Object) key} of the parameter to get.
45     * @param targetType is the {@link Class} reflecting the type to convert the value to. Supports common Java standard
46     *        types such as {@link String}, {@link Long}, {@link Double}, {@link BigDecimal}, etc.
47     * @param required - <code>true</code> if the value is required and a {@link BadRequestException} is thrown if it is
48     *        not present, <code>false</code> otherwise (if optional).
49     * @return the value for the given <code>key</code> converted to the given <code>targetType</code>. May be
50     *         <code>null</code> if <code>required</code> is <code>false</code> .
51     * @throws WebApplicationException if an error occurred. E.g. {@link BadRequestException} if a required parameter is
52     *         missing or {@link InternalServerErrorException} if the given <code>targetType</code> is not supported.
53     */
54    @SuppressWarnings("unchecked")
55    public <T> T get(String key, Class<T> targetType, boolean required) throws WebApplicationException {
56  
57      String value = get(key);
58      if (value == null) {
59        if (required) {
60          throw new BadRequestException("Missing parameter: " + key);
61        }
62        Object result = null;
63        if (targetType.isPrimitive()) {
64          if (targetType == boolean.class) {
65            result = Boolean.FALSE;
66          } else if (targetType == int.class) {
67            result = Integer.valueOf(0);
68          } else if (targetType == long.class) {
69            result = Long.valueOf(0);
70          } else if (targetType == double.class) {
71            result = Double.valueOf(0);
72          } else if (targetType == float.class) {
73            result = Float.valueOf(0);
74          } else if (targetType == byte.class) {
75            result = Byte.valueOf((byte) 0);
76          } else if (targetType == short.class) {
77            result = Short.valueOf((short) 0);
78          } else if (targetType == char.class) {
79            result = '\0';
80          }
81        }
82        return (T) result;
83      }
84      try {
85        return convertValue(value, targetType);
86      } catch (WebApplicationException e) {
87        throw e;
88      } catch (Exception e) {
89        throw new BadRequestException("Failed to convert '" + value + "' to type " + targetType);
90      }
91    }
92  
93    /**
94     * Converts the given <code>value</code> to the given <code>targetType</code>.
95     *
96     * @param <T> is the generic type of <code>targetType</code>.
97     * @param value is the value to convert.
98     * @param targetType is the {@link Class} reflecting the type to convert the value to.
99     * @return the converted value.
100    * @throws ParseException if parsing of the given <code>value</code> failed while converting.
101    */
102   @SuppressWarnings("unchecked")
103   protected <T> T convertValue(String value, Class<T> targetType) throws ParseException {
104 
105     if (value == null) {
106       return null;
107     }
108     Object result;
109     if (targetType == String.class) {
110       result = value;
111     } else if (targetType.isEnum()) {
112       for (T instance : targetType.getEnumConstants()) {
113         Enum<?> e = (Enum<?>) instance;
114         if (e.name().equalsIgnoreCase(value)) {
115           return instance;
116         }
117       }
118       throw new IllegalArgumentException("Enum constant not found!");
119     } else if ((targetType == boolean.class) || (targetType == Boolean.class)) {
120       result = Boolean.parseBoolean(value);
121     } else if ((targetType == int.class) || (targetType == Integer.class)) {
122       result = Integer.valueOf(value);
123     } else if ((targetType == long.class) || (targetType == Long.class)) {
124       result = Long.valueOf(value);
125     } else if ((targetType == double.class) || (targetType == Double.class)) {
126       result = Double.valueOf(value);
127     } else if ((targetType == float.class) || (targetType == Float.class)) {
128       result = Float.valueOf(value);
129     } else if ((targetType == short.class) || (targetType == Short.class)) {
130       result = Short.valueOf(value);
131     } else if ((targetType == byte.class) || (targetType == Byte.class)) {
132       result = Byte.valueOf(value);
133     } else if (targetType == BigDecimal.class) {
134       result = new BigDecimal(value);
135     } else if (targetType == BigInteger.class) {
136       result = new BigInteger(value);
137     } else if (targetType == Date.class) {
138       result = new SimpleDateFormat("YYYY-MM-dd'T'HH:mm:ss").parseObject(value);
139     } else {
140       throw new InternalServerErrorException("Unsupported type " + targetType);
141     }
142     // do not use type.cast() as not working for primitive types.
143     return (T) result;
144   }

146   /**
147    * Gets the parameter as single value with the given <code>key</code> as {@link String}.
148    *
149    * @param key is the {@link java.util.Map#get(Object) key} of the parameter to get.
150    * @return the requested parameter. Will be <code>null</code> if the parameter is not present.
151    * @throws BadRequestException if the parameter is defined multiple times (see {@link #getList(String)}).
152    */
153   public String get(String key) throws BadRequestException {
154 
155     List<String> list = this.parameters.get(key);
156     if ((list == null) || (list.isEmpty())) {
157       return null;
158     }
159     if (list.size() > 1) {
160       throw new BadRequestException("Duplicate parameter: " + key);
161     }
162     return list.get(0);
163   }
164 
165   /**
166    * Gets the parameter with the given <code>key</code> as {@link String}. Unlike {@link #get(String)} this method will
167    * not throw an exception if the parameter is multi-valued but just return the first value.
168    *
169    * @param key is the {@link java.util.Map#get(Object) key} of the parameter to get.
170    * @return the first value of the requested parameter. Will be <code>null</code> if the parameter is not present.
171    */
172   public String getFirst(String key) {
173 
174     return this.parameters.getFirst(key);
175   }
176 
177   /**
178    * Gets the {@link List} of all value for the parameter with with the given <code>key</code>. In general you should
179    * avoid multi-valued parameters (e.g. http://host/path?query=a&query=b). The JAX-RS API supports this exotic case as
180    * first citizen so we expose it here but only use it if you know exactly what you are doing.
181    *
182    * @param key is the {@link java.util.Map#get(Object) key} of the parameter to get.
183    * @return the {@link List} with all values of the requested parameter. Will be an {@link Collections#emptyList()
184    *         empty list} if the parameter is not present.
185    */
186   public List<String> getList(String key) {
187 
188     List<String> list = this.parameters.get(key);
189     if (list == null) {
190       list = Collections.emptyList();
191     }
192     return list;
193   }

195   /**
196    * @param uriInfo is the {@link UriInfo}.
197    * @return a new instance of {@link RequestParameters} for {@link UriInfo#getQueryParameters()}.
198    */
199   public static RequestParameters fromQuery(UriInfo uriInfo) {
200 
201     return new RequestParameters(uriInfo.getQueryParameters());
202   }
