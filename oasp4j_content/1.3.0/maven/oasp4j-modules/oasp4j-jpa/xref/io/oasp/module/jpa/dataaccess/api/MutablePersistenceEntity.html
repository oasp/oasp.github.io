3   package io.oasp.module.jpa.dataaccess.api;
4   
5   import net.sf.mmm.util.entity.api.MutableGenericEntity;
6   import net.sf.mmm.util.entity.api.PersistenceEntity;
7   
8   /**
9    * This is the interface for a {@link PersistenceEntity} in OASP.
10   *
11   * @param <ID> is the type of the {@link #getId() primary key}.
12   *
13   * @see AbstractPersistenceEntity
14   *
15   * @author hohwille
16   */
17  public interface MutablePersistenceEntity<ID> extends PersistenceEntity<ID>, MutableGenericEntity<ID> {
18  
19    /**
20     * This method sets the {@link #getRevision() revision} of this entity. <br>
21     * <b>ATTENTION:</b><br>
22     * This operation should only be used in specific cases and if you are aware of what you are doing as this attribute
23     * is managed by the persistence. However, for final freedom we decided to add this method to the API (e.g. to copy
24     * from transfer-object to persistent-entity and vice-versa).
25     *
26     * @param revision is the new value of {@link #getRevision()}.
27     */
28    void setRevision(Number revision);
29  
30  }
