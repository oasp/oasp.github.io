31  /**
32   * This is the abstract base-implementation of the {@link GenericDao} interface.
33   *
34   * @param <ID> is the generic type if the {@link PersistenceEntity#getId() primary key}.
35   * @param <E> is the generic type of the managed {@link PersistenceEntity}.
36   *
37   */
38  // @Repository
39  public abstract class AbstractGenericDao<ID, E extends PersistenceEntity<ID>> implements GenericDao<ID, E> {
40  
41    /** Logger instance. */
42    private static final Logger LOG = LoggerFactory.getLogger(AbstractGenericDao.class);
43  
44    private EntityManager entityManager;
45  
46    /**
47     * The constructor.
48     */
49    public AbstractGenericDao() {
50  
51      super();
52    }

54    /**
55     * @return the {@link Class} reflecting the managed entity.
56     */
57    protected abstract Class<E> getEntityClass();
58  
59    /**
60     * @return the {@link EntityManager} instance.
61     */
62    protected EntityManager getEntityManager() {
63  
64      return this.entityManager;
65    }
66  
67    /**
68     * @param entityManager the {@link EntityManager} to inject.
69     */
70    @PersistenceContext
71    public void setEntityManager(EntityManager entityManager) {
72  
73      this.entityManager = entityManager;
74    }
75  
76    /**
77     * @return the name of the managed entity.
78     */
79    protected String getEntityName() {
80  
81      return getEntityClass().getSimpleName();
82    }
83  
84    @Override
85    public E save(E entity) {
86  
87      if (isNew(entity)) {
88        this.entityManager.persist(entity);
89        LOG.debug("Saved new {} with id {}.", getEntityName(), entity.getId());
90        return entity;
91      } else {
92        if (this.entityManager.find(entity.getClass(), entity.getId()) != null) {
93          E update = this.entityManager.merge(entity);
94          LOG.debug("Updated {} with id {}.", getEntityName(), entity.getId());
95          return update;
96        } else {
97          throw new EntityNotFoundException("Entity not found");
98        }
99      }
100   }
101 
102   /**
103    * Determines if the given {@link PersistenceEntity} is {@link PersistenceEntity#STATE_NEW new}.
104    *
105    * @param entity is the {@link PersistenceEntity} to check.
106    * @return {@code true} if {@link PersistenceEntity#STATE_NEW new}, {@code false} otherwise (e.g.
107    *         {@link PersistenceEntity#STATE_DETACHED detached} or {@link PersistenceEntity#STATE_MANAGED managed}.
108    */
109   protected boolean isNew(E entity) {
110 
111     return entity.getId() == null;
112   }
113 
114   @Override
115   public void save(Iterable<? extends E> entities) {
116 
117     for (E entity : entities) {
118       save(entity);
119     }
120   }

122   @Override
123   public void forceIncrementModificationCounter(E entity) {
124 
125     getEntityManager().lock(entity, LockModeType.OPTIMISTIC_FORCE_INCREMENT);
126   }

128   @Override
129   public E findOne(ID id) {
130 
131     E entity = this.entityManager.find(getEntityClass(), id);
132     return entity;
133   }
134 
135   @Override
136   public E find(ID id) throws ObjectNotFoundUserException {
137 
138     E entity = findOne(id);
139     if (entity == null) {
140       throw new ObjectNotFoundUserException(getEntityClass().getSimpleName(), id);
141     }
142     return entity;
143   }
144 
145   @Override
146   public boolean exists(ID id) {
147 
148     // pointless...
149     return findOne(id) != null;
150   }

152   /**
153    * @return an {@link Iterable} to find ALL {@link #getEntityClass() managed entities} from the persistent store. Not
154    *         exposed to API by default as this might not make sense for all kind of entities.
155    */
156   protected List<E> findAll() {
157 
158     CriteriaQuery<E> query = this.entityManager.getCriteriaBuilder().createQuery(getEntityClass());
159     Root<E> root = query.from(getEntityClass());
160     query.select(root);
161     TypedQuery<E> typedQuery = this.entityManager.createQuery(query);
162     List<E> resultList = typedQuery.getResultList();
163     LOG.debug("Query for all {} objects returned {} hit(s).", getEntityName(), resultList.size());
164     return resultList;
165   }
166 
167   @Override
168   public List<E> findAll(Iterable<ID> ids) {
169 
170     CriteriaBuilder builder = this.entityManager.getCriteriaBuilder();
171     CriteriaQuery<E> query = builder.createQuery(getEntityClass());
172     Root<E> root = query.from(getEntityClass());
173     query.select(root);
174     query.where(root.get("id").in(ids));
175     TypedQuery<E> typedQuery = this.entityManager.createQuery(query);
176     List<E> resultList = typedQuery.getResultList();
177     LOG.debug("Query for selection of {} objects returned {} hit(s).", getEntityName(), resultList.size());
178     return resultList;
179   }
180 
181   @Override
182   public void delete(ID id) {
183 
184     E entity = this.entityManager.getReference(getEntityClass(), id);
185     this.entityManager.remove(entity);
186     LOG.debug("Deleted {} with ID {}.", getEntityName(), id);
187   }
188 
189   @Override
190   public void delete(E entity) {
191 
192     // entity might be detached and could cause trouble in entityManager on remove
193     if (this.entityManager.contains(entity)) {
194       this.entityManager.remove(entity);
195       LOG.debug("Deleted {} with ID {}.", getEntityName(), entity.getId());
196     } else {
197       delete(entity.getId());
198     }
199 
200   }
201 
202   @Override
203   public void delete(Iterable<? extends E> entities) {
204 
205     for (E entity : entities) {
206       delete(entity);
207     }
208   }

216   /**
217    * Returns a paginated list of entities according to the supplied {@link SearchCriteriaTo criteria}.
218    * <p>
219    * Applies {@code limit} and {@code offset} values to the supplied {@code query} according to the supplied
220    * {@link PaginationTo pagination} information inside {@code criteria}.
221    * <p>
222    * If a {@link PaginationTo#isTotal() total count} of available entities is requested, will also execute a second
223    * query, without pagination parameters applied, to obtain said count.
224    * <p>
225    * Will install a query timeout if {@link SearchCriteriaTo#getSearchTimeout()} is not null.
226    *
227    * @param criteria contains information about the requested page.
228    * @param query is a query which is preconfigured with the desired conditions for the search.
229    * @param expr is used for the final mapping from the SQL result to the entities.
230    * @return a paginated list.
231    */
232   protected PaginatedListTo<E> findPaginated(SearchCriteriaTo criteria, JPAQuery query, Expression<E> expr) {
233 
234     applyCriteria(criteria, query);
235 
236     PaginationTo pagination = criteria.getPagination();
237 
238     PaginationResultTo paginationResult = createPaginationResult(pagination, query);
239 
240     applyPagination(pagination, query);
241     List<E> paginatedList = query.list(expr);
242 
243     return new PaginatedListTo<>(paginatedList, paginationResult);
244   }
245 
246   /**
247    * Creates a {@link PaginationResultTo pagination result} for the given {@code pagination} and {@code query}.
248    * <p>
249    * Needs to be called before pagination is applied to the {@code query}.
250    *
251    * @param pagination contains information about the requested page.
252    * @param query is a query preconfigured with the desired conditions for the search.
253    * @return information about the applied pagination.
254    */
255   protected PaginationResultTo createPaginationResult(PaginationTo pagination, JPAQuery query) {
256 
257     Long total = calculateTotalBeforePagination(pagination, query);
258 
259     return new PaginationResultTo(pagination, total);
260   }
261 
262   /**
263    * Calculates the total number of entities the given {@link JPAQuery query} would return without pagination applied.
264    * <p>
265    * Needs to be called before pagination is applied to the {@code query}.
266    *
267    * @param pagination is the pagination information as requested by the client.
268    * @param query is the {@link JPAQuery query} for which to calculate the total.
269    * @return the total count, or {@literal null} if {@link PaginationTo#isTotal()} is {@literal false}.
270    */
271   protected Long calculateTotalBeforePagination(PaginationTo pagination, JPAQuery query) {
272 
273     Long total = null;
274     if (pagination.isTotal()) {
275       total = query.clone().count();
276     }
277 
278     return total;
279   }

281   /**
282    * Applies the {@link PaginationTo pagination criteria} to the given {@link JPAQuery}.
283    *
284    * @param pagination is the {@link PaginationTo pagination criteria} to apply.
285    * @param query is the {@link JPAQuery} to apply to.
286    */
287   protected void applyPagination(PaginationTo pagination, JPAQuery query) {
288 
289     if (pagination == PaginationTo.NO_PAGINATION) {
290       return;
291     }
292 
293     Integer limit = pagination.getSize();
294     if (limit != null) {
295       query.limit(limit);
296 
297       int page = pagination.getPage();
298       if (page > 0) {
299         query.offset((page - 1) * limit);
300       }
301     }
302   }

304   /**
305    * Applies the meta-data of the given {@link AbstractSearchCriteria search criteria} to the given {@link JPAQuery}.
306    *
307    * @param criteria is the {@link AbstractSearchCriteria search criteria} to apply.
308    * @param query is the {@link JPAQuery} to apply to.
309    */
310   protected void applyCriteria(AbstractSearchCriteria criteria, JPAQuery query) {
311 
312     Integer limit = criteria.getMaximumHitCount();
313     if (limit != null) {
314       query.limit(limit);
315     }
316     int offset = criteria.getHitOffset();
317     if (offset > 0) {
318       query.offset(offset);
319     }
320     Long timeout = criteria.getSearchTimeout();
321     if (timeout != null) {
322       query.setHint("javax.persistence.query.timeout", timeout.intValue());
323     }
324   }
325 
326   /**
327    * Applies the meta-data of the given {@link AbstractSearchCriteria search criteria} to the given {@link Query}.
328    *
329    * @param criteria is the {@link AbstractSearchCriteria search criteria} to apply.
330    * @param query is the {@link Query} to apply to.
331    */
332   protected void applyCriteria(AbstractSearchCriteria criteria, Query query) {
333 
334     Integer limit = criteria.getMaximumHitCount();
335     if (limit != null) {
336       query.setMaxResults(limit);
337     }
338     int offset = criteria.getHitOffset();
339     if (offset > 0) {
340       query.setFirstResult(offset);
341     }
342     Long timeout = criteria.getSearchTimeout();
343     if (timeout != null) {
344       query.setHint("javax.persistence.query.timeout", timeout.intValue());
345     }
346   }
