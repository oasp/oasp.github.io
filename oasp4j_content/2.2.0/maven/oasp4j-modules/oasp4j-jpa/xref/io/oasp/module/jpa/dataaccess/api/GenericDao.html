5   import net.sf.mmm.util.entity.api.PersistenceEntity;
6   import net.sf.mmm.util.exception.api.ObjectNotFoundUserException;

8   /**
9    * This is the interface for a <em>Data Access Object</em> (DAO). It acts as a manager responsible for the persistence
10   * operations on a specific {@link PersistenceEntity entity} {@literal <E>}.<br/>
11   * This is base interface contains the CRUD operations:
12   * <ul>
13   * <li>Create: call {@link #save(PersistenceEntity)} on a new entity.</li>
14   * <li>Retrieve: use <code>find*</code> methods such as {@link #findOne(Object)}. More specific queries will be added in
15   * dedicated DAO interfaces.</li>
16   * <li>Update: done automatically by JPA vendor (hibernate) on commit or call {@link #save(PersistenceEntity)} to
17   * {@link javax.persistence.EntityManager#merge(Object) merge} an entity.</li>
18   * <li>Delete: call {@link #delete(PersistenceEntity)} or {@link #delete(Object)}.</li>
19   * </ul>
20   * For each (non-abstract) implementation of {@link PersistenceEntity entity} <code>MyEntity</code> you should create an
21   * interface interface <code>MyEntityDao</code> that inherits from this {@link GenericDao} interface. Also you create an
22   * implementation of that interface <code>MyEntityDaoImpl</code> that you derive from
23   * {@link io.oasp.module.jpa.dataaccess.base.AbstractGenericDao}.
24   *
25   * @param <ID> is the generic type if the {@link PersistenceEntity#getId() primary key}.
26   * @param <E> is the generic type of the {@link PersistenceEntity}.
27   *
28   */
29  public interface GenericDao<ID, E extends PersistenceEntity<ID>> {
30  
31    /**
32     * Saves a given entity. Use the returned instance for further operations as the save operation might have changed the
33     * entity instance completely.
34     *
35     * @param entity the {@link PersistenceEntity entity} to save
36     * @return the saved entity
37     */
38    E save(E entity);
39  
40    /**
41     * Saves all given entities.
42     *
43     * @param entities the {@link PersistenceEntity entities} to save
44     */
45    void save(Iterable<? extends E> entities);
46  
47    /**
48     * Retrieves an entity by its id.
49     *
50     * @param id must not be {@literal null}.
51     * @return the entity with the given id or {@literal null} if none found
52     * @throws ObjectNotFoundUserException if the requested entity does not exists (use {@link #findOne(Object)} to
53     *         prevent).
54     */
55    E find(ID id) throws ObjectNotFoundUserException;
56  
57    /**
58     * Retrieves an entity by its id.
59     *
60     * @param id must not be {@literal null}.
61     * @return the entity with the given id or {@literal null} if none found
62     * @throws IllegalArgumentException if {@code id} is {@code null}
63     */
64    E findOne(ID id) throws IllegalArgumentException;
65  
66    /**
67     * Returns whether an entity with the given id exists.
68     *
69     * @param id must not be {@literal null}.
70     * @return true if an entity with the given id exists, {@literal false} otherwise
71     */
72    boolean exists(ID id);
73  
74    /**
75     * Returns all instances of the type with the given IDs.
76     *
77     * @param ids are the IDs of all entities to retrieve e.g. as {@link java.util.List}.
78     * @return an {@link Iterable} with all {@link PersistenceEntity entites} for the given <code>ids</code>.
79     */
80    List<E> findAll(Iterable<ID> ids);
81  
82    /**
83     * Deletes the entity with the given id.
84     *
85     * @param id must not be {@literal null}.
86     * @throws IllegalArgumentException in case the given {@code id} is {@code null}
87     */
88    void delete(ID id) throws IllegalArgumentException;
89  
90    /**
91     * Deletes a given entity.
92     *
93     * @param entity the {@link PersistenceEntity entity} to delete
94     */
95    void delete(E entity);
96  
97    /**
98     * Deletes the given entities.
99     *
100    * @param entities the {@link PersistenceEntity entities} to delete
101    */
102   void delete(Iterable<? extends E> entities);
103 
104   /**
105    * Enforces to increment the {@link PersistenceEntity#getModificationCounter() modificationCounter} e.g. to enforce
106    * that a parent object gets locked when its children are modified.<br>
107    * As an example we assume that we have the two optimistic locked entities {@code Order} and its contained
108    * {@code OrderPosition}. By default the users can modify an {@code Order} and each of its {@code OrderPosition}s in
109    * parallel without getting a locking conflict. This can be desired. However, it can also be a demand that an
110    * {@code Order} gets approved and the user doing that is responsible for the total price as the sum of the prices of
111    * each {@code OrderPosition}. Now if another user is adding or changing an {@code OrderPostion} belonging to that
112    * {@code Order} in parallel the {@code Order} will get approved but the approved total price will differ from what
113    * the user has actually seen when he clicked on approve. To prevent this the use-case to modify an
114    * {@code OrderPosition} can use this method to trigger a locking on the associated {@code Order}. The implication is
115    * also that two users changing an {@code OrderPosition} associated with the same {@code Order} in parallel will get a
116    * conflict.
117    *
118    * @param entity that is getting checked.
119    */
120   void forceIncrementModificationCounter(E entity);
121 
122 }
