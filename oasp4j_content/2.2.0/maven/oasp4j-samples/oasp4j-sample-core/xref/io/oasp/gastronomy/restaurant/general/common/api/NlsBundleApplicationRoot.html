30     * @param newState is the new state for the entity that is illegal.

33    @NlsBundleMessage("The entity {entity} in state {currentState} can not be changed to state {newState}!")

37    /**
38     * @see io.oasp.gastronomy.restaurant.general.common.api.exception.IllegalEntityStateException
39     *
40     * @param object is the entity relevant for the error.

43     */
44    @NlsBundleMessage("The property {property} of object {object} can not be changed!")
45    NlsMessage errorIllegalPropertyChange(@Named("object") Object object, @Named("property") Object property);
