3   package io.oasp.module.jpa.dataaccess.api;
4   
5   import java.util.List;
6   
7   import javax.persistence.PersistenceException;
8   
9   import net.sf.mmm.util.entity.api.RevisionedEntity;
10  import net.sf.mmm.util.exception.api.ObjectNotFoundException;
11  
12  /**
13   * This is the interface for a {@link GenericDao} with the ability of revision-control. It organizes a revision-history
14   * (journal) of the managed entities.
15   *
16   * @see RevisionedEntity
17   *
18   * @param <ID> is the type of the {@link RevisionedEntity#getId() primary key}.
19   * @param <ENTITY> is the type of the managed entity.
20   *
21   */
22  public interface GenericRevisionedDao<ID, ENTITY extends MutablePersistenceEntity<ID>> extends GenericDao<ID, ENTITY> {
23  
24    /**
25     * This method will get the {@link List} of historic {@link MutablePersistenceEntity#getRevision() revisions} of the
26     * {@link MutablePersistenceEntity entity} with the given <code>id</code>.<br>
27     * If the <code>entity</code> is NOT revision controlled, an {@link java.util.Collections#emptyList() empty list} is
28     * returned.
29     *
30     * @param id the {@link MutablePersistenceEntity#getId() primary key} of the {@link MutablePersistenceEntity entity}
31     *        to retrieve the history for.
32     * @return the {@link List} of historic {@link RevisionedEntity#getRevision() revisions}.
33     */
34    List<Number> getRevisionHistory(ID id);
35  
36    /**
37     * This method will get the {@link List} of {@link RevisionMetadata} from the {@link RevisionedEntity#getRevision()
38     * revision}-history of the entity with the given <code>id</code>.
39     *
40     * @param id is the {@link RevisionedEntity#getId() primary key} of the entity for which the history-metadata is
41     *        requested.
42     * @return the requested {@link List} of {@link RevisionMetadata}.
43     */
44    List<RevisionMetadata> getRevisionHistoryMetadata(Object id);
45  
46    /**
47     * This method loads a historic {@link RevisionedEntity#getRevision() revision} of the {@link RevisionedEntity} with
48     * the given <code>id</code> from the persistent store. <br>
49     * However if the given <code>revision</code> is {@link RevisionedEntity#LATEST_REVISION} the {@link #find(Object)
50     * latest revision will be loaded}. <br>
51     * <b>ATTENTION:</b><br>
52     * You should not make assumptions about the <code>revision</code> numbering of the underlying implementation. Please
53     * use {@link #getRevisionHistory(Object)} or {@link #getRevisionHistoryMetadata(Object)} to find revision numbers.
54     *
55     * @param id is the {@link RevisionedEntity#getId() primary key} of the requested {@link RevisionedEntity entity}.
56     * @param revision is the {@link RevisionedEntity#getRevision() revision} of the requested entity or
57     *        {@link RevisionedEntity#LATEST_REVISION} to get the {@link #find(Object) latest} revision. A specific
58     *        revision has to be greater than <code>0</code>.
59     * @return the requested {@link RevisionedEntity entity}.
60     * @throws ObjectNotFoundException if the requested {@link RevisionedEntity entity} could NOT be found.
61     */
62    ENTITY load(ID id, Number revision) throws ObjectNotFoundException;
63  
64    /**
65     * {@inheritDoc}
66     *
67     * The behavior of this method depends on the revision-control strategy of the implementation. <br>
68     * <ul>
69     * <li>In case of an <em>audit-proof revision-history</em> the deletion of the
70     * {@link RevisionedEntity#LATEST_REVISION latest revision} of an entity will only move it to the history while the
71     * deletion of a {@link RevisionedEntity#getRevision() historic entity} is NOT permitted and will cause a
72     * {@link PersistenceException}.</li>
73     * <li>In case of an <em>on-demand revision-history</em> the deletion of the {@link RevisionedEntity#LATEST_REVISION
74     * latest revision} of an entity will either move it to the history or</li>
75     * </ul>
76     * If the given <code>entity</code> is a {@link RevisionedEntity#getRevision() historic entity} the according historic
77     */
78    @Override
79    void delete(ENTITY entity);
80  
81  }
