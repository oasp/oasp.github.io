10  import org.junit.Test;

12  import org.springframework.test.context.TestExecutionListeners;

15  /**
16   * Test class to test the {@link GenericDao}.
17   *
18   */
19  @TestExecutionListeners({ DependencyInjectionTestExecutionListener.class })
20  @ContextConfiguration({ "classpath:config/app/dataaccess/beans-dataaccess.xml" })
21  public class AbstractGenericDaoTest extends ComponentTest {
22  
23    @Inject
24    private GenericDaoForceIncrementModificationTestBean testBean;
25  
26    /**
27     * Test of {@link GenericDao#forceIncrementModificationCounter(net.sf.mmm.util.entity.api.PersistenceEntity)}. Ensures
28     * that the modification counter is updated after the call of that method when the transaction is closed.
29     */
30    @Test
31    public void testForceIncrementModificationCounter() {
32  
33      // given
34      TestEntity entity = this.testBean.create();
35      assertThat(entity.getId()).isNotNull();
36      assertThat(entity.getModificationCounter()).isEqualTo(0);
37  
38      // when
39      TestEntity updatedEntity = this.testBean.incrementModificationCounter(entity.getId());
40  
41      // then
42      assertThat(updatedEntity.getModificationCounter()).isEqualTo(1);
43    }
44  
45    /**
46     * This type provides methods in a transactional environment for the containing test class. All methods, annotated
47     * with the {@link Transactional} annotation, are executed in separate transaction, thus one test case can execute
48     * multiple transactions. Unfortunately this does not work when the transactional methods are directly in the
49     * top-level class of the test-case itself.
50     */
51    @Named
52    public static class GenericDaoForceIncrementModificationTestBean {
53  
54      @Inject
55      private TestDao genericDao;
56  
57      /**
58       * Creates a new {@link TestEntity}, persist it and surround everything with a transaction.
59       *
60       * @return entity the new {@link TestEntity}.
61       */
62      @Transactional
63      public TestEntity create() {
64  
65        TestEntity entity = new TestEntity();
66        this.genericDao.save(entity);
67        return entity;
68      }
69  
70      /**
71       * Loads the {@link TestEntity} with the given {@code id} and
72       * {@link GenericDao#forceIncrementModificationCounter(net.sf.mmm.util.entity.api.PersistenceEntity) increments the
73       * modification counter}.
74       *
75       * @param id of the {@link TestEntity} to load and increment.
76       * @return entity the updated {@link TestEntity}.
77       */
78      @Transactional
79      public TestEntity incrementModificationCounter(long id) {
80  
81        TestEntity entity = this.genericDao.find(id);
82        this.genericDao.forceIncrementModificationCounter(entity);
83        return entity;
84      }
85    }
86  
87  };
